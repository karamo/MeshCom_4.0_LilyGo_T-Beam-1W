32#define SFN_OPEN_USES_CHKSUM 0
33bool FatFile::open(FatFile* dirFile, FatSfn_t* fname, oflag_t oflag) {
34 uint16_t date;
35 uint16_t time;
36 uint8_t ms10;
37 bool emptyFound = false;
38#if SFN_OPEN_USES_CHKSUM
39 uint8_t checksum;
40#endif // SFN_OPEN_USES_CHKSUM
41 uint8_t lfnOrd = 0;
42 uint16_t emptyIndex = 0;
43 uint16_t index = 0;
44 DirFat_t* dir;
45 DirLfn_t* ldir;
46
47 dirFile->rewind();
48 while (true) {
49 dir = dirFile->readDirCache(true);
50 if (!dir) {
51 if (dirFile->getError()) {
52 DBG_FAIL_MACRO;
53 goto fail;
54 }
55 // At EOF if no error.
56 break;
57 }
58 if (dir->name[0] == FAT_NAME_DELETED || dir->name[0] == FAT_NAME_FREE) {
59 if (!emptyFound) {
60 emptyIndex = index;
61 emptyFound = true;
62 }
63 if (dir->name[0] == FAT_NAME_FREE) {
64 break;
65 }
66 lfnOrd = 0;
67 } else if (isFatFileOrSubdir(dir)) {
68 if (!memcmp(fname->sfn, dir->name, 11)) {
69 // don't open existing file if O_EXCL
70 if (oflag & O_EXCL) {
71 DBG_FAIL_MACRO;
72 goto fail;
73 }
74#if SFN_OPEN_USES_CHKSUM
75 if (lfnOrd && checksum != lfnChecksum(dir->name)) {
76 DBG_FAIL_MACRO;
77 goto fail;
78 }
79#endif // SFN_OPEN_USES_CHKSUM
80 if (!openCachedEntry(dirFile, index, oflag, lfnOrd)) {
81 DBG_FAIL_MACRO;
82 goto fail;
83 }
84 return true;
85 } else {
86 lfnOrd = 0;
87 }
88 } else if (isFatLongName(dir)) {
89 ldir = reinterpret_cast<DirLfn_t*>(dir);
90 if (ldir->order & FAT_ORDER_LAST_LONG_ENTRY) {
91 lfnOrd = ldir->order & 0X1F;
92#if SFN_OPEN_USES_CHKSUM
93 checksum = ldir->checksum;
94#endif // SFN_OPEN_USES_CHKSUM
95 }
96 } else {
97 lfnOrd = 0;
98 }
99 index++;
100 }
101 // don't create unless O_CREAT and write mode
102 if (!(oflag & O_CREAT) || !isWriteMode(oflag)) {
103 DBG_FAIL_MACRO;
104 goto fail;
105 }
106 if (emptyFound) {
107 index = emptyIndex;
108 } else {
109 if (!dirFile->addDirCluster()) {
110 DBG_FAIL_MACRO;
111 goto fail;
112 }
113 }
114 dir = reinterpret_cast<DirFat_t*>(dirFile->cacheDir(index));
115 if (!dir) {
116 DBG_FAIL_MACRO;
117 goto fail;
118 }
119 // initialize as empty file
120 memset(dir, 0, sizeof(DirFat_t));
121 memcpy(dir->name, fname->sfn, 11);
122
123 // Set base-name and extension lower case bits.
124 dir->caseFlags = (FAT_CASE_LC_BASE | FAT_CASE_LC_EXT) & fname->flags;
125
126 // Set timestamps.
127 if (FsDateTime::callback) {
128 // call user date/time function
129 FsDateTime::callback(&date, &time, &ms10);
130 setLe16(dir->createDate, date);
131 setLe16(dir->createTime, time);
132 dir->createTimeMs = ms10;
133 } else {
134 setLe16(dir->createDate, FS_DEFAULT_DATE);
135 setLe16(dir->modifyDate, FS_DEFAULT_DATE);
136 setLe16(dir->accessDate, FS_DEFAULT_DATE);
137 if (FS_DEFAULT_TIME) {
138 setLe16(dir->createTime, FS_DEFAULT_TIME);
139 setLe16(dir->modifyTime, FS_DEFAULT_TIME);
140 }
141 }
142 // Force write of entry to device.
143 dirFile->m_vol->cacheDirty();
144
145 // open entry in cache.
146 return openCachedEntry(dirFile, index, oflag, 0);
147
148fail:
149 return false;
150}

153 FatSfn_t fname;
154 auto vol = FatVolume::cwv();
155 while (*path == '/') {
156 path++;
157 }

178bool FatFile::openSFN(FatSfn_t* fname) {
179 DirFat_t dir;
180 DirLfn_t* ldir;
181 auto vol = m_vol;
182 uint8_t lfnOrd = 0;
183 if (!isDir()) {
184 DBG_FAIL_MACRO;
185 goto fail;
186 }
187 while (true) {
188 if (read(&dir, 32) != 32) {
189 DBG_FAIL_MACRO;
190 goto fail;
191 }
192 if (dir.name[0] == 0) {
193 DBG_FAIL_MACRO;
194 goto fail;
195 }
196 if (isFatFileOrSubdir(&dir) && memcmp(fname->sfn, dir.name, 11) == 0) {
197 uint16_t saveDirIndex = (m_curPosition - 32) >> 5;
198 uint32_t saveDirCluster = m_firstCluster;
199 memset(this, 0, sizeof(FatFile));
200 m_attributes = dir.attributes & FS_ATTRIB_COPY;
201 m_flags = FILE_FLAG_READ;
202 if (isFatFile(&dir)) {
203 m_attributes |= FILE_ATTR_FILE;
204 if (!isReadOnly()) {
205 m_attributes |= FS_ATTRIB_ARCHIVE;
206 m_flags |= FILE_FLAG_WRITE;
207 }
208 }
209 m_lfnOrd = lfnOrd;
210 m_firstCluster = (uint32_t)getLe16(dir.firstClusterHigh) << 16;
211 m_firstCluster |= getLe16(dir.firstClusterLow);
212 m_fileSize = getLe32(dir.fileSize);
213 m_vol = vol;
214 m_dirCluster = saveDirCluster;
215 m_dirSector = m_vol->cacheSectorNumber();
216 m_dirIndex = saveDirIndex;
217 return true;
218 } else if (isFatLongName(&dir)) {
219 ldir = reinterpret_cast<DirLfn_t*>(&dir);
220 if (ldir->order & FAT_ORDER_LAST_LONG_ENTRY) {
221 lfnOrd = ldir->order & 0X1F;
222 }
223 } else {
224 lfnOrd = 0;
225 }
226 }
227
228fail:
229 return false;
230}

232// format directory name field from a 8.3 name string
233bool FatFile::parsePathName(const char* path, FatSfn_t* fname,
234 const char** ptr) {
235 uint8_t uc = 0;
236 uint8_t lc = 0;
237 uint8_t bit = FNAME_FLAG_LC_BASE;
238 // blank fill name and extension
239 for (uint8_t i = 0; i < 11; i++) {
240 fname->sfn[i] = ' ';
241 }
242 for (uint8_t i = 0, n = 7;; path++) {
243 uint8_t c = *path;
244 if (c == 0 || isDirSeparator(c)) {
245 // Done.
246 break;
247 }
248 if (c == '.' && n == 7) {
249 n = 10; // max index for full 8.3 name
250 i = 8; // place for extension
251
252 // bit for extension.
253 bit = FNAME_FLAG_LC_EXT;
254 } else {
255 if (sfnReservedChar(c) || i > n) {
256 DBG_FAIL_MACRO;
257 goto fail;
258 }
259 if ('a' <= c && c <= 'z') {
260 c += 'A' - 'a';
261 lc |= bit;
262 } else if ('A' <= c && c <= 'Z') {
263 uc |= bit;
264 }
265 fname->sfn[i++] = c;
266 }
267 }
268 // must have a file name, extension is optional
269 if (fname->sfn[0] == ' ') {
270 DBG_FAIL_MACRO;
271 goto fail;
272 }
273 // Set base-name and extension bits.
274 fname->flags = (lc & uc) ? 0 : lc;
275 while (isDirSeparator(*path)) {
276 path++;
277 }
278 *ptr = path;
279 return true;
280
281fail:
282 return false;
283}

286bool FatFile::remove() {
287 DirFat_t* dir;
288 // Can't remove if LFN or not open for write.
289 if (!isWritable() || isLFN()) {
290 DBG_FAIL_MACRO;
291 goto fail;
292 }
293 // Free any clusters.
294 if (m_firstCluster && !m_vol->freeChain(m_firstCluster)) {
295 DBG_FAIL_MACRO;
296 goto fail;
297 }
298 // Cache directory entry.
299 dir = cacheDirEntry(FsCache::CACHE_FOR_WRITE);
300 if (!dir) {
301 DBG_FAIL_MACRO;
302 goto fail;
303 }
304 // Mark entry deleted.
305 dir->name[0] = FAT_NAME_DELETED;
306
307 // Set this file closed.
308 m_attributes = FILE_ATTR_CLOSED;
309 m_flags = 0;
310
311 // Write entry to device.
312 return m_vol->cacheSync();
313
314fail:
315 return false;
316}
