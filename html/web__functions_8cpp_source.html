1110{
1111 int iRead = toPhoneRead;
1112 if (bDEBUG)
1113 Serial.printf("toPhoneWrite:%i toPhoneRead:%i\n", toPhoneWrite, toPhoneRead);
1114
1115 if (toPhoneWrite == 0)
1116 {
1117 web_client.printf("<p>No messages available.</p>");

1119
1120 while (toPhoneWrite != iRead)
1121 {
1122 if (bDEBUG)
1123 Serial.printf("iRead:%i [1]:%02X\n", iRead, BLEtoPhoneBuff[iRead][1]);
1124
1125 uint8_t toPhoneBuff[MAX_MSG_LEN_PHONE] = {0}; // we need to insert the first byte text msg flag
1126 uint8_t blelen = BLEtoPhoneBuff[iRead][0]; // MAXIMUM PACKET Length over BLE is 245 (MTU=247 bytes), two get lost, otherwise we need to split it up!
1127
1128 if (BLEtoPhoneBuff[iRead][1] == 0x91)
1129 { // Mheard
1130 // memcpy(toPhoneBuff, BLEtoPhoneBuff[iRead]+1, blelen-1);
1131 }
1132 else if (BLEtoPhoneBuff[iRead][1] == 0x44)
1133 { // Data Message (JSON)
1134 // memcpy(toPhoneBuff, BLEtoPhoneBuff[iRead]+1, blelen);
1135 }
1136 else
1137 { // Text Message and Position
1138 uint8_t tbuffer[5];
1139 unsigned long unix_time = 0;
1140 char timestamp[21];
1141 String ccheck = "";
1142
1143 memcpy(toPhoneBuff, BLEtoPhoneBuff[iRead] + 1, blelen - 4);
1144 memcpy(tbuffer, BLEtoPhoneBuff[iRead] + 1 + (blelen - 4), 4);
1145 unix_time = (tbuffer[0] << 24) | (tbuffer[1] << 16) | (tbuffer[2] << 8) | tbuffer[3];
1146 time_t unix_t = (time_t)(unix_time + (long)(meshcom_settings.node_utcoff * 60 * 60));
1147 struct tm *oldt = gmtime(&unix_t);
1148 strftime(timestamp, 20, "%Y-%m-%d %H:%M:%S", oldt);
1149 struct aprsMessage aprsmsg;
1150 uint8_t msg_type_b_lora = decodeAPRS(toPhoneBuff, blelen, aprsmsg); // print which message type we got
1151 int icheck = checkOwnTx(aprsmsg.msg_id);
1152
1153 if (icheck >= 0)
1154 {
1155 if (own_msg_id[icheck][4] == 1)
1156 { // 00...not heard, 01...heard, 02...ACK
1157 ccheck = "&#x2713&nbsp;";
1158 }
1159
1160 if (own_msg_id[icheck][4] == 2)
1161 { // 00...not heard, 01...heard, 02...ACK
1162 ccheck = "&#x2611;&nbsp;";
1163 }
1164 }
1165
1166 // Textmessage
1167 if (msg_type_b_lora == 0x3A)
1168 {
1169 if (aprsmsg.msg_payload.indexOf(":ack") < 1)
1170 {
1171 String msgtxt = aprsmsg.msg_payload;
1172 if (bDEBUG)
1173 Serial.printf("aprsmsg.msg_source_call.c_str():%s, aprsmsg.msg_gateway_call.c_str():%s, aprsmsg.msg_destination_call.c_str():%s, aprsmsg.msg_payload.c_str():%s\n", aprsmsg.msg_source_call.c_str(), aprsmsg.msg_source_last.c_str(), aprsmsg.msg_destination_call.c_str(), aprsmsg.msg_payload.c_str());
1174
1175 if (msgtxt.indexOf('{') > 0)
1176 msgtxt = aprsmsg.msg_payload.substring(0, msgtxt.indexOf('{'));
1177
1178 // messages by others
1179 if (strcmp(meshcom_settings.node_call, aprsmsg.msg_source_call.c_str()) == 0)
1180 {
1181 web_client.printf("<div class=\"message message-send\"><div>");
1182
1183 web_client.printf("<p class=\"font-small font-bold\">%s", ccheck.c_str());
1184 web_client.printf("<a target=\"_blank\" href=\"https://aprs.fi/?call=%s\">%s</a>", aprsmsg.msg_source_path.c_str(), aprsmsg.msg_source_path.c_str());
1185 web_client.printf("%s%s</p>", (char *)">", aprsmsg.msg_destination_path.c_str());
1186
1187 web_client.printf("<p class=\"font-small font-bold\">%s</p>", timestamp);
1188 web_client.printf("<p class=\"font-normal\">%s</p>", msgtxt.c_str());
1189 web_client.printf("</div></div>");
1190 }
1191 // own messages
1192 else
1193 {
1194 web_client.printf("<div class=\"message message-received\"><div>");
1195
1196 web_client.printf("<p class=\"font-small font-bold\">%s", ccheck.c_str());
1197 web_client.printf("<a target=\"_blank\" href=\"https://aprs.fi/?call=%s\">%s</a>", aprsmsg.msg_source_path.c_str(), aprsmsg.msg_source_path.c_str());

1199
1200 web_client.printf("<p class=\"font-small font-bold\">%s</p>", timestamp);
1201 web_client.printf("<p class=\"font-normal\">%s</p>", msgtxt.c_str());
1202 web_client.printf("</div></div>");
1203 }
1204 }
1205 }
1206 }
1207 iRead++;
1208 if (iRead >= MAX_RING)
1209 iRead = 0;
1210 }
