33bool FatFile::ls(print_t* pr, uint8_t flags, uint8_t indent) {
34 FatFile file;
35 if (!isDir()) {
36 DBG_FAIL_MACRO;
37 goto fail;
38 }
39 rewind();
40 while (file.openNext(this, O_RDONLY)) {
41 // indent for dir level
42 if (!file.isHidden() || (flags & LS_A)) {
43 for (uint8_t i = 0; i < indent; i++) {
44 pr->write(' ');
45 }
46 if (flags & LS_DATE) {
47 file.printModifyDateTime(pr);
48 pr->write(' ');
49 }
50 if (flags & LS_SIZE) {
51 file.printFileSize(pr);
52 pr->write(' ');
53 }
54 file.printName(pr);
55 if (file.isDir()) {
56 pr->write('/');
57 }
58 pr->write('\r');
59 pr->write('\n');
60 if ((flags & LS_R) && file.isDir()) {
61 file.ls(pr, flags, indent + 2);
62 }
63 }
64 file.close();
65 }
66 if (getError()) {
67 DBG_FAIL_MACRO;
68 goto fail;
69 }
70 return true;
71
72fail:
73 return false;
74}
