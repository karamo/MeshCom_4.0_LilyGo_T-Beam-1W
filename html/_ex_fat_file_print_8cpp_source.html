178size_t ExFatFile::printName8(print_t* pr) {
179 DirName_t* dn;
180 uint16_t hs = 0;
181 uint32_t cp;
182 size_t n = 0;
183 uint8_t in;
184 char buf[5];
185 if (!isOpen()) {
186 DBG_FAIL_MACRO;
187 goto fail;
188 }
189 for (uint8_t is = 2; is <= m_setCount; is++) {
190 dn = reinterpret_cast<DirName_t*>(dirCache(is, FsCache::CACHE_FOR_READ));
191 if (!dn || dn->type != EXFAT_TYPE_NAME) {
192 DBG_FAIL_MACRO;
193 goto fail;
194 }
195 for (in = 0; in < 15; in++) {
196 uint16_t c = getLe16(dn->unicode + 2 * in);
197 if (hs) {
198 if (!FsUtf::isLowSurrogate(c)) {
199 DBG_FAIL_MACRO;
200 goto fail;
201 }
202 cp = FsUtf::u16ToCp(hs, c);
203 hs = 0;
204 } else if (!FsUtf::isSurrogate(c)) {
205 if (c == 0) {
206 break;
207 }
208 cp = c;
209 } else if (FsUtf::isHighSurrogate(c)) {
210 hs = c;
211 continue;
212 } else {
213 DBG_FAIL_MACRO;
214 goto fail;
215 }
216 char* str = FsUtf::cpToMb(cp, buf, buf + sizeof(buf));
217 if (!str) {
218 DBG_FAIL_MACRO;
219 goto fail;
220 }
221 n += pr->write(reinterpret_cast<uint8_t*>(buf), str - buf);
222 }
223 }
224 return n;
225
226fail:
227 return 0;
228}
