203int NrfETH::getUDP()
204{
205 char source_call[20] = {0};
206 char destination_call[20] = {0};
207
208 uint8_t convBuffer[UDP_TX_BUF_SIZE+5]; // we need an extra buffer for udp tx, as we add other stuff (ID, RSSI, SNR, MODE)
209
210 udp_is_busy = true; //setting the busy flag
211
212 int packetSize = Udp.parsePacket(); // If there's data available, read a packet.
213
214 // HEARTBEAT keine Ausgabe
215 //if(packetSize != 22 && packetSize > 0 && bDEBUG)
216 if(packetSize > 0 && bDEBUG)
217 Serial.printf("[UDP_ETH] UDP Packet received with length: %i\n", packetSize);
218
219 if (packetSize <= UDP_TX_BUF_SIZE && packetSize > 0)
220 {
221 // read the packet
222 Udp.read(inc_udp_buffer, UDP_TX_BUF_SIZE); // Read the packet into packetBufffer.
223
224 // if more than n values are 00 we might have received a faulty message
225 uint8_t zerocount = 0;
226
227 for (int i = 0; i < packetSize; i+=2)
228 {
229 if (inc_udp_buffer[i] == 0x00 && inc_udp_buffer[i + 1] == 0x00)
230 {
231 zerocount += 2;
232 }
233 else
234 zerocount = 0;
235 }
236
237 if(packetSize > 0 && bDEBUG)// && bDEBUG)
238 Serial.printf("[UDP_ETH] UDP zerocount: %i ? > 6\n", zerocount);
239
240 if (zerocount <= MAX_ZEROS)
241 {
242 /* we now need to distinguish if we got a LoRa packet to send from the server
243 or it is a config message. First 4 Bytes indicate if it is
244 GATE: 0x47 41 54 45
245 CONF: 0x43 4F 4E 46
246 */
247
248 // get the first 4 bytes of the incoming udp message
249 char indicator_b[UDP_MSG_INDICATOR_LEN];
250
251 memcpy(indicator_b, inc_udp_buffer, UDP_MSG_INDICATOR_LEN);
252
253 char gate[] = "GATE";
254 char conf[] = "CONF";
255 char beat[] = "BEAT";
256
257 if (memcmp(indicator_b, gate, UDP_MSG_INDICATOR_LEN) == 0)
258 {
259
260 if(bDEBUG)
261 Serial.printf("[GATE] Received a LoRa packet to transmit\n");
262
263 last_upd_timer = millis();
264
265 lora_tx_msg_len = packetSize - UDP_MSG_INDICATOR_LEN;
266 if (lora_tx_msg_len > UDP_TX_BUF_SIZE)
267 lora_tx_msg_len = UDP_TX_BUF_SIZE; // zur Sicherheit
268
269 memcpy(RcvBuffer, inc_udp_buffer+UDP_MSG_INDICATOR_LEN, lora_tx_msg_len);
270
271 // send JSON to Extern IP
272 if(bEXTUDP)
273 sendExtern(true, (char*)"udp", RcvBuffer, (uint8_t)lora_tx_msg_len);
274
275 // printout message type
276 uint8_t msg_type_b = RcvBuffer[0];
277
278 switch (msg_type_b)
279 {
280 case 0x3A: DEBUG_MSG("UDP", "Received Textmessage"); break; // ':'
281 case 0x21: DEBUG_MSG("UDP", "Received PosInfo"); break;
282 case 0x40: DEBUG_MSG("UDP", "Received Weather"); break;
283 default: DEBUG_MSG("UDP", "Received unknown"); break;
284 }
285
286 if (msg_type_b == 0x3A || msg_type_b == 0x21 || msg_type_b == 0x40)
287 {
288 bool bBLELoopOut = true;
289
290 struct aprsMessage aprsmsg;
291
292 // print which message type we got
293 uint8_t msg_type_b_lora = decodeAPRS(RcvBuffer, lora_tx_msg_len, aprsmsg);
294
295 if(msg_type_b_lora > 0)
296 {
297 if(bDisplayInfo)
298 {
299 printBuffer_aprs((char*)"RX-UDP ", aprsmsg);
300 Serial.println();
301 }
302
303 bool bUDPtoLoraSend = true;
304
305 snprintf(source_call, sizeof(source_call), "%s", aprsmsg.msg_source_call.c_str());
306 snprintf(destination_call, sizeof(destination_call), "%s", aprsmsg.msg_destination_call.c_str());
307
308 aprsmsg.msg_source_path.concat(',');
309 aprsmsg.msg_source_path.concat(meshcom_settings.node_call);
310
311 aprsmsg.msg_server = true;
312
313 aprsmsg.msg_last_hw = BOARD_HARDWARE | 0x80; // hardware last sending node
314 aprsmsg.msg_source_mod = (getMOD() & 0xF) | (meshcom_settings.node_country << 4); // modulation & country
315
316 if(bDEBUG)
317 {
318 Serial.printf("RX-UDP Source-Path:%s\n", aprsmsg.msg_source_path.c_str());
319 }
320
321
322 memset(convBuffer, 0x00, UDP_TX_BUF_SIZE);
323
324 uint16_t size = encodeAPRS(convBuffer, aprsmsg);
325
326 if(size > UDP_TX_BUF_SIZE)
327 size = UDP_TX_BUF_SIZE;
328
329 if(bDEBUG)
330 {
331 Serial.printf("RX-UDP Check-payload (%i):%02X \n", size, msg_type_b);
332 }
333
334 if(msg_type_b == 0x3A)
335 {
336 if(memcmp(aprsmsg.msg_payload.c_str(), "{SET}", 5) == 0)
337 {
338 sendDisplayText(aprsmsg, (int16_t)99, (int8_t)0);
339 }
340 else
341 if(memcmp(aprsmsg.msg_payload.c_str(), "{CET}", 5) == 0)
342 {
343 sendDisplayText(aprsmsg, (int16_t)99, (int8_t)0);
344 }
345 else
346 if((strcmp(destination_call, "*") == 0 && !bNoMSGtoALL) || strcmp(destination_call, meshcom_settings.node_call) == 0 || CheckGroup(destination_call) > 0)
347 {
348 // wenn eine Meldung via UDP kommt und den eigene Node betrifft dann keine weiterleitung an LoRa TX
349 if(strcmp(destination_call, meshcom_settings.node_call) == 0)
350 bUDPtoLoraSend=false;
351
352 unsigned int iAckId = 0;
353
354 int iAckPos=aprsmsg.msg_payload.indexOf(":ack");
355 int iRefPos=aprsmsg.msg_payload.indexOf(":rej");
356 int iEnqPos=aprsmsg.msg_payload.indexOf("{", 1);
357
358 if(strcmp(destination_call, "*") == 0)
359 {
360 iAckPos=0;
361 iRefPos=0;
362 iEnqPos=0;
363 }
364
365 if(iAckPos > 0 || iRefPos > 0)
366 {
367 unsigned int iAckId = (aprsmsg.msg_payload.substring(iAckPos+4)).toInt();
368 msg_counter = ((_GW_ID & 0x3FFFFF) << 10) | (iAckId & 0x3FF);
369
370 uint8_t print_buff[30];
371
372 print_buff[0]=0x41;
373 print_buff[1]=msg_counter & 0xFF;
374 print_buff[2]=(msg_counter >> 8) & 0xFF;
375 print_buff[3]=(msg_counter >> 16) & 0xFF;
376 print_buff[4]=(msg_counter >> 24) & 0xFF;
377 print_buff[5]=0x01; // ACK
378 print_buff[6]=0x00;
379
380 if(bDisplayInfo)
381 Serial.printf("\n[UDP-MSGID] ack_msg_id:%02X%02X%02X%02X\n", print_buff[1], print_buff[2], print_buff[3], print_buff[4]);
382
383 int iackcheck = checkOwnTx(msg_counter);
384 if(iackcheck >= 0)
385 {
386 own_msg_id[iackcheck][4] = 0x02; // 02...ACK
387 }
388
389 addBLEOutBuffer(print_buff, 7);
390
391 if(strcmp(source_call, meshcom_settings.node_call) == 0)
392 bUDPtoLoraSend=false;
393
394 bBLELoopOut=false;
395 }
396
397 if(iEnqPos > 0)
398 {
399 iAckId = (aprsmsg.msg_payload.substring(iEnqPos+1)).toInt();
400 aprsmsg.msg_payload = aprsmsg.msg_payload.substring(0, iEnqPos);
401 }
402
403 if(iAckPos <= 0)
404 {
405 if(!bGATEWAY)
406 sendDisplayText(aprsmsg, (int16_t)99, (int8_t)0);
407 }
408
409 aprsmsg.max_hop = aprsmsg.max_hop | 0x20; // msg_app_offline true
410
411 uint8_t tempRcvBuffer[UDP_TX_BUF_SIZE];
412
413 aprsmsg.msg_last_hw = BOARD_HARDWARE | 0x80; // hardware last sending node
414 aprsmsg.msg_source_mod = (getMOD() & 0xF) | (meshcom_settings.node_country << 4); // modulation & country
415
416 uint16_t tempsize = encodeAPRS(tempRcvBuffer, aprsmsg);
417
418 addBLEOutBuffer(tempRcvBuffer, tempsize);
419
420 bBLELoopOut=false;
421
422 // DM message for lokal Node
423 if(iAckId > 0)
424 {
425 strSource_call = source_call;
426 SendAckMessage(strSource_call, iAckId);
427 }
428 }
429 }
430
431 // resend only Packet
432 if(bUDPtoLoraSend)
433 {
434 // store last message to compare later on
435 insertOwnTx(aprsmsg.msg_id);
436
437 ringBuffer[iWrite][0] = size;
438 if (msg_type_b == 0x3A) // only Messages
439 {
440 if(aprsmsg.msg_payload.startsWith("{") > 0)
441 ringBuffer[iWrite][1] = 0xFF; // retransmission Status ...0xFF no retransmission on {CET} & Co.
442 else
443 ringBuffer[iWrite][1] = 0x00; // retransmission Status ...0xFF no retransmission
444 }
445 else
446 ringBuffer[iWrite][1] = 0xFF; // retransmission Status ...0xFF no retransmission
447 memcpy(ringBuffer[iWrite] + 2, convBuffer, size);
448 iWrite++;
449 if (iWrite >= MAX_RING) // if the buffer is full we start at index 0 -> take care of overwriting!
450 iWrite = 0;
451
452 if(bDEBUG)
453 {
454 Serial.printf("RX-UDP addLoraRxBuffer\n");
455 }
456
457 addLoraRxBuffer(aprsmsg.msg_id, true);
458
459 // add rcvMsg to BLE out Buff
460 // size message is int -> uint16_t buffer size
461 if(isPhoneReady == 1 && bBLELoopOut) // wird schon vorher abgehandelt
462 {
463 if(bDEBUG)
464 {
465 Serial.printf("RX-UDP addBLEOutBuffer\n");
466 }
467
468 addBLEOutBuffer(convBuffer, size);
469 }
470 }
471 }
472 }
473 }
474 else if (memcmp(indicator_b, conf, UDP_MSG_INDICATOR_LEN) == 0)
475 {
476
477 if(bDisplayInfo)
478 {
479 Serial.print(getTimeString());
480 Serial.printf("[CONF] received from server\n");
481 }
482
483 last_upd_timer = millis();
484
485 had_initial_udp_conn = true;
486
487 /* Handling Config Messages (sticking for now without own method)
488 * first 4 bytes are 'CONF' (already checked at this point)
489 * 5th byte is an indicator which setting is coming:
490 * 0x00 -> Callsign - length of chars - chars - 0x01 - length - shortname
491 * CONF 0x00 LL bytes(rufzeichen) - 0x01 - shortname(3)
492 */
493
494 uint8_t config_buf[UDP_CONF_BUFF_SIZE] = {0};
495
496 if (packetSize <= UDP_CONF_BUFF_SIZE && packetSize >= UDP_MSG_INDICATOR_LEN)
497 {
498 memcpy(config_buf, inc_udp_buffer + UDP_MSG_INDICATOR_LEN, packetSize - UDP_MSG_INDICATOR_LEN);
499 // fill rest of buffer with 0
500 for (int i = 0; i < UDP_CONF_BUFF_SIZE; i++)
501 {
502 config_buf[packetSize - UDP_MSG_INDICATOR_LEN + i] = 0x00;
503 }
504
505 // print the message
506 // printBuffer(config_buf, packetSize - UDP_MSG_INDICATOR_LEN);
507
508 // check which config arrived and proceed
509 if (config_buf[0] == 0x00)
510 {
511 // we got a callsign from server
512 int call_len = config_buf[1];
513 char call_arr[call_len + 1];
514 call_arr[call_len] = '\0';
515 memcpy(call_arr, config_buf + 2, call_len);
516 _longname = String(call_arr);
517 DEBUG_MSG("CONF", "Got callsign (longanme) from server: ");
518 Serial.println(_longname);
519 DEBUG_MSG("CONF", "Callsign Length: %d", call_len);
520
521 // shortname
522 int short_len=0;
523 if (config_buf[2 + call_len] == 0x01)
524 {
525 short_len = config_buf[2 + call_len + 1];
526 char short_arr[short_len + 1];
527 memcpy(short_arr, config_buf + (2 + call_len + 2), short_len);
528 short_arr[short_len] = '\0';
529 shortname = String(short_arr);
530 DEBUG_MSG_VAL("CONF", short_len, "Shortname received: ");
531 Serial.println(shortname);
532 }
533
534 int inpos= 2 + call_len + short_len + 2;
535
536 // lat
537 if (config_buf[inpos] == 0x02)
538 {
539 _lat=config_buf[inpos+1] | (config_buf[inpos+2] << 8) | (config_buf[inpos+3] << 16) | (config_buf[inpos+4] << 24);
540 DEBUG_MSG_VAL("CONF", _lat, "LAT received: ");
541 inpos=inpos+5;
542 }
543
544 // lon
545 if (config_buf[inpos] == 0x03)
546 {
547 _lon=config_buf[inpos+1] | (config_buf[inpos+2] << 8) | (config_buf[inpos+3] << 16) | (config_buf[inpos+4] << 24);
548 DEBUG_MSG_VAL("CONF", _lon, "LON received: ");
549 inpos=inpos+5;
550 }
551
552 // lat
553 if (config_buf[inpos] == 0x04)
554 {
555 _alt=config_buf[inpos+1] | (config_buf[inpos+2] << 8) | (config_buf[inpos+3] << 16) | (config_buf[inpos+4] << 24);
556 DEBUG_MSG_VAL("CONF", _alt, "ALT received: ");
557 inpos=inpos+5;
558 }
559 }
560 else
561 {
562 Serial.printf("[ERROR] Incoming config message not known! Discarding!\n");
563 }
564 }
565 else
566 {
567 Serial.printf("[ERROR] Incoming config message not known! Discarding!\n");
568 }
569 }
570 else if (memcmp(indicator_b, beat, UDP_MSG_INDICATOR_LEN) == 0)
571 {
572
573 // we got an heartbeat from server which we use to check connection (saving time we got it)
574 if(bDEBUG)
575 {
576 Serial.print(getTimeString());
577 Serial.printf(" [BEAT] Heartbeat from server\n");
578 }
579
580 last_upd_timer = millis();
581
587 }
588 else
589 {
590 Serial.printf("[ERROR] Received udp message without indicator\n");
591 last_upd_timer = millis();
592 }
593
594 // zero out the inc buffer
595 memset(inc_udp_buffer, 0, UDP_TX_BUF_SIZE);
596
597 udp_is_busy = false; //setting the busy flag
598 return 0;
599 }
600 else
601 {
602 Serial.printf("[ERROR] UDP Message has too much Zeros\n");
603 resetDHCP();
604 }
605 }
606
607 // zero out the inc buffer
608 memset(inc_udp_buffer, 0, UDP_TX_BUF_SIZE);
609
610 udp_is_busy = false; //setting the busy flag
611 return 1;
612}

656void NrfETH::getMyMac()
657{
658 /*https://devzone.nordicsemi.com/f/nordic-q-a/22089/specific-unique-id-for-device-detection/86795#86795
659
660 * NRF_FICR->DEVICEADDR[] is array of 32-bit words.
661 * NRF_FICR->DEVICEADDR yields type (unit32_t*)
662 * Cast: (uint64_t*) NRF_FICR->DEVICEADDR yields type (unit64_t*)
663 * Dereferencing: *(uint64_t*) NRF_FICR->DEVICEADDR yields type uint64_t
664 *
665 * Nordic doc asserts upper two bytes read all ones.
666 */
667
668 uint64_t result = *(uint64_t *)NRF_FICR->DEVICEADDR;
669
670 // Mask off upper bytes, to match over-the-air length of 6 bytes.
671 result = result & MAX_DEVICE_ID;
672
673 macaddr[0] = 0x00; //we set the first byte fix, otherwise troubles with DHCP server
674
675 memcpy(macaddr +1, &result, sizeof(result) -3);
676
677 /*
678 Serial.print("MAC ADDR: ");
679
680 for (int i = 0; i < 6; i++)
681 {
682 Serial.printf("%02X:", macaddr[i]);
683 }
684 Serial.println("");
685 */
686
687 // setting now the GW-ID
688 DEBUG_MSG("Radio", "Setting GW-ID");
689
690 //memcpy makes little endian, so we need to shift bytes our self
691 _GW_ID = macaddr[5] | (macaddr[4] << 8) | (macaddr[3] << 16) | (macaddr[2] << 24);
692
693 DEBUG_MSG("Radio", "GW-ID: 0x%04X", _GW_ID);
694}

754int NrfETH::startETH()
755{
756 if(hasIPaddress)
757 return 0;
758
