420//------------------------------------------------------------------------------
421static bool cardCommand(uint32_t xfertyp, uint32_t arg) {
422 DBG_IRQSTAT();
423 if (waitTimeout(isBusyCommandInhibit)) {
424 return false; // Caller will set errorCode.
425 }
426 SDHC_CMDARG = arg;
427#if defined(__IMXRT1062__)
428 // Set MIX_CTRL if data transfer.
429 if (xfertyp & SDHC_XFERTYP_DPSEL) {
430 SDHC_MIX_CTRL &= ~SDHC_MIX_CTRL_MASK;
431 SDHC_MIX_CTRL |= xfertyp & SDHC_MIX_CTRL_MASK;
432 }
433 xfertyp &= ~SDHC_MIX_CTRL_MASK;
434#endif // defined(__IMXRT1062__)
435 SDHC_XFERTYP = xfertyp;
436 if (waitTimeout(isBusyCommandComplete)) {
437 return false; // Caller will set errorCode.
438 }
439 m_irqstat = SDHC_IRQSTAT;
440 SDHC_IRQSTAT = m_irqstat;
441
442 return (m_irqstat & SDHC_IRQSTAT_CC) && !(m_irqstat & SDHC_IRQSTAT_CMD_ERROR);
443}

445static bool cardACMD13(sds_t* scr) {
446 // ACMD13 returns 64 bytes.
447 if (waitTimeout(isBusyCMD13)) {
448 return sdError(SD_CARD_ERROR_CMD13);
449 }
450 enableDmaIrs();
451 SDHC_DSADDR = (uint32_t)scr;
452 SDHC_BLKATTR = SDHC_BLKATTR_BLKCNT(1) | SDHC_BLKATTR_BLKSIZE(64);
453 SDHC_IRQSIGEN = SDHC_IRQSIGEN_MASK;
454 if (!cardAcmd(m_rca, ACMD13_XFERTYP, 0)) {
455 return sdError(SD_CARD_ERROR_ACMD13);
456 }
457 if (!waitDmaStatus()) {
458 return sdError(SD_CARD_ERROR_DMA);
459 }
460 return true;
461}

485//------------------------------------------------------------------------------
486static void initSDHC() {
487 initClock();
488
489 // Disable GPIO clock.
490 enableGPIO(false);
491
492#if defined(__IMXRT1062__)
493 SDHC_MIX_CTRL |= 0x80000000;
494#endif // (__IMXRT1062__)
495
496 // Reset SDHC. Use default Water Mark Level of 16.
497 SDHC_SYSCTL |= SDHC_SYSCTL_RSTA | SDHC_SYSCTL_SDCLKFS(0x80);
498
499 while (SDHC_SYSCTL & SDHC_SYSCTL_RSTA) {
500 }
501
502 // Set initial SCK rate.
503 setSdclk(SD_MAX_INIT_RATE_KHZ);
504
505 enableGPIO(true);
506
507 // Enable desired IRQSTAT bits.
508 SDHC_IRQSTATEN = SDHC_IRQSTATEN_MASK;
509
510 attachInterruptVector(IRQ_SDHC, sdIrs);
511 NVIC_SET_PRIORITY(IRQ_SDHC, 6 * 16);
512 NVIC_ENABLE_IRQ(IRQ_SDHC);
513
514 // Send 80 clocks to card.
515 SDHC_SYSCTL |= SDHC_SYSCTL_INITA;
516 while (SDHC_SYSCTL & SDHC_SYSCTL_INITA) {
517 }
518}

546static bool rdWrSectors(uint32_t xfertyp, uint32_t sector, uint8_t* buf,
547 size_t n) {
548 if ((3 & (uint32_t)buf) || n == 0) {
549 return sdError(SD_CARD_ERROR_DMA);
550 }
551 if (yieldTimeout(isBusyCMD13)) {
552 return sdError(SD_CARD_ERROR_CMD13);
553 }
554 enableDmaIrs();
555 SDHC_DSADDR = (uint32_t)buf;
556 SDHC_BLKATTR = SDHC_BLKATTR_BLKCNT(n) | SDHC_BLKATTR_BLKSIZE(512);
557 SDHC_IRQSIGEN = SDHC_IRQSIGEN_MASK;
558 if (!cardCommand(xfertyp, m_highCapacity ? sector : 512 * sector)) {
559 return false;
560 }
561 return waitDmaStatus();
562}
563//------------------------------------------------------------------------------
564// Read 16 byte CID or CSD register.
565static bool readReg16(uint32_t xfertyp, void* data) {
566 uint8_t* d = reinterpret_cast<uint8_t*>(data);
567 if (!cardCommand(xfertyp, m_rca)) {
568 return false; // Caller will set errorCode.
569 }
570 uint32_t sr[] = {SDHC_CMDRSP0, SDHC_CMDRSP1, SDHC_CMDRSP2, SDHC_CMDRSP3};
571 for (int i = 0; i < 15; i++) {
572 d[14 - i] = sr[i / 4] >> 8 * (i % 4);
573 }
574 d[15] = 0;
575 return true;
576}
577//------------------------------------------------------------------------------
578static void setSdclk(uint32_t kHzMax) {
579 const uint32_t DVS_LIMIT = 0X10;
580 const uint32_t SDCLKFS_LIMIT = 0X100;
581 uint32_t dvs = 1;
582 uint32_t sdclkfs = 1;
583 uint32_t maxSdclk = 1000 * kHzMax;
584 uint32_t base = baseClock();
585
586 while ((base / (sdclkfs * DVS_LIMIT) > maxSdclk) &&
587 (sdclkfs < SDCLKFS_LIMIT)) {
588 sdclkfs <<= 1;
589 }
590 while ((base / (sdclkfs * dvs) > maxSdclk) && (dvs < DVS_LIMIT)) {
591 dvs++;
592 }
593 m_sdClkKhz = base / (1000 * sdclkfs * dvs);
594 sdclkfs >>= 1;
595 dvs--;
596#if defined(__MK64FX512__) || defined(__MK66FX1M0__)
597 // Disable SDHC clock.
598 SDHC_SYSCTL &= ~SDHC_SYSCTL_SDCLKEN;
599#endif // defined(__MK64FX512__) || defined(__MK66FX1M0__)
600
601 // Change dividers.
602 uint32_t sysctl =
603 SDHC_SYSCTL & ~(SDHC_SYSCTL_DTOCV_MASK | SDHC_SYSCTL_DVS_MASK |
604 SDHC_SYSCTL_SDCLKFS_MASK);
605
606 SDHC_SYSCTL = sysctl | SDHC_SYSCTL_DTOCV(0x0E) | SDHC_SYSCTL_DVS(dvs) |
607 SDHC_SYSCTL_SDCLKFS(sdclkfs);
608
609 // Wait until the SDHC clock is stable.
610 while (!(SDHC_PRSSTAT & SDHC_PRSSTAT_SDSTB)) {
611 }
612
613#if defined(__MK64FX512__) || defined(__MK66FX1M0__)
614 // Enable the SDHC clock.
615 SDHC_SYSCTL |= SDHC_SYSCTL_SDCLKEN;
616#endif // defined(__MK64FX512__) || defined(__MK66FX1M0__)
617}
618//------------------------------------------------------------------------------
619static bool transferStop() {
620 // This fix allows CDIHB to be cleared in Tennsy 3.x without a reset.
621 SDHC_PROCTL &= ~SDHC_PROCTL_SABGREQ;
622 if (!cardCommand(CMD12_XFERTYP, 0)) {
623 return sdError(SD_CARD_ERROR_CMD12);
624 }
625 if (yieldTimeout(isBusyDat)) {
626 return sdError(SD_CARD_ERROR_CMD13);
627 }
628 if (SDHC_PRSSTAT & SDHC_PRSSTAT_CDIHB) {
629 // This should not happen after above fix.
630 // Save registers before reset DAT lines.
631 uint32_t irqsststen = SDHC_IRQSTATEN;
632 uint32_t proctl = SDHC_PROCTL & ~SDHC_PROCTL_SABGREQ;
633 // Do reset to clear CDIHB. Should be a better way!
634 SDHC_SYSCTL |= SDHC_SYSCTL_RSTD;
635 // Restore registers.
636 SDHC_IRQSTATEN = irqsststen;
637 SDHC_PROCTL = proctl;
638 }
639 return true;
640}
641//------------------------------------------------------------------------------
642// Return true if timeout occurs.
643static bool yieldTimeout(bool (*fcn)()) {
644 m_busyFcn = fcn;
645 uint32_t m = micros();
646 while (fcn()) {
647 if ((micros() - m) > BUSY_TIMEOUT_MICROS) {
648 m_busyFcn = 0;
649 return true;
650 }
651 yield();
652 }
653 m_busyFcn = 0;
654 return false; // Caller will set errorCode.
655}
656//------------------------------------------------------------------------------
657static bool waitDmaStatus() {
658 if (yieldTimeout(isBusyDMA)) {
659 return false; // Caller will set errorCode.
660 }
661 return (m_irqstat & SDHC_IRQSTAT_TC) && !(m_irqstat & SDHC_IRQSTAT_ERROR);
662}
663//------------------------------------------------------------------------------
664// Return true if timeout occurs.
665static bool waitTimeout(bool (*fcn)()) {
666 uint32_t m = micros();
667 while (fcn()) {
668 if ((micros() - m) > BUSY_TIMEOUT_MICROS) {
669 return true;
670 }
671 }
672 return false; // Caller will set errorCode.
673}
674//------------------------------------------------------------------------------
675static bool waitTransferComplete() {
676 if (!m_transferActive) {
677 return true;
678 }
679 bool timeOut = waitTimeout(isBusyTransferComplete);
680 m_transferActive = false;
681 m_irqstat = SDHC_IRQSTAT;
682 SDHC_IRQSTAT = m_irqstat;
683 if (timeOut || (m_irqstat & SDHC_IRQSTAT_ERROR)) {
684 return sdError(SD_CARD_ERROR_TRANSFER_COMPLETE);
685 }
686 return true;

688//==============================================================================
689// Start of SdioCard member functions.
690//==============================================================================
691bool SdioCard::begin(SdioConfig sdioConfig) {
692 uint32_t kHzSdClk;
693 uint32_t arg;
694 m_sdioConfig = sdioConfig;
695 m_curState = IDLE_STATE;
696 m_initDone = false;
697 m_errorCode = SD_CARD_ERROR_NONE;
698 m_highCapacity = false;
699 m_version2 = false;
700
701 // initialize controller.
702 initSDHC();
703 if (!cardCommand(CMD0_XFERTYP, 0)) {
704 return sdError(SD_CARD_ERROR_CMD0);
705 }
706 // Try several times for case of reset delay.
707 for (uint32_t i = 0; i < CMD8_RETRIES; i++) {
708 if (cardCommand(CMD8_XFERTYP, 0X1AA)) {
709 if (SDHC_CMDRSP0 != 0X1AA) {
710 return sdError(SD_CARD_ERROR_CMD8);
711 }
712 m_version2 = true;
713 break;
714 }
715 SDHC_SYSCTL |= SDHC_SYSCTL_RSTA;
716 while (SDHC_SYSCTL & SDHC_SYSCTL_RSTA) {
717 }
718 }
719 // Must support 3.2-3.4 Volts
720 arg = m_version2 ? 0X40300000 : 0x00300000;
721 int m = micros();
722 do {
723 if (!cardAcmd(0, ACMD41_XFERTYP, arg) ||
724 ((micros() - m) > BUSY_TIMEOUT_MICROS)) {
725 return sdError(SD_CARD_ERROR_ACMD41);
726 }
727 } while ((SDHC_CMDRSP0 & 0x80000000) == 0);
728 m_ocr = SDHC_CMDRSP0;
729 if (SDHC_CMDRSP0 & 0x40000000) {
730 // Is high capacity.
731 m_highCapacity = true;
732 }
733 if (!cardCommand(CMD2_XFERTYP, 0)) {
734 return sdError(SD_CARD_ERROR_CMD2);
735 }
736 if (!cardCommand(CMD3_XFERTYP, 0)) {
737 return sdError(SD_CARD_ERROR_CMD3);
738 }
739 m_rca = SDHC_CMDRSP0 & 0xFFFF0000;
740
741 if (!readReg16(CMD9_XFERTYP, &m_csd)) {
742 return sdError(SD_CARD_ERROR_CMD9);
743 }
744 if (!readReg16(CMD10_XFERTYP, &m_cid)) {
745 return sdError(SD_CARD_ERROR_CMD10);
746 }
747 if (!cardCommand(CMD7_XFERTYP, m_rca)) {
748 return sdError(SD_CARD_ERROR_CMD7);
749 }
750 // Set card to bus width four.
751 if (!cardAcmd(m_rca, ACMD6_XFERTYP, 2)) {
752 return sdError(SD_CARD_ERROR_ACMD6);
753 }
754 // Set SDHC to bus width four.
755 SDHC_PROCTL &= ~SDHC_PROCTL_DTW_MASK;
756 SDHC_PROCTL |= SDHC_PROCTL_DTW(SDHC_PROCTL_DTW_4BIT);
757
758 SDHC_WML = SDHC_WML_RDWML(FIFO_WML) | SDHC_WML_WRWML(FIFO_WML);
759
760 if (!cardACMD51(&m_scr)) {
761 return false;
762 }
763 if (!cardACMD13(&m_sds)) {
764 return false;
765 }
766 // Determine if High Speed mode is supported and set frequency.
767 // Check status[16] for error 0XF or status[16] for new mode 0X1.
768 uint8_t status[64];
769 if (m_scr.sdSpec() > 0 && cardCMD6(0X00FFFFFF, status) && (2 & status[13]) &&
770 cardCMD6(0X80FFFFF1, status) && (status[16] & 0XF) == 1) {
771 kHzSdClk = 50000;
772 } else {
773 kHzSdClk = 25000;
774 }
775 // Disable GPIO.
776 enableGPIO(false);
777
778 // Set the SDHC SCK frequency.
779 setSdclk(kHzSdClk);
780
781 // Enable GPIO.
782 enableGPIO(true);
783 m_initDone = true;
784 return true;
785}

787bool SdioCard::cardCMD6(uint32_t arg, uint8_t* status) {
788 // CMD6 returns 64 bytes.
789 if (waitTimeout(isBusyCMD13)) {
790 return sdError(SD_CARD_ERROR_CMD13);

792 enableDmaIrs();

794 SDHC_BLKATTR = SDHC_BLKATTR_BLKCNT(1) | SDHC_BLKATTR_BLKSIZE(64);
795 SDHC_IRQSIGEN = SDHC_IRQSIGEN_MASK;
796 if (!cardCommand(CMD6_XFERTYP, arg)) {
797 return sdError(SD_CARD_ERROR_CMD6);

799 if (!waitDmaStatus()) {
800 return sdError(SD_CARD_ERROR_DMA);
801 }
802 return true;
803}
804//------------------------------------------------------------------------------
805bool SdioCard::erase(uint32_t firstSector, uint32_t lastSector) {
806 if (m_curState != IDLE_STATE && !syncDevice()) {
807 return false;
808 }
809 // check for single sector erase
810 if (!m_csd.eraseSingleBlock()) {
811 // erase size mask
812 uint8_t m = m_csd.eraseSize() - 1;
813 if ((firstSector & m) != 0 || ((lastSector + 1) & m) != 0) {
814 // error card can't erase specified area
815 return sdError(SD_CARD_ERROR_ERASE_SINGLE_SECTOR);
816 }
817 }
818 if (!m_highCapacity) {
819 firstSector <<= 9;
820 lastSector <<= 9;
821 }
822 if (!cardCommand(CMD32_XFERTYP, firstSector)) {
823 return sdError(SD_CARD_ERROR_CMD32);
824 }
825 if (!cardCommand(CMD33_XFERTYP, lastSector)) {
826 return sdError(SD_CARD_ERROR_CMD33);
827 }
828 if (!cardCommand(CMD38_XFERTYP, 0)) {
829 return sdError(SD_CARD_ERROR_CMD38);
830 }
831 if (waitTimeout(isBusyCMD13)) {
832 return sdError(SD_CARD_ERROR_ERASE_TIMEOUT);
833 }
834 return true;

879bool SdioCard::readData(uint8_t* dst) {
880 DBG_IRQSTAT();
881 uint32_t* p32 = reinterpret_cast<uint32_t*>(dst);
882
883 if (!(SDHC_PRSSTAT & SDHC_PRSSTAT_RTA)) {
884 SDHC_PROCTL &= ~SDHC_PROCTL_SABGREQ;
885 noInterrupts();
886 SDHC_PROCTL |= SDHC_PROCTL_CREQ;
887 SDHC_PROCTL |= SDHC_PROCTL_SABGREQ;
888 interrupts();
889 }
890 if (waitTimeout(isBusyFifoRead)) {
891 return sdError(SD_CARD_ERROR_READ_FIFO);
892 }
893 for (uint32_t iw = 0; iw < 512 / (4 * FIFO_WML); iw++) {
894 while (0 == (SDHC_PRSSTAT & SDHC_PRSSTAT_BREN)) {
895 }
896 for (uint32_t i = 0; i < FIFO_WML; i++) {
897 p32[i] = SDHC_DATPORT;
898 }
899 p32 += FIFO_WML;
900 }
901 if (waitTimeout(isBusyTransferComplete)) {
902 return sdError(SD_CARD_ERROR_READ_TIMEOUT);
903 }
904 m_irqstat = SDHC_IRQSTAT;
905 SDHC_IRQSTAT = m_irqstat;
906 return (m_irqstat & SDHC_IRQSTAT_TC) && !(m_irqstat & SDHC_IRQSTAT_ERROR);
