321struct csd_t {
323 uint8_t csd[16];
324 // Extract big endian fields.
326 uint32_t capacity() const {
327 uint32_t c_size;
328 uint8_t ver = csd[0] >> 6;
329 if (ver == 0) {
330 c_size = (uint32_t)(csd[6] & 3) << 10;
331 c_size |= (uint32_t)csd[7] << 2 | csd[8] >> 6;
332 uint8_t c_size_mult = (csd[9] & 3) << 1 | csd[10] >> 7;
333 uint8_t read_bl_len = csd[5] & 15;
334 return (c_size + 1) << (c_size_mult + read_bl_len + 2 - 9);
335 } else if (ver == 1) {
336 c_size = (uint32_t)(csd[7] & 63) << 16;
337 c_size |= (uint32_t)csd[8] << 8;
338 c_size |= csd[9];
339 return (c_size + 1) << 10;
340 } else {
341 return 0;
342 }
343 }
344
345 bool eraseSingleBlock() const { return csd[10] & 0X40; }
347 int eraseSize() const { return ((csd[10] & 0X3F) << 1 | csd[11] >> 7) + 1; }
349 bool copy() const { return csd[14] & 0X40; }
351 bool permWriteProtect() const { return csd[14] & 0X20; }
353 bool tempWriteProtect() const { return csd[14] & 0X10; }
354};

360struct scr_t {
362 uint8_t scr[8];
364 uint8_t srcStructure() const { return scr[0] >> 4; }
366 uint8_t sdSpec() const { return scr[0] & 0XF; }
368 bool dataAfterErase() const { return scr[1] & 0X80; }
370 uint8_t sdSecurity() const { return (scr[1] >> 4) & 0X7; }
372 uint8_t sdBusWidths() const { return scr[1] & 0XF; }
374 bool sdSpec3() const { return scr[2] & 0X80; }
376 bool sdSpec4() const { return scr[2] & 0X4; }
379 uint8_t sdSpecX() const { return (scr[2] & 0X3) << 2 | scr[3] >> 6; }
381 uint8_t cmdSupport() const { return scr[3] & 0XF; }
383 int16_t sdSpecVer() const {
384 if (sdSpec() > 2) {
385 return -1;
386 } else if (sdSpec() < 2) {
387 return sdSpec() ? 110 : 101;
388 } else if (!sdSpec3()) {
389 return 200;
390 } else if (!sdSpec4() && !sdSpecX()) {
391 return 300;
392 }
393 return 400 + 100 * sdSpecX();
394 }
395};
