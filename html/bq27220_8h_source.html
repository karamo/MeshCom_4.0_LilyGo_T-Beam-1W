191 bool i2cReadBytes(uint8_t reg, uint8_t * dest, uint8_t count) {
192 wire->beginTransmission(addr);
193 wire->write(reg);
194 wire->endTransmission(true);
195
196 wire->requestFrom(addr, count);
197 for(int i = 0; i < count; i++) {
198 dest[i] = wire->read();
199 }
200 return true;
201 }

203 bool i2cWriteBytes(uint8_t reg, uint8_t * src, uint8_t count) {
204 wire->beginTransmission(addr);
205 wire->write(reg);
206 for(int i = 0; i < count; i++) {
207 wire->write(src[i]);
208 }
209 wire->endTransmission(true);
210 return true;
