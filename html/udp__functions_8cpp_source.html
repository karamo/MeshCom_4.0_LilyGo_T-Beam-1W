94void getMeshComUDPpacket(unsigned char inc_udp_buffer[UDP_TX_BUF_SIZE], int packetSize)
95{
96 char source_call[20] = {0};
97 char destination_call[20] = {0};
98
99 udp_is_busy = true;
100 // if more than n values are 00 we might have received a faulty message
101 uint8_t zerocount = 0;
102
103 for (int i = 0; i < packetSize; i+=2)
104 {
105 if (inc_udp_buffer[i] == 0x00 && inc_udp_buffer[i + 1] == 0x00)
106 {
107 zerocount += 2;
108 }
109 else
110 zerocount = 0;
111 }
112
113 //Serial.printf("zerocount:%i\n", zerocount);
114
115 if (zerocount <= MAX_ZEROS)
116 {
117 /* we now need to distinguish if we got a LoRa packet to send from the server
118 or it is a config message. First 4 Bytes indicate if it is
119 GATE: 0x47 41 54 45
120 CONF: 0x43 4F 4E 46
121 */
122
123 // get the first 4 bytes of the incoming udp message
124 char indicator_b[UDP_MSG_INDICATOR_LEN];
125
126 memcpy(indicator_b, inc_udp_buffer, UDP_MSG_INDICATOR_LEN);
127
128 char gate[] = "GATE";
129 char beat[] = "BEAT";
130
131 if (memcmp(indicator_b, gate, UDP_MSG_INDICATOR_LEN) == 0)
132 {
133 DEBUG_MSG("UDP", "Received a LoRa packet to transmit");
134
135 // Buffer filling
136 lora_tx_msg_len = packetSize - UDP_MSG_INDICATOR_LEN;
137 if (lora_tx_msg_len > UDP_TX_BUF_SIZE)
138 lora_tx_msg_len = UDP_TX_BUF_SIZE; // zur Sicherheit
139
140 // printout message type
141 uint8_t msg_type_b = inc_udp_buffer[UDP_MSG_INDICATOR_LEN];
142
143 switch (msg_type_b)
144 {
145 case 0x3A: DEBUG_MSG("UDP", "Received Textmessage"); break; // ':'
146 case 0x21: DEBUG_MSG("UDP", "Received PosInfo"); break; // '!'
147 case 0x40: DEBUG_MSG("UDP", "Received Hey"); break; // '@'
148 default: DEBUG_MSG("UDP", "Received unknown"); break;
149 }
150
151 if (msg_type_b == 0x3A || msg_type_b == 0x21 || msg_type_b == 0x40)
152 {
153 bool bBLELoopOut = true;
154
155 last_upd_timer = millis();
156
157 memcpy(convBuffer, inc_udp_buffer + UDP_MSG_INDICATOR_LEN, lora_tx_msg_len);
158
159 // send JSON to Extern IP
160 if(hasExternIPaddress)
161 {
162 if(bEXTUDP)
163 sendExtern(true, (char*)"udp", convBuffer, lora_tx_msg_len);
164 }
165
166 struct aprsMessage aprsmsg;
167
168 // print which message type we got
169 decodeAPRS(convBuffer, lora_tx_msg_len, aprsmsg);
170
171 snprintf(source_call, sizeof(source_call), "%s", aprsmsg.msg_source_call.c_str());
172 snprintf(destination_call, sizeof(destination_call), "%s", aprsmsg.msg_destination_call.c_str());
173
174 bool bUDPtoLoraSend = true;
175
176 if(msg_type_b == 0x21)
177 {
178 sendDisplayPosition(aprsmsg, 99, 0);
179
180 // add rcvMsg to forward to LoRa TX
181 addLoraRxBuffer(aprsmsg.msg_id, true);
182
183 if(bGATEWAY_NOPOS)
184 bUDPtoLoraSend=false;
185 }
186
187 // print aprs message
188 if(bDisplayInfo)
189 {
190 printBuffer_aprs((char*)"RX-UDP ", aprsmsg);
191 Serial.println();
192 }
193
194 bLED_ORANGE = true;
195
196 aprsmsg.msg_source_path.concat(',');
197 aprsmsg.msg_source_path.concat(meshcom_settings.node_call);
198
199 aprsmsg.msg_server = true;
200
201 aprsmsg.msg_last_hw = BOARD_HARDWARE | 0x80; // hardware last sending node
202 aprsmsg.msg_source_mod = (getMOD() & 0xF) | (meshcom_settings.node_country << 4); // modulation & country
203
204 memset(convBuffer, 0x00, UDP_TX_BUF_SIZE);
205
206 uint16_t size = encodeAPRS(convBuffer, aprsmsg);
207
208 if(size > UDP_TX_BUF_SIZE)
209 size = UDP_TX_BUF_SIZE;
210
211 if(msg_type_b == 0x3A)
212 {
213 if(memcmp(aprsmsg.msg_payload.c_str(), "{SET}", 5) == 0)
214 {
215 sendDisplayText(aprsmsg, 99, 0);
216 }
217 else
218 if(memcmp(aprsmsg.msg_payload.c_str(), "{CET}", 5) == 0)
219 {
220 sendDisplayText(aprsmsg, 99, 0);
221 }
222 else
223 if((strcmp(destination_call, "*") == 0 && !bNoMSGtoALL) || strcmp(destination_call, meshcom_settings.node_call) == 0 || CheckGroup(destination_call) > 0)
224 {
225 // wenn eine Meldung via UDP kommt und den eigene Node betrifft dann keine weiterleitung an LoRa TX
226 if(strcmp(destination_call, meshcom_settings.node_call) == 0)
227 bUDPtoLoraSend=false;
228
229 unsigned int iAckId = 0;
230
231 int iAckPos=aprsmsg.msg_payload.indexOf(":ack");
232 int iRefPos=aprsmsg.msg_payload.indexOf(":rej");
233 int iEnqPos=aprsmsg.msg_payload.indexOf("{", 1);
234
235 if(strcmp(destination_call, "*") == 0)
236 {
237 iAckPos=0;
238 iRefPos=0;
239 iEnqPos=0;
240 }
241
242 if(iAckPos > 0 || iRefPos > 0)
243 {
244 unsigned int iAckId = (aprsmsg.msg_payload.substring(iAckPos+4)).toInt();
245 msg_counter = ((_GW_ID & 0x3FFFFF) << 10) | (iAckId & 0x3FF);
246
247 uint8_t print_buff[30];
248
249 print_buff[0]=0x41;
250 print_buff[1]=msg_counter & 0xFF;
251 print_buff[2]=(msg_counter >> 8) & 0xFF;
252 print_buff[3]=(msg_counter >> 16) & 0xFF;
253 print_buff[4]=(msg_counter >> 24) & 0xFF;
254 print_buff[5]=0x01; // ACK
255 print_buff[6]=0x00;
256
257 int iackcheck = checkOwnTx(msg_counter);
258 if(iackcheck >= 0)
259 {
260 own_msg_id[iackcheck][4] = 0x02; // 02...ACK
261 print_buff[5]=0x02; // 02...ACK
262 }
263
264 if(bDisplayInfo)
265 Serial.printf("[UDP-MSGID] ack_msg_id:%02X%02X%02X%02X ACK...%02X\n", print_buff[4], print_buff[3], print_buff[2], print_buff[1], print_buff[5]);
266
267 addBLEOutBuffer(print_buff, 7);
268
269 if(strcmp(source_call, meshcom_settings.node_call) == 0)
270 bUDPtoLoraSend=false;
271
272 bBLELoopOut=false;
273 }
274 if(iEnqPos > 0)
275 {
276 iAckId = (aprsmsg.msg_payload.substring(iEnqPos+1)).toInt();
277 aprsmsg.msg_payload = aprsmsg.msg_payload.substring(0, iEnqPos);
278 }
279
280 if(iAckPos <= 0)
281 {
282 sendDisplayText(aprsmsg, 99, 0);
283 }
284
285 aprsmsg.max_hop = aprsmsg.max_hop | 0x20; // msg_app_offline true
286
287 uint8_t tempRcvBuffer[UDP_TX_BUF_SIZE];
288
289 aprsmsg.msg_last_hw = BOARD_HARDWARE | 0x80; // hardware last sending node
290 aprsmsg.msg_source_mod = (getMOD() & 0xF) | (meshcom_settings.node_country << 4); // modulation & country
291
292 uint16_t tempsize = encodeAPRS(tempRcvBuffer, aprsmsg);
293
294 addBLEOutBuffer(tempRcvBuffer, tempsize);
295
296 bBLELoopOut=false;
297
298 // DM message for lokal Node
299 if(iAckId > 0)
300 {
301 strSource_call = source_call;
302 SendAckMessage(strSource_call, iAckId);
303 }
304 }
305 }
306
307 // first byte is always the len of the msg
308 // UDP messages send to LoRa TX
309 // resend only Packet to all
310 if(bUDPtoLoraSend)
311 {
312 // store last message to compare later on
313 insertOwnTx(aprsmsg.msg_id);
314
315 ringBuffer[iWrite][0] = size;
316 if (msg_type_b == 0x3A) // only Messages
317 {
318 if(aprsmsg.msg_payload.startsWith("{") > 0)
319 ringBuffer[iWrite][1] = 0xFF; // retransmission Status ...0xFF no retransmission on {CET} & Co.
320 else
321 ringBuffer[iWrite][1] = 0x00; // retransmission Status ...0xFF no retransmission
322 }
323 else
324 ringBuffer[iWrite][1] = 0xFF; // retransmission Status ...0xFF no retransmission
325 memcpy(ringBuffer[iWrite] + 2, convBuffer, size);
326
327 addRingPointer(iWrite, iRead, MAX_RING);
328
329 /*
330 iWrite++;
331 if (iWrite >= MAX_RING) // if the buffer is full we start at index 0 -> take care of overwriting!
332 iWrite = 0;
333 */
334
335 addLoraRxBuffer(aprsmsg.msg_id, true);
336
337 // add rcvMsg to BLE out Buff
338 // size message is int -> uint16_t buffer size
339 if(isPhoneReady == 1 && bBLELoopOut) // wird schon vorher abgehandelt
340 {
341 addBLEOutBuffer(convBuffer, size);
342 }
343 }
344 }
345
346 // zero out the inc buffer
347 memset(inc_udp_buffer, 0, UDP_TX_BUF_SIZE);
348
349 udp_is_busy = false; //setting the busy flag
350
351 return;
352 }
353 // Heartbeat from Server
354 else if (memcmp(indicator_b, beat, UDP_MSG_INDICATOR_LEN) == 0)
355 {
356
357 // we got an heartbeat from server which we use to check connection (saving time we got it)
358 DEBUG_MSG("BEAT", "Heartbeat from server received");
359
365 last_upd_timer = millis();
366 }
367 else
368 {
369 DEBUG_MSG("ERROR", "Received udp message without indicator");
370 last_upd_timer = millis();
371 }
372 }
373 else
374 {
375 DEBUG_MSG("ERROR", "UDP Message has too much Zeros");
376 resetMeshComUDP();
377 }
378
379 udp_is_busy = false; //setting the busy flag
380}

384void sendMeshComUDP()
385{
386 if(bWIFIAP)
387 return;
388
389 if(!hasIPaddress)
390 return;
391
392 if(udpWrite != udpRead)
393 {
394 if(!udp_is_busy)
395 {
396 uint16_t msg_len = (uint16_t)ringBufferUDPout[udpRead][0];
397
398 // send it over UDP
399
400 Udp.beginPacket(node_hostip , UDP_PORT);
401
402 if (!Udp.write(ringBufferUDPout[udpRead] + 1, msg_len))
403 {
404 if(bDisplayCont)
405 Serial.println("[ERROR]...Sending UDP Packet failed");
406
407 err_cnt_udp_tx++;
408 // if we have too much errors sending, reset UDP
409 if (err_cnt_udp_tx >= MAX_ERR_UDP_TX)
410 {
411 // avoid TX and UDP
412 hasIPaddress = false;
413 meshcom_settings.node_hasIPaddress = hasIPaddress;
414 //cmd_counter = 50;
415
416 if(bDisplayCont)
417 Serial.println("[ERROR]...resetMeshComUDP");
418
419 err_cnt_udp_tx = 0;
420
421 resetMeshComUDP();
422 }
423 }
424
425 Udp.endPacket();
426
427 memcpy(convBuffer, ringBufferUDPout[udpRead] + 1 + 36, msg_len);
428
429 if(convBuffer[0] == 0x3A || convBuffer[0] == 0x21 || convBuffer[0] == 0x40)
430 {
431 struct aprsMessage aprsmsg;
432
433 // print which message type we got
434 decodeAPRS(convBuffer, msg_len, aprsmsg);
435
436 // print aprs message
437 if(bDisplayInfo)
438 {
439 printBuffer_aprs((char*)"TX-UDP ", aprsmsg);
440 Serial.println("");
441 }
442 }
443
444 // zero out sent buffer
445 memset(ringBufferUDPout[udpRead], 0, UDP_TX_BUF_SIZE);
446
447 udpRead++;
448 if (udpRead >= MAX_RING_UDP)
449 udpRead = 0;
450
451 }
452 else
453 {
454 DEBUG_MSG("UDP", "UDP busy. Sending asap");
455 }
456 }
457}
