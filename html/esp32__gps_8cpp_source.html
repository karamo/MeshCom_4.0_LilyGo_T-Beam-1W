463bool checkGPS(uint32_t Baudrate)
464{
465 Serial.printf("GPS: trying %u baud <%i>\n", Baudrate, maxStateCount);

481unsigned int getGPS(void)
482{
483 if(!bGPSON)
484 {
485 if(meshcom_settings.node_postime > 0)
486 return (unsigned long)meshcom_settings.node_postime;
487
488 posinfo_fix = false;
489 posinfo_satcount = 0;
490 posinfo_hdop = 0;
491
492 return POSINFO_INTERVAL;
493 }
494
495 if(bMitHardReset)
496 {
497 Serial.print("===== STATE ");
498 Serial.println(state);
499 }
500
501 #if defined (BOARD_TRACKER)
502 if(GPS.available())
503 {
504 return readGPS();
505 }
506
507 GPS.begin(GPS_BAUDRATE_MODUL, SERIAL_8N1, GPS_TX_PIN, GPS_RX_PIN);
508
509 return POSINFO_INTERVAL;
510 #endif
511
512 #if defined GPS_BAUDRATE
513 gpsBaudrate = GPS_BAUDRATE;
514 #endif
515
516 bool bw=true;
517
518 switch (state)
519 {
520 case 0: // auto-baud connection, then switch to 38400 and save config
521
522 while(bw)
523 {
524 if (checkGPS(gpsBaudrate)) //priorisierte Baudrate zuerst testen
525 {
526 bw=false;
527 }
528 else
529 if (checkGPS(9600))
530 {
531 bw=false;
532 }
533 else
534 if (checkGPS(38400))
535 {
536 bw=false;
537 }
538 else
539 if (checkGPS(57600))
540 {
541 bw=false;
542 }
543 else
544 if (checkGPS(115200))
545 {
546 bw=false;
547 }
548
549 delay(200); //Wait a bit before trying again to limit the Serial output flood
550 maxStateCount++;
551
552 if(maxStateCount > 10)
553 return POSINFO_INTERVAL;
554 }
555
556 if(bGPSON)
557 {
558 myGPS.setUART1Output(COM_TYPE_NMEA); //Set the UART port to output NMEA only
559 delay(100);
560 myGPS.enableNMEAMessage(UBX_NMEA_GLL, COM_PORT_UART1);
561 delay(100);
562 myGPS.enableNMEAMessage(UBX_NMEA_GSA, COM_PORT_UART1);
563 delay(100);
564 myGPS.enableNMEAMessage(UBX_NMEA_RMC, COM_PORT_UART1);
565 delay(100);
566 myGPS.enableNMEAMessage(UBX_NMEA_VTG, COM_PORT_UART1);
567 delay(100);
568 myGPS.enableNMEAMessage(UBX_NMEA_GGA, COM_PORT_UART1);
569 delay(100);
570 myGPS.setMeasurementRate(1000);
571 delay(100);
572 myGPS.saveConfiguration(); //Save the current settings to flash and BBR
573 delay(100);
574
575 if(gpsBaudrate != GPS_BAUDRATE)
576 {
577 myGPS.setSerialRate(GPS_BAUDRATE, COM_PORT_UART1);
578
579 GPS.end();
580 delay(100);
581 GPS.begin(GPS_BAUDRATE);
582 delay(100);
583
584 myGPS.saveConfiguration();
585 delay(100);
586 }
587
588 Serial.println("GPS serial connected, saved config");
589
590 state++;
591 }
592
593 break;
594
595 case 1: // hardReset, expect to see GPS back at 38400 baud
596 if(bMitHardReset)
597 {
598 Serial.println("Issuing hardReset (cold start)");
599
600 myGPS.hardReset();
601 delay(3000);
602 GPS.begin(gpsBaudrate);
603
604 if (myGPS.begin(GPS))
605 {
606 Serial.println("Success.");
607 }
608 else
609 {
610 Serial.printf("*** GPS did not respond at %i baud, starting over.\n", gpsBaudrate);
611 state = 2;
612 //bMitHardReset=false;
613 break;
614 }
615 }
616
617 state++;
618
619 break;
