95static size_t searchPair16(const pair16_t* table, size_t size, uint16_t key) {
96 size_t left = 0;
97 size_t right = size;
98 while (right - left > 1) {
99 size_t mid = left + (right - left) / 2;
100 if (readTable16(table[mid].key) <= key) {
101 left = mid;
102 } else {
103 right = mid;
104 }
105 }
106 return left;
107}

109uint16_t toUpcase(uint16_t chr) {
110 uint16_t i, first;
111 // Optimize for simple ASCII.
112 if (chr < 127) {
113 return chr - ('a' <= chr && chr <= 'z' ? 'a' - 'A' : 0);
114 }
115 i = searchPair16(reinterpret_cast<const pair16_t*>(mapTable), MAP_DIM, chr);
116 first = readTable16(mapTable[i].base);
117 if (first <= chr && (chr - first) < readTable8(mapTable[i].count)) {
118 int8_t off = readTable8(mapTable[i].off);
119 if (off == 1) {
120 return chr - ((chr - first) & 1);
121 }
122 return chr + (off ? off : -0x1C60);
123 }
124 i = searchPair16(lookupTable, LOOKUP_DIM, chr);
125 if (readTable16(lookupTable[i].key) == chr) {
126 return readTable16(lookupTable[i].val);
127 }
128 return chr;
129}

131uint32_t upcaseChecksum(uint16_t uc, uint32_t sum) {
132 sum = (sum << 31) + (sum >> 1) + (uc & 0XFF);
133 sum = (sum << 31) + (sum >> 1) + (uc >> 8);
134 return sum;
135}
