60bool keypad_init(int address)
61{
62 if(!i2cIsInit(0)){
63 Wire.begin(BOARD_KEYBOARD_SDA, BOARD_KEYBOARD_SCL);
64 Wire.beginTransmission(address);
65 Wire.endTransmission(true);
66 }
67
68 if (!keypad.begin(address, &Wire)) {
69 // Serial.println("keypad not found, check wiring & pullups!");
70 log_e("keypad not found, check wiring & pullups!");
71 return false;
72 }
73
74 // configure the size of the keypad matrix.
75 // all other pins will be inputs
76 keypad.matrix(KEYPAD_ROWS, KEYPAD_COLS);
77
78 // flush the internal buffer
79 keypad.flush();
80
81 return true;
82}

110void keypad_loop(void)
111{
112 char c = -1;
113 int state = -1;
114 int row, col;
115 int k = keypad.getEvent();
116 int v = keypad.available();
117
118 if(k >=KEYPAD_RELEASE_VAL_MIN && k <= KEYPAD_RELEASE_VAL_MAX)
119 { // release event
120 k = k - KEYPAD_RELEASE_VAL_MIN;
121 state = KEYPAD_RELEASE;
122 }
123
124 if(k >=KEYPAD_PRESS_VAL_MIN && k <= KEYPAD_PRESS_VAL_MAX)
125 { // press event
126 k = k - KEYPAD_PRESS_VAL_MIN;
127 state = KEYPAD_PRESS;
128 }
129
130 if(state != -1)
131 {
132 row = k / KEYPAD_COLS;
133 col = (KEYPAD_COLS-1) - k % KEYPAD_COLS;
134
135 switch (ikeypad_layer)
136 {
137 case 1:
138 c = keymap1[row][col];
139 break;
140
141 case 2:
142 c = keymap2[row][col];
143 break;
144
145 case 3:
146 c = keymap3[row][col];
147 break;
148
149 default:
150 c = keymap0[row][col];
151 break;
152 }
153
154 if(bTDECKDEBUG)
155 Serial.printf("state=%d k=%d, v=%d, layer:%i/%i press:%d, %d, %c\n", state, k, v, ikeypad_layer, ikeypad_layer_save, row, col, c);
156
157 keypad_curr_val = 0x00;
158
159 if(state == 1)
160 {
161 // CAP char
162 if(c == 0xFC)
163 {
164 if(ikeypad_layer == 1)
165 {
166 ikeypad_layer = 0;
167 c=0xF0;
168 }
169 else
170 {
171 ikeypad_layer = 1;
172 c=0xF1;
173 }
174
175 ikeypad_layer_save = ikeypad_layer;
176 }
177 else
178 // SYM
179 if(c == 0xFF)
180 {
181 if(ikeypad_layer == 2)
182 {
183 ikeypad_layer = 0;
184 c=0xF0;
185 }
186 else
187 {
188 ikeypad_layer = 2;
189 c=0xF2;
190 }
191
192 ikeypad_layer_save = ikeypad_layer;
193 }
194 else
195 // ALT
196 if(c == 0xFE)
197 {
198 ikeypad_layer = 3;
199 c=0xF3;
200 }
201
202 if(keypad_listener && !meshcom_settings.node_keyboardlock && (c==0XF0 || c==0XF1 || c==0XF2))
203 keypad_listener(state, c);
204 }
205 else
206 {
207 // state = 0
208 keypad_curr_val = c;
209 keypad_state = state;
210 keypad_update = true;
211
212
213 bool bNORM = true;
214
215 // spezial Keys
216 if(ikeypad_layer == 3)
217 {
218 if(c == 'T' || c == 't') // ALT + T
219 {
220 c = 0xF9;
221
222 bNORM = true;
223 }
224 else
225 if(c == 'R' || c == 'r') // ALT + R
226 {
227 c = 0xFA;
228
229 bNORM = true;
230 }
231 else
232 if(c == 'G' || c == 'g') // ALT + G
233 {
234 bGPSON = ! bGPSON;
235
236 bNORM = true;
237 }
238 else
239 if(c == 'D' || c == 'd') // ALT + D
240 {
241 bTDECKDEBUG = ! bTDECKDEBUG;
242
243 bNORM = false;
244 }
245 else
246 if(c == 'L' || c == 'l') // ALT + L
247 {
248 meshcom_settings.node_backlightlock = !meshcom_settings.node_backlightlock;
249
250 if(meshcom_settings.node_backlightlock)
251 digitalWrite(BOARD_KEYBOARD_LED, HIGH);
252 else
253 digitalWrite(BOARD_KEYBOARD_LED, LOW);
254
255 //if(!meshcom_settings.node_backlightlock)
256 // tft_off();
257
258 bNORM = false;
259 }
260 else
261 if(c == 'K' || c == 'k') // ALT + K
262 {
263 meshcom_settings.node_keyboardlock = !meshcom_settings.node_keyboardlock;
264
265 bNORM = false;
266 }
267 else
268 if(c == 'O' || c == 'o') // ALT + O
269 {
270 if (meshcom_settings.node_map > 0)
271 meshcom_settings.node_map--;
272
273 //set_map(meshcom_settings.node_map);
274
275 bNORM = false;
276 }
277 else
278 if(c == 'I' || c == 'i') // ALT + I
279 {
280 if (meshcom_settings.node_map < MAX_MAP-1)
281 meshcom_settings.node_map++;
282
283 //set_map(meshcom_settings.node_map);
284
285 bNORM = false;
286 }
287 else
288 if((c == 'B' || c == 'b') && (!meshcom_settings.node_keyboardlock)) // ALT + B
289 {
290 //cycleBrightness(); // TODO
291 bNORM = false;
292 }
293 else
294 if((c == 'M' || c == 'm') && (!meshcom_settings.node_keyboardlock)) // ALT + M
295 {
296 meshcom_settings.node_mute = !meshcom_settings.node_mute;
297 bNORM = false;
298
299 }
300
301 ikeypad_layer = ikeypad_layer_save;
302 }
303
304
305 if(keypad_listener && bNORM && !meshcom_settings.node_keyboardlock)
306 keypad_listener(state, c);
307 }
308
309 }
