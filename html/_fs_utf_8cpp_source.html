30char* cpToMb(uint32_t cp, char* str, const char* end) {
31 size_t n = end - str;
32 if (cp < 0X80) {
33 if (n < 1) goto fail;
34 *(str++) = static_cast<uint8_t>(cp);
35 } else if (cp < 0X800) {
36 if (n < 2) goto fail;
37 *(str++) = static_cast<uint8_t>((cp >> 6) | 0XC0);
38 *(str++) = static_cast<uint8_t>((cp & 0X3F) | 0X80);
39 } else if (cp < 0X10000) {
40 if (n < 3) goto fail;
41 *(str++) = static_cast<uint8_t>((cp >> 12) | 0XE0);
42 *(str++) = static_cast<uint8_t>(((cp >> 6) & 0X3F) | 0X80);
43 *(str++) = static_cast<uint8_t>((cp & 0X3F) | 0X80);
44 } else {
45 if (n < 4) goto fail;
46 *(str++) = static_cast<uint8_t>((cp >> 18) | 0XF0);
47 *(str++) = static_cast<uint8_t>(((cp >> 12) & 0X3F) | 0X80);
48 *(str++) = static_cast<uint8_t>(((cp >> 6) & 0X3F) | 0X80);
49 *(str++) = static_cast<uint8_t>((cp & 0X3F) | 0X80);
50 }
51 return str;
52
53fail:
54 return nullptr;
55}

58const char* mbToCp(const char* str, const char* end, uint32_t* rtn) {
59 size_t n;
60 uint32_t cp;
61 if (str >= end) {
62 return nullptr;
63 }
64 uint8_t ch = str[0];
65 if ((ch & 0X80) == 0) {
66 *rtn = ch;
67 return str + 1;
68 }
69 if ((ch & 0XE0) == 0XC0) {
70 cp = ch & 0X1F;
71 n = 2;
72 } else if ((ch & 0XF0) == 0XE0) {
73 cp = ch & 0X0F;
74 n = 3;
75 } else if ((ch & 0XF8) == 0XF0) {
76 cp = ch & 0X07;
77 n = 4;
78 } else {
79 return nullptr;
80 }
81 if ((str + n) > end) {
82 return nullptr;
83 }
84 for (size_t i = 1; i < n; i++) {
85 ch = str[i];
86 if ((ch & 0XC0) != 0X80) {
87 return nullptr;
88 }
89 cp <<= 6;
90 cp |= ch & 0X3F;
91 }
92 // Don't allow over long as ASCII.
93 if (cp < 0X80 || !isValidCp(cp)) {
94 return nullptr;
95 }
96 *rtn = cp;
97 return str + n;
98}

100const char* mbToU16(const char* str, const char* end, uint16_t* hs,
101 uint16_t* ls) {
102 uint32_t cp;
103 const char* ptr = mbToCp(str, end, &cp);
104 if (!ptr) {
105 return nullptr;
106 }
107 if (cp <= 0XFFFF) {
108 *hs = cp;
109 *ls = 0;
110 } else {
111 *hs = highSurrogate(cp);
112 *ls = lowSurrogate(cp);
113 }

115}
