13void BaseDisplay::drawPixel(int16_t x, int16_t y, uint16_t color) {
14 // Rotate the pixel
15 int16_t x1=0, y1=0;
16 switch(rotation) {
17 case 0: // No rotation
18 x1=x;
19 y1=y;
20 break;
21 case 1: // 90deg clockwise
22 x1 = (drawing_width - 1) - y;
23 y1 = x;
24 break;
25 case 2: // 180deg
26 x1 = (drawing_width - 1) - x;
27 y1 = (drawing_height - 1) - y;
28 break;
29 case 3: // 270deg clockwise
30 x1 = y;
31 y1 = (drawing_height - 1) - x;
32 break;
33 }
34 x = x1;
35 y = y1;
36
37 // Handle flip
38 if (imgflip & Flip::HORIZONTAL) {
39 if (rotation % 2) // If landscape
40 y = (drawing_height - 1) - y;
41 else // If portrait
42 x = (drawing_width - 1) - x;
43 }
44 if (imgflip & Flip::VERTICAL) {
45 if (rotation % 2) // If landscape
46 x = (drawing_width - 1) - x;
47 else // If portrait
48 y = (drawing_height - 1) - y;
49 }
50
51 // Check if pixel falls in our page
52 if((uint16_t) x >= winrot_left && (uint16_t) y >= page_top && (uint16_t) y <= page_bottom && (uint16_t) x <= winrot_right) {
53
54 uint16_t byte_offset;
55 uint8_t bit_offset;
56 calculatePixelPageOffset(x, y, byte_offset, bit_offset); // Position of pixel within the page files. Overriden if no "partial window" support
57
58 // Insert the correct color values into the appropriate location
59 uint8_t bitmask = ~(1 << bit_offset);
60 page_black[byte_offset] &= bitmask;
61 page_black[byte_offset] |= (color & WHITE) << bit_offset;
62
63 // Red, if display supports
64 if (supportsColor(RED)) {
65 page_red[byte_offset] &= bitmask;
66 page_red[byte_offset] |= (color >> 1) << bit_offset;
67 }
68 }
69}

71// Where should the pixel be placed in the pagefile - overriden by derived display classes which do not support "partial window"
72void BaseDisplay::calculatePixelPageOffset(uint16_t x, uint16_t y, uint16_t &byte_offset, uint8_t &bit_offset) {
73 // Calculate a memory location (byte) for our pixel
74 byte_offset = (y - page_top) * ((winrot_right - winrot_left + 1) / 8);
75 byte_offset += ((x - winrot_left) / 8);
76 bit_offset = x % 8; // Find the location of the bit in which the value will be stored
77 bit_offset = (7 - bit_offset); // For some reason, the screen wants the bit order flipped. MSB vs LSB?
78}
