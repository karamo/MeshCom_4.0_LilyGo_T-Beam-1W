53bool ExFatFormatter::format(FsBlockDevice* dev, uint8_t* secBuf, print_t* pr) {
54#if !PRINT_FORMAT_PROGRESS
55 (void)pr;
56#endif // !PRINT_FORMAT_PROGRESS
57 MbrSector_t* mbr;
58 ExFatPbs_t* pbs;
59 DirUpcase_t* dup;
60 DirBitmap_t* dbm;
61 DirLabel_t* label;
62 uint32_t bitmapSize;
63 uint32_t checksum = 0;
64 uint32_t clusterCount;
65 uint32_t clusterHeapOffset;
66 uint32_t fatLength;
67 uint32_t fatOffset;
68 uint32_t m;
69 uint32_t ns;
70 uint32_t partitionOffset;
71 uint32_t sector;
72 uint32_t sectorsPerCluster;
73 uint32_t volumeLength;
74 uint32_t sectorCount;
75 uint8_t sectorsPerClusterShift;
76 uint8_t vs;
77
78 m_dev = dev;
79 m_secBuf = secBuf;
80 sectorCount = dev->sectorCount();
81 // Min size is 512 MB
82 if (sectorCount < 0X100000) {
83 writeMsg(pr, "Device is too small\r\n");
84 DBG_FAIL_MACRO;
85 goto fail;
86 }
87 // Determine partition layout.
88 for (m = 1, vs = 0; m && sectorCount > m; m <<= 1, vs++) {
89 }
90 sectorsPerClusterShift = vs < 29 ? 8 : (vs - 11) / 2;
91 sectorsPerCluster = 1UL << sectorsPerClusterShift;
92 fatLength = 1UL << (vs < 27 ? 13 : (vs + 1) / 2);
93 fatOffset = fatLength;
94 partitionOffset = 2 * fatLength;
95 clusterHeapOffset = 2 * fatLength;
96 clusterCount = (sectorCount - 4 * fatLength) >> sectorsPerClusterShift;
97 volumeLength = clusterHeapOffset + (clusterCount << sectorsPerClusterShift);
98
99 // make Master Boot Record. Use fake CHS.
100 memset(secBuf, 0, BYTES_PER_SECTOR);
101 mbr = reinterpret_cast<MbrSector_t*>(secBuf);
102 mbr->part->beginCHS[0] = 1;
103 mbr->part->beginCHS[1] = 1;
104 mbr->part->beginCHS[2] = 0;
105 mbr->part->type = 7;
106 mbr->part->endCHS[0] = 0XFE;
107 mbr->part->endCHS[1] = 0XFF;
108 mbr->part->endCHS[2] = 0XFF;
109 setLe32(mbr->part->relativeSectors, partitionOffset);
110 setLe32(mbr->part->totalSectors, volumeLength);
111 setLe16(mbr->signature, MBR_SIGNATURE);
112 if (!dev->writeSector(0, secBuf)) {
113 DBG_FAIL_MACRO;
114 goto fail;
115 }
116 // Partition Boot sector.
117 memset(secBuf, 0, BYTES_PER_SECTOR);
118 pbs = reinterpret_cast<ExFatPbs_t*>(secBuf);
119 pbs->jmpInstruction[0] = 0XEB;
120 pbs->jmpInstruction[1] = 0X76;
121 pbs->jmpInstruction[2] = 0X90;
122 pbs->oemName[0] = 'E';
123 pbs->oemName[1] = 'X';
124 pbs->oemName[2] = 'F';
125 pbs->oemName[3] = 'A';
126 pbs->oemName[4] = 'T';
127 pbs->oemName[5] = ' ';
128 pbs->oemName[6] = ' ';
129 pbs->oemName[7] = ' ';
130 setLe64(pbs->bpb.partitionOffset, partitionOffset);
131 setLe64(pbs->bpb.volumeLength, volumeLength);
132 setLe32(pbs->bpb.fatOffset, fatOffset);
133 setLe32(pbs->bpb.fatLength, fatLength);
134 setLe32(pbs->bpb.clusterHeapOffset, clusterHeapOffset);
135 setLe32(pbs->bpb.clusterCount, clusterCount);
136 setLe32(pbs->bpb.rootDirectoryCluster, ROOT_CLUSTER);
137 setLe32(pbs->bpb.volumeSerialNumber, sectorCount);
138 setLe16(pbs->bpb.fileSystemRevision, 0X100);
139 setLe16(pbs->bpb.volumeFlags, 0);
140 pbs->bpb.bytesPerSectorShift = BYTES_PER_SECTOR_SHIFT;
141 pbs->bpb.sectorsPerClusterShift = sectorsPerClusterShift;
142 pbs->bpb.numberOfFats = 1;
143 pbs->bpb.driveSelect = 0X80;
144 pbs->bpb.percentInUse = 0;
145
146 // Fill boot code like official SDFormatter.
147 for (size_t i = 0; i < sizeof(pbs->bootCode); i++) {
148 pbs->bootCode[i] = 0XF4;
149 }
150 setLe16(pbs->signature, PBR_SIGNATURE);
151 for (size_t i = 0; i < BYTES_PER_SECTOR; i++) {
152 if (i == offsetof(ExFatPbs_t, bpb.volumeFlags[0]) ||
153 i == offsetof(ExFatPbs_t, bpb.volumeFlags[1]) ||
154 i == offsetof(ExFatPbs_t, bpb.percentInUse)) {
155 continue;
156 }
157 checksum = exFatChecksum(checksum, secBuf[i]);
158 }
159 sector = partitionOffset;
160 if (!dev->writeSector(sector, secBuf) ||
161 !dev->writeSector(sector + BOOT_BACKUP_OFFSET, secBuf)) {
162 DBG_FAIL_MACRO;
163 goto fail;
164 }
165 sector++;
166 // Write eight Extended Boot Sectors.
167 memset(secBuf, 0, BYTES_PER_SECTOR);
168 setLe16(pbs->signature, PBR_SIGNATURE);
169 for (int j = 0; j < 8; j++) {
170 for (size_t i = 0; i < BYTES_PER_SECTOR; i++) {
171 checksum = exFatChecksum(checksum, secBuf[i]);
172 }
173 if (!dev->writeSector(sector, secBuf) ||
174 !dev->writeSector(sector + BOOT_BACKUP_OFFSET, secBuf)) {
175 DBG_FAIL_MACRO;
176 goto fail;
177 }
178 sector++;
179 }
180 // Write OEM Parameter Sector and reserved sector.
181 memset(secBuf, 0, BYTES_PER_SECTOR);
182 for (int j = 0; j < 2; j++) {
183 for (size_t i = 0; i < BYTES_PER_SECTOR; i++) {
184 checksum = exFatChecksum(checksum, secBuf[i]);
185 }
186 if (!dev->writeSector(sector, secBuf) ||
187 !dev->writeSector(sector + BOOT_BACKUP_OFFSET, secBuf)) {
188 DBG_FAIL_MACRO;
189 goto fail;
190 }
191 sector++;
192 }
193 // Write Boot CheckSum Sector.
194 for (size_t i = 0; i < BYTES_PER_SECTOR; i += 4) {
195 setLe32(secBuf + i, checksum);
196 }
197 if (!dev->writeSector(sector, secBuf) ||
198 !dev->writeSector(sector + BOOT_BACKUP_OFFSET, secBuf)) {
199 DBG_FAIL_MACRO;
200 goto fail;
201 }
202 // Initialize FAT.
203 writeMsg(pr, "Writing FAT ");
204 sector = partitionOffset + fatOffset;
205 ns = ((clusterCount + 2) * 4 + BYTES_PER_SECTOR - 1) / BYTES_PER_SECTOR;
206
207 memset(secBuf, 0, BYTES_PER_SECTOR);
208 // Allocate two reserved clusters, bitmap, upcase, and root clusters.
209 secBuf[0] = 0XF8;
210 for (size_t i = 1; i < 20; i++) {
211 secBuf[i] = 0XFF;
212 }
213 for (uint32_t i = 0; i < ns; i++) {
214 if (i % (ns / 32) == 0) {
215 writeMsg(pr, ".");
216 }
217 if (!dev->writeSector(sector + i, secBuf)) {
218 DBG_FAIL_MACRO;
219 goto fail;
220 }
221 if (i == 0) {
222 memset(secBuf, 0, BYTES_PER_SECTOR);
223 }
224 }
225 writeMsg(pr, "\r\n");
226 // Write cluster two, bitmap.
227 sector = partitionOffset + clusterHeapOffset;
228 bitmapSize = (clusterCount + 7) / 8;
229 ns = (bitmapSize + BYTES_PER_SECTOR - 1) / BYTES_PER_SECTOR;
230 if (ns > sectorsPerCluster) {
231 DBG_FAIL_MACRO;
232 goto fail;
233 }
234 memset(secBuf, 0, BYTES_PER_SECTOR);
235 // Allocate clusters for bitmap, upcase, and root.
236 secBuf[0] = 0X7;
237 for (uint32_t i = 0; i < ns; i++) {
238 if (!dev->writeSector(sector + i, secBuf)) {
239 DBG_FAIL_MACRO;
240 goto fail;
241 }
242 if (i == 0) {
243 secBuf[0] = 0;
244 }
245 }
246 // Write cluster three, upcase table.
247 writeMsg(pr, "Writing upcase table\r\n");
248 if (!writeUpcase(partitionOffset + clusterHeapOffset + sectorsPerCluster)) {
249 DBG_FAIL_MACRO;
250 goto fail;
251 }
252 if (m_upcaseSize > BYTES_PER_SECTOR * sectorsPerCluster) {
253 DBG_FAIL_MACRO;
254 goto fail;
255 }
256 // Initialize first sector of root.
257 writeMsg(pr, "Writing root\r\n");
258 ns = sectorsPerCluster;
259 sector = partitionOffset + clusterHeapOffset + 2 * sectorsPerCluster;
260 memset(secBuf, 0, BYTES_PER_SECTOR);
261
262 // Unused Label entry.
263 label = reinterpret_cast<DirLabel_t*>(secBuf);
264 label->type = EXFAT_TYPE_LABEL & 0X7F;
265
266 // bitmap directory entry.
267 dbm = reinterpret_cast<DirBitmap_t*>(secBuf + 32);
268 dbm->type = EXFAT_TYPE_BITMAP;
269 setLe32(dbm->firstCluster, BITMAP_CLUSTER);
270 setLe64(dbm->size, bitmapSize);
271
272 // upcase directory entry.
273 dup = reinterpret_cast<DirUpcase_t*>(secBuf + 64);
274 dup->type = EXFAT_TYPE_UPCASE;
275 setLe32(dup->checksum, m_upcaseChecksum);
276 setLe32(dup->firstCluster, UPCASE_CLUSTER);
277 setLe64(dup->size, m_upcaseSize);
278
279 // Write root, cluster four.
280 for (uint32_t i = 0; i < ns; i++) {
281 if (!dev->writeSector(sector + i, secBuf)) {
282 DBG_FAIL_MACRO;
283 goto fail;
284 }
285 if (i == 0) {
286 memset(secBuf, 0, BYTES_PER_SECTOR);
287 }
288 }
289 writeMsg(pr, "Format done\r\n");
290 return true;
291
292fail:
293 writeMsg(pr, "Format failed\r\n");
294 return false;
295}

297bool ExFatFormatter::syncUpcase() {
298 uint16_t index = m_upcaseSize & SECTOR_MASK;
299 if (!index) {
300 return true;
301 }
302 for (size_t i = index; i < BYTES_PER_SECTOR; i++) {
303 m_secBuf[i] = 0;
304 }
305 return m_dev->writeSector(m_upcaseSector, m_secBuf);
306}
307//------------------------------------------------------------------------------
308bool ExFatFormatter::writeUpcaseByte(uint8_t b) {
309 uint16_t index = m_upcaseSize & SECTOR_MASK;
310 m_secBuf[index] = b;
311 m_upcaseChecksum = exFatChecksum(m_upcaseChecksum, b);
312 m_upcaseSize++;
313 if (index == SECTOR_MASK) {
314 return m_dev->writeSector(m_upcaseSector++, m_secBuf);
315 }
316 return true;
317}
318//------------------------------------------------------------------------------
319bool ExFatFormatter::writeUpcaseUnicode(uint16_t unicode) {
320 return writeUpcaseByte(unicode) && writeUpcaseByte(unicode >> 8);
321}
322//------------------------------------------------------------------------------
323bool ExFatFormatter::writeUpcase(uint32_t sector) {
324 uint32_t n;
325 uint32_t ns;
326 uint32_t ch = 0;
327 uint16_t uc;
328
329 m_upcaseSize = 0;
330 m_upcaseChecksum = 0;
331 m_upcaseSector = sector;
332
333 while (ch < 0X10000) {
334 uc = toUpcase(ch);
335 if (uc != ch) {
336 if (!writeUpcaseUnicode(uc)) {
337 DBG_FAIL_MACRO;
338 goto fail;
339 }
340 ch++;
341 } else {
342 for (n = ch + 1; n < 0X10000 && n == toUpcase(n); n++) {
343 }
344 ns = n - ch;
345 if (ns >= MINIMUM_UPCASE_SKIP) {
346 if (!writeUpcaseUnicode(0XFFFF) || !writeUpcaseUnicode(ns)) {
347 DBG_FAIL_MACRO;
348 goto fail;
349 }
350 ch = n;
351 } else {
352 while (ch < n) {
353 if (!writeUpcaseUnicode(ch++)) {
354 DBG_FAIL_MACRO;
355 goto fail;
356 }
357 }
358 }
359 }
360 }
361 if (!syncUpcase()) {
362 DBG_FAIL_MACRO;
363 goto fail;
364 }
365 return true;
366
367fail:
368 return false;
369}
