45//------------------------------------------------------------------------------
46/* Disable DMA Controller. */
47static void dmac_disable() { DMAC->DMAC_EN &= (~DMAC_EN_ENABLE); }
48/* Enable DMA Controller. */
49static void dmac_enable() { DMAC->DMAC_EN = DMAC_EN_ENABLE; }
50/* Disable DMA Channel. */
51static void dmac_channel_disable(uint32_t ul_num) {
52 DMAC->DMAC_CHDR = DMAC_CHDR_DIS0 << ul_num;
53}
54/* Enable DMA Channel. */
55static void dmac_channel_enable(uint32_t ul_num) {
56 DMAC->DMAC_CHER = DMAC_CHER_ENA0 << ul_num;
57}
58/* Poll for transfer complete. */
59static bool dmac_channel_transfer_done(uint32_t ul_num) {
60 return (DMAC->DMAC_CHSR & (DMAC_CHSR_ENA0 << ul_num)) ? false : true;
61}
62//------------------------------------------------------------------------------
63// start RX DMA
64static void spiDmaRX(uint8_t* dst, uint16_t count) {
65 dmac_channel_disable(SPI_DMAC_RX_CH);
66 DMAC->DMAC_CH_NUM[SPI_DMAC_RX_CH].DMAC_SADDR = (uint32_t)&SPI0->SPI_RDR;
67 DMAC->DMAC_CH_NUM[SPI_DMAC_RX_CH].DMAC_DADDR = (uint32_t)dst;
68 DMAC->DMAC_CH_NUM[SPI_DMAC_RX_CH].DMAC_DSCR = 0;
69 DMAC->DMAC_CH_NUM[SPI_DMAC_RX_CH].DMAC_CTRLA =
70 count | DMAC_CTRLA_SRC_WIDTH_BYTE | DMAC_CTRLA_DST_WIDTH_BYTE;
71 DMAC->DMAC_CH_NUM[SPI_DMAC_RX_CH].DMAC_CTRLB =
72 DMAC_CTRLB_SRC_DSCR | DMAC_CTRLB_DST_DSCR | DMAC_CTRLB_FC_PER2MEM_DMA_FC |
73 DMAC_CTRLB_SRC_INCR_FIXED | DMAC_CTRLB_DST_INCR_INCREMENTING;
74 DMAC->DMAC_CH_NUM[SPI_DMAC_RX_CH].DMAC_CFG =
75 DMAC_CFG_SRC_PER(SPI_RX_IDX) | DMAC_CFG_SRC_H2SEL | DMAC_CFG_SOD |
76 DMAC_CFG_FIFOCFG_ASAP_CFG;
77 dmac_channel_enable(SPI_DMAC_RX_CH);
78}
79//------------------------------------------------------------------------------
80// start TX DMA
81static void spiDmaTX(const uint8_t* src, uint16_t count) {
82 static uint8_t ff = 0XFF;
83 uint32_t src_incr = DMAC_CTRLB_SRC_INCR_INCREMENTING;
84 if (!src) {
85 src = &ff;
86 src_incr = DMAC_CTRLB_SRC_INCR_FIXED;
87 }
88 dmac_channel_disable(SPI_DMAC_TX_CH);
89 DMAC->DMAC_CH_NUM[SPI_DMAC_TX_CH].DMAC_SADDR = (uint32_t)src;
90 DMAC->DMAC_CH_NUM[SPI_DMAC_TX_CH].DMAC_DADDR = (uint32_t)&SPI0->SPI_TDR;
91 DMAC->DMAC_CH_NUM[SPI_DMAC_TX_CH].DMAC_DSCR = 0;
92 DMAC->DMAC_CH_NUM[SPI_DMAC_TX_CH].DMAC_CTRLA =
93 count | DMAC_CTRLA_SRC_WIDTH_BYTE | DMAC_CTRLA_DST_WIDTH_BYTE;
94
95 DMAC->DMAC_CH_NUM[SPI_DMAC_TX_CH].DMAC_CTRLB =
96 DMAC_CTRLB_SRC_DSCR | DMAC_CTRLB_DST_DSCR | DMAC_CTRLB_FC_MEM2PER_DMA_FC |
97 src_incr | DMAC_CTRLB_DST_INCR_FIXED;
98
99 DMAC->DMAC_CH_NUM[SPI_DMAC_TX_CH].DMAC_CFG =
100 DMAC_CFG_DST_PER(SPI_TX_IDX) | DMAC_CFG_DST_H2SEL | DMAC_CFG_SOD |
101 DMAC_CFG_FIFOCFG_ALAP_CFG;
102
103 dmac_channel_enable(SPI_DMAC_TX_CH);
104}

107void SdSpiArduinoDriver::activate() {
108 SPI.beginTransaction(m_spiSettings);
109
110 Spi* pSpi = SPI0;
111 // Save the divisor
112 uint32_t scbr = pSpi->SPI_CSR[SPI_CHIP_SEL] & 0XFF00;
113 // Disable SPI
114 pSpi->SPI_CR = SPI_CR_SPIDIS;
115 // reset SPI
116 pSpi->SPI_CR = SPI_CR_SWRST;
117 // no mode fault detection, set master mode
118 pSpi->SPI_MR = SPI_PCS(SPI_CHIP_SEL) | SPI_MR_MODFDIS | SPI_MR_MSTR;
119 // mode 0, 8-bit,
120 pSpi->SPI_CSR[SPI_CHIP_SEL] = scbr | SPI_CSR_CSAAT | SPI_CSR_NCPHA;
121 // enable SPI
122 pSpi->SPI_CR |= SPI_CR_SPIEN;
123}

160uint8_t SdSpiArduinoDriver::receive(uint8_t* buf, size_t count) {
161 Spi* pSpi = SPI0;
162 int rtn = 0;
163#if USE_SAM3X_DMAC
164 // clear overrun error
165 while (pSpi->SPI_SR & (SPI_SR_OVRES | SPI_SR_RDRF)) {
166 pSpi->SPI_RDR;
167 }
168 spiDmaRX(buf, count);
169 spiDmaTX(0, count);
170
171 uint32_t m = millis();
172 while (!dmac_channel_transfer_done(SPI_DMAC_RX_CH)) {
173 if ((millis() - m) > SAM3X_DMA_TIMEOUT) {
174 dmac_channel_disable(SPI_DMAC_RX_CH);
175 dmac_channel_disable(SPI_DMAC_TX_CH);
176 rtn = 2;
177 break;
178 }
179 }
180 if (pSpi->SPI_SR & SPI_SR_OVRES) {
181 rtn |= 1;
182 }
183#else // USE_SAM3X_DMAC
184 for (size_t i = 0; i < count; i++) {
185 pSpi->SPI_TDR = 0XFF;
186 while ((pSpi->SPI_SR & SPI_SR_RDRF) == 0) {
187 }
188 buf[i] = pSpi->SPI_RDR;
189 }
190#endif // USE_SAM3X_DMAC
191 return rtn;
192}

196void SdSpiArduinoDriver::send(const uint8_t* buf, size_t count) {
197 Spi* pSpi = SPI0;
198#if USE_SAM3X_DMAC
199 spiDmaTX(buf, count);
200 while (!dmac_channel_transfer_done(SPI_DMAC_TX_CH)) {
201 }
202#else // #if USE_SAM3X_DMAC
203 while ((pSpi->SPI_SR & SPI_SR_TXEMPTY) == 0) {
204 }
205 for (size_t i = 0; i < count; i++) {
206 pSpi->SPI_TDR = buf[i];
207 while ((pSpi->SPI_SR & SPI_SR_TDRE) == 0) {
208 }
209 }
210#endif // #if USE_SAM3X_DMAC
211 while ((pSpi->SPI_SR & SPI_SR_TXEMPTY) == 0) {
212 }
213 // leave RDR empty
214 while (pSpi->SPI_SR & (SPI_SR_OVRES | SPI_SR_RDRF)) {
215 pSpi->SPI_RDR;
216 }
217}
