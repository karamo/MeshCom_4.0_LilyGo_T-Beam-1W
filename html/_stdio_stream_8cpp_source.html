61char* StdioStream::fgets(char* str, size_t num, size_t* len) {
62 char* s = str;
63 if (num-- == 0) {
64 return 0;
65 }
66 while (num) {
67 size_t n;
68 if ((n = m_r) == 0) {
69 if (!fillBuf()) {
70 if (s == str) {
71 return 0;
72 }
73 break;
74 }
75 n = m_r;
76 }
77 if (n > num) {
78 n = num;
79 }
80 uint8_t* end = reinterpret_cast<uint8_t*>(memchr(m_p, '\n', n));
81 if (end != 0) {
82 n = ++end - m_p;
83 memcpy(s, m_p, n);
84 m_r -= n;
85 m_p = end;
86 s += n;
87 break;
88 }
89 memcpy(s, m_p, n);
90 m_r -= n;
91 m_p += n;
92 s += n;
93 num -= n;
94 }
95 *s = 0;
96 if (len) {
97 *len = s - str;
98 }
99 return str;
100}

102bool StdioStream::fopen(const char* path, const char* mode) {
103 oflag_t oflag;
104 uint8_t m;
105 switch (*mode++) {
106 case 'a':
107 m = O_WRONLY;
108 oflag = O_CREAT | O_APPEND;
109 m_status = S_SWR;
110 break;
111
112 case 'r':
113 m = O_RDONLY;
114 oflag = 0;
115 m_status = S_SRD;
116 break;
117
118 case 'w':
119 m = O_WRONLY;
120 oflag = O_CREAT | O_TRUNC;
121 m_status = S_SWR;
122 break;
123
124 default:
125 goto fail;
126 }
127 while (*mode) {
128 switch (*mode++) {
129 case '+':
130 m_status = S_SRW;
131 m = O_RDWR;
132 break;
133
134 case 'b':
135 break;
136
137 case 'x':
138 oflag |= O_EXCL;
139 break;
140
141 default:
142 goto fail;
143 }
144 }
145 oflag |= m;
146 if (!StreamBaseFile::open(path, oflag)) {
147 goto fail;
148 }
149 m_r = 0;
150 m_w = 0;
151 m_p = m_buf;
152 return true;
153
154fail:
155 m_status = 0;
156 return false;
157}

164size_t StdioStream::fread(void* ptr, size_t size, size_t count) {
165 uint8_t* dst = reinterpret_cast<uint8_t*>(ptr);
166 size_t total = size * count;
167 if (total == 0) {
168 return 0;
169 }
170 size_t need = total;
171 while (need > m_r) {
172 memcpy(dst, m_p, m_r);
173 dst += m_r;
174 m_p += m_r;
175 need -= m_r;
176 if (!fillBuf()) {
177 return (total - need) / size;
178 }
179 }
180 memcpy(dst, m_p, need);
181 m_r -= need;
182 m_p += need;
183 return count;
184}

187 int32_t pos;

202 }
203 break;
204

208 }
209 break;
210
211 case SEEK_END:

219 }
220 m_r = 0;
221 m_p = m_buf;
222 return 0;
223
224fail:
225 return EOF;
226}

229 uint32_t pos = StreamBaseFile::curPosition();
230 if (m_status & S_SRD) {
231 if (m_r > pos) {
232 return -1L;
233 }
234 pos -= m_r;
235 } else if (m_status & S_SWR) {
236 pos += m_p - m_buf;
237 }
238 return pos;

245int StdioStream::write(const void* buf, size_t count) {
246 const uint8_t* src = static_cast<const uint8_t*>(buf);
247 size_t todo = count;
248
249 while (todo > m_w) {
250 memcpy(m_p, src, m_w);
251 m_p += m_w;
252 src += m_w;
253 todo -= m_w;
254 if (!flushBuf()) {
255 return EOF;
256 }
257 }
258 memcpy(m_p, src, todo);
259 m_p += todo;
260 m_w -= todo;
261 return count;
262}

359int StdioStream::ungetc(int c) {
360 // error if EOF.
361 if (c == EOF) {
362 return EOF;
363 }
364 // error if not reading.
365 if ((m_status & S_SRD) == 0) {
366 return EOF;
367 }

373 m_status &= ~S_EOF;
374 return *--m_p = (uint8_t)c;

388bool StdioStream::fillBuf() {
389 if (!(m_status & S_SRD)) { // check for S_ERR and S_EOF ??/////////////////
390 if (!(m_status & S_SRW)) {
391 m_status |= S_ERR;
392 return false;
393 }
394 if (m_status & S_SWR) {
395 if (!flushBuf()) {
396 return false;
397 }
398 m_status &= ~S_SWR;
399 m_status |= S_SRD;
400 m_w = 0;
401 }
402 }
403 m_p = m_buf + UNGETC_BUF_SIZE;
404 int nr = StreamBaseFile::read(m_p, sizeof(m_buf) - UNGETC_BUF_SIZE);
405 if (nr <= 0) {
406 m_status |= nr < 0 ? S_ERR : S_EOF;
407 m_r = 0;
408 return false;
409 }
410 m_r = nr;
411 return true;
412}

415bool StdioStream::flushBuf() {
416 if (!(m_status & S_SWR)) {
417 if (!(m_status & S_SRW)) {
418 m_status |= S_ERR;
419 return false;
420 }
421 m_status &= ~S_SRD;
422 m_status |= S_SWR;
423 m_r = 0;
424 m_w = sizeof(m_buf);
425 m_p = m_buf;
426 return true;
427 }
428 uint8_t n = m_p - m_buf;
429 m_p = m_buf;
430 m_w = sizeof(m_buf);
431 if (StreamBaseFile::write(m_buf, n) == n) {
432 return true;
433 }
434 m_status |= S_ERR;
435 return false;
436}
