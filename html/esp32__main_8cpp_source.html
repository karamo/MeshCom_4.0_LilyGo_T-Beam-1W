440unsigned int getMacAddr(void)
441{
442 uint64_t dmac64 = ESP.getEfuseMac();
443
444 dmac[0] = dmac64 >> 40;
445 dmac[1] = dmac64 >> 32;
446 dmac[2] = dmac64 >> 24;
447 dmac[3] = dmac64 >> 16;
448 dmac[4] = dmac64 >> 8;
449 dmac[5] = dmac64;
450
451 unsigned int gw_id = dmac[3] << 24 | dmac[2] << 16 | dmac[1] << 8 | dmac[0];
452
453 return gw_id;
454}
