90// Inform the display of selected memory area
91void BaseDisplay::setMemoryArea(uint16_t sx, uint16_t sy, uint16_t ex, uint16_t ey) {
92 // Split into bytes
93 uint8_t sy1, sy2, ey1, ey2;
94 sy1 = sy & 0xFF;
95 sy2 = (sy >> 8) & 0xFF;
96 ey1 = ey & 0xFF;
97 ey2 = (ey >> 8) & 0xFF;
98
99 // Data entry mode - Left to Right, Top to Bottom
100 sendCommand(0x11);
101 sendData(0x03);
102
103 // Inform the panel hardware of our chosen memory location
104 sendCommand(0x44); // Memory X start - end
105 sendData(sx);
106 sendData(ex);
107 sendCommand(0x45); // Memory Y start - end
108 sendData(sy1);
109 sendData(sy2);
110 sendData(ey1);
111 sendData(ey2);
112 sendCommand(0x4E); // Memory cursor X
113 sendData(sx);
114 sendCommand(0x4F); // Memory cursor y
115 sendData(sy1);
116 sendData(sy2);
117}

119// Prepare display controller to receive image data, then transfer
120void BaseDisplay::sendImageData() {
121
122 // IF Fastmode OFF
123 // -------------
124 if (fastmode_state == OFF) {
125
126 // Send black
127 sendCommand(0x24); // Write "BLACK" memory
128 for (uint16_t i = 0; i < pagefile_length; i++)
129 sendData(page_black[i]);
130
131 // If supports red, send red
132 if ( supportsColor(RED) ) { // If 3-Color red display
133 sendCommand(0x26); // Write memory for red(1)/white (0)
134 for (uint16_t i = 0; i < pagefile_length; i++)
135 sendData(page_red[i]);
136 }
137
138 // If mono, send black data to red memory, for future partial refresh (differential update)
139 else {
140 sendCommand(0x26);
141 for (uint16_t i = 0; i < pagefile_length; i++)
142 sendData(page_black[i]);
143 }
144 }
145
146 // IF Fastmode ON - first pass
147 // OR Fastmode TURBO
148 // -------------------------
149 else if (!fastmode_secondpass) {
150 // Send black
151 sendCommand(0x24); // Write "BLACK" memory
152 for (uint16_t i = 0; i < pagefile_length; i++)
153 sendData(page_black[i]);
154 }
155
156 // IF Fastmode OFF - second pass
157 // ----------------------------
158 else {
159 // Send black data to red memory, for differential update
160 sendCommand(0x26);
161 for (uint16_t i = 0; i < pagefile_length; i++)
162 sendData(page_black[i]);
163 }
164}

166// Used by clear()
167void BaseDisplay::sendBlankImageData() {
168 // Calculate memory values needed to display default_color
169 uint8_t black_byte, red_byte;
170 black_byte = (default_color & WHITE) * 255; // Get the black mem value for default color
171 if (supportsColor(RED))
172 red_byte = ((default_color & RED) >> 1) * 255; // Get the red mem value for default color
173 else
174 red_byte = black_byte;
175
176 // Determine how many bytes to write
177 uint16_t pagefile_size = (panel_width / 8) * panel_height;
178
179 // Write the data
180 sendCommand(0x24); // Write "BLACK" memory
181 for (uint16_t i = 0; i < pagefile_size; i++)
182 sendData(black_byte);
183
184 // Also write the RED memory, so long as we're not clearing in fastmode (breaks differential update)
185 if (fastmode_state == OFF || fastmode_state == NOT_SET) {
186 sendCommand(0x26); // Write "RED" memory
187 for (uint16_t i = 0; i < pagefile_size; i++)
188 sendData(red_byte);
189 }
190}

201// Clear the screen in one optimized step - public
202void BaseDisplay::clear() {
203 clearAllMemories();
204
205 // Trigger the display update
206 // --------------------------
207
208 // We might need to change to Fastmode::OFF for the update - remember where we started
209 Fastmode original_state = fastmode_state;
210
211 // If sketch just started, might need to load settings
212 if (fastmode_state == NOT_SET)
213 fastmodeOff();
214
215 // Trigger the display changes
216 activate();
217
218 // If we *didn't* want to be in Fastmode::OFF, return to original state
219 if (original_state == ON)
220 fastmodeOn();
221 else if (original_state == TURBO)
222 fastmodeTurbo();
223
224 // Track state of display memory (re:customPowerOn)
225 display_cleared = true;
226 just_restarted = false;
227
228}

230// Clears the display memory, and if PRESERVE_IMAGE, the pagefile too
231void BaseDisplay::clearAllMemories() {
232
233 // Perform hardware init, if not already done
234 begin();
235
236 // Fill the image memory with blank data, if drawing unpaged
237 if (PRESERVE_IMAGE && pagefile_height == panel_height)
238 clearPage();
239
240 // Fill the display's memory with blank data
241 int16_t sx, sy, ex, ey;
242 calculateMemoryArea(sx, sy, ex, ey, 0, 0, panel_width - 1, panel_height - 1 ); // Virtual, derived class
243 setMemoryArea(sx, sy, ex, ey);
244 sendBlankImageData(); // Transfer (blank) image via SPI
245 endImageTxQuiet();
246}

278 // Send power-off signal to your custom power switching circuit, and set the display pins to prevent unwanted current flow
279 void BaseDisplay::customPowerOff(uint16_t pause) {
280 // Just in-case customPowerOff() gets called in some weird place
281 begin();
282
283 // If poweroff is called immediately after drawing, it can compromise the image
284 delay(pause);
285
286 // Stop SPI
287 display_spi->end();
288
289 // Send the power-down signal
290 digitalWrite(pin_power, !switch_type);
291
292 // Set the logic pins: prevent a small current return path
293 pinMode(pin_dc, OUTPUT);
294 pinMode(pin_cs, OUTPUT);
295 pinMode(pin_sdi, OUTPUT);
296 pinMode(pin_clk, OUTPUT);
297 digitalWrite(pin_dc, switch_type);
298 digitalWrite(pin_cs, switch_type);
299 digitalWrite(pin_sdi, switch_type);
300 digitalWrite(pin_clk, switch_type);
301
302 // Prevent some strange leakage through the busy pin (supposedly high-impedance, but..)
303 if (switch_type == PNP) {
304 pinMode(pin_busy, OUTPUT);
305 digitalWrite(pin_busy, LOW);
306 }
307
308 // Same, if we're using sd card
309 if (pin_cs_card != 0xFF) {
310 digitalWrite(pin_cs_card, switch_type);
311 digitalWrite(pin_miso, switch_type);
312 }
313 }

315 // Send power-on signal to your custom power switching circuit, then re-init display
316 void BaseDisplay::customPowerOn() {
317 // We set this to OUTPUT in customPowerOff() to prevent current leakage
318 pinMode(pin_busy, INPUT);
319
320 // CS pin deselcted; power-up the display
321 digitalWrite(pin_cs, HIGH);
322 digitalWrite(pin_power, switch_type);
323
324 // Wait for powerup
325 delay(50);
326
327 // SPI resumes
328 Platform::beginSPI(display_spi, pin_sdi, pin_miso, pin_clk);
329
330 // SAMD21: Move the SPI pins back to custom location
331 #ifdef __SAMD21G18A__
332 if (pin_sdi != MOSI || pin_clk != SCK || pin_miso != MISO)
333 Platform::setSPIPins(pin_sdi, pin_clk, pin_miso);
334 #endif
335
336 // Re-load settings for full-refresh
337 fastmodeOff();
338
339 // Mark display as potentially out of sync with memory
340 just_restarted = true;
341
342 // Re-initialize display memory
343 if (PRESERVE_IMAGE && pagefile_height == panel_height) // If not paged - write old image back to display memory
344 writePage();
345 else // If paged - No record of old image, just fill blank
346 clearAllMemories();
347 }
