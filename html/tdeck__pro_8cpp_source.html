90static void flush_timer_cb(lv_timer_t *t)
91{

132static void disp_flush(lv_disp_drv_t * disp_drv, const lv_area_t * area, lv_color_t * color_p)
133{
134 uint32_t w = (area->x2 - area->x1);
135 uint32_t h = (area->y2 - area->y1);
136
137 uint16_t epd_idx = 0;
138
139 union flush_buf_pixel pixel;
140
141 for(int i = 0; i < w * h; i += 8) {
142 pixel.bit.b1 = (color_p + i + 7)->full;
143 pixel.bit.b2 = (color_p + i + 6)->full;
144 pixel.bit.b3 = (color_p + i + 5)->full;
145 pixel.bit.b4 = (color_p + i + 4)->full;
146 pixel.bit.b5 = (color_p + i + 3)->full;
147 pixel.bit.b6 = (color_p + i + 2)->full;
148 pixel.bit.b7 = (color_p + i + 1)->full;
149 pixel.bit.b8 = (color_p + i + 0)->full;
150 decodebuffer[epd_idx] = pixel.full;
151 epd_idx++;
152 }
153
154 // Serial.printf("x1=%d, y1=%d, x2=%d, y2=%d\n", area->x1, area->y1, area->x2, area->y2);
155
156 /*IMPORTANT!!!
157 *Inform the graphics library that you are ready with the flushing*/
158 lv_disp_flush_ready(disp_drv);
159}

219static bool bq25896_init(void)
220{
221 // BQ25896 --- 0x6B
222 Wire.beginTransmission(BOARD_I2C_ADDR_BQ25896);
223 if (Wire.endTransmission() == 0)
224 {
225 // battery_25896.begin();
226 PPM.init(Wire, BOARD_I2C_SDA, BOARD_I2C_SCL, BOARD_I2C_ADDR_BQ25896);
227 // Set the minimum operating voltage. Below this voltage, the PPM will protect
228 PPM.setSysPowerDownVoltage(3300);
229
230 // Set input current limit, default is 500mA
231 PPM.setInputCurrentLimit(3250);
232
233 Serial.printf("getInputCurrentLimit: %d mA\n",PPM.getInputCurrentLimit());
234
235 // Disable current limit pin
236 PPM.disableCurrentLimitPin();
237
238 // Set the charging target voltage, Range:3840 ~ 4608mV ,step:16 mV
239 PPM.setChargeTargetVoltage(4208);
240
241 // Set the precharge current , Range: 64mA ~ 1024mA ,step:64mA
242 PPM.setPrechargeCurr(64);
243
244 // The premise is that Limit Pin is disabled, or it will only follow the maximum charging current set by Limi tPin.
245 // Set the charging current , Range:0~5056mA ,step:64mA
246 PPM.setChargerConstantCurr(832);
247
248 // Get the set charging current
249 PPM.getChargerConstantCurr();
250 Serial.printf("getChargerConstantCurr: %d mA\n",PPM.getChargerConstantCurr());
251
252 PPM.enableADCMeasure();
253
254 PPM.enableCharge();
255
256 // Turn off charging function
257 // If USB is used as the only power input, it is best to turn off the charging function,
258 // otherwise the VSYS power supply will have a sawtooth wave, affecting the discharge output capability.
259 // PPM.disableCharge();
260
261
262 // The OTG function needs to enable OTG, and set the OTG control pin to HIGH
263 // After OTG is enabled, if an external power supply is plugged in, OTG will be turned off
264
265 PPM.enableOTG();
266 PPM.disableOTG();
267 // pinMode(OTG_ENABLE_PIN, OUTPUT);
268 // digitalWrite(OTG_ENABLE_PIN, HIGH);
269
270 return true;
271 }
272 return false;

410void initTDeck_pro()
411{
412 // LORASDEPD use the same SPI, in order to avoid mutual influence;
413 // before powering on, all CS signals should be pulled high and in an unselected state;
414 pinMode(BOARD_EPD_CS, OUTPUT);
415 digitalWrite(BOARD_EPD_CS, HIGH);
416 pinMode(BOARD_SD_CS, OUTPUT);
417 digitalWrite(BOARD_SD_CS, HIGH);
418 pinMode(BOARD_LORA_CS, OUTPUT);
419 digitalWrite(BOARD_LORA_CS, HIGH);
420
421 Serial.begin(115200);
422
423 // IO
424 pinMode(BOARD_KEYBOARD_LED, OUTPUT);
425 pinMode(BOARD_MOTOR_PIN, OUTPUT);
426 pinMode(BOARD_6609_EN, OUTPUT); // enable 7682 module
427 pinMode(BOARD_LORA_EN, OUTPUT); // enable LORA module
428 pinMode(BOARD_GPS_EN, OUTPUT); // enable GPS module
429 pinMode(BOARD_1V8_EN, OUTPUT); // enable gyroscope module
430 pinMode(BOARD_A7682E_PWRKEY, OUTPUT);
431 digitalWrite(BOARD_KEYBOARD_LED, LOW);
432 digitalWrite(BOARD_MOTOR_PIN, LOW);
433 digitalWrite(BOARD_6609_EN, HIGH);
434 digitalWrite(BOARD_LORA_EN, HIGH);
435 digitalWrite(BOARD_GPS_EN, HIGH);
436 digitalWrite(BOARD_1V8_EN, LOW); // KBC GYRO OFF
437 digitalWrite(BOARD_A7682E_PWRKEY, LOW); // KBC A7682E OFF
438
439 // i2c devices
440 byte error, address;
441 int nDevices = 0;
442 Wire.begin(BOARD_I2C_SDA, BOARD_I2C_SCL);
443 Serial.printf("------------- I2C ------------- \n");
444 for(address = 0x01; address < 0x7F; address++){
445 Wire.beginTransmission(address);
446 error = Wire.endTransmission();
447 if(error == 0){ // 0: success.
448 nDevices++;
449 if(address == BOARD_I2C_ADDR_TOUCH){
450 // flag_Touch_init = true;
451 Serial.printf("[0x%x] TOUCH find!\n", address);
452 } else if (address == BOARD_I2C_ADDR_LTR_553ALS) {
453 Serial.printf("[0x%x] LTR_553ALS find!\n", address);
454 } else if (address == BOARD_I2C_ADDR_GYROSCOPDE) {
455 Serial.printf("[0x%x] GYROSCOPDE find!\n", address);
456 } else if (address == BOARD_I2C_ADDR_KEYBOARD) {
457 Serial.printf("[0x%x] KEYBOARD find!\n", address);
458 } else if (address == BOARD_I2C_ADDR_BQ27220) {
459 Serial.printf("[0x%x] BQ27220 find!\n", address);
460 } else if (address == BOARD_I2C_ADDR_BQ25896) {
461 Serial.printf("[0x%x] BQ25896 find!\n", address);
462 }
463 }
464 }
465
466 Serial.printf("------------- SPIFFS ------------- \n");
467
468 if(!SPIFFS.begin(true)){
469 Serial.println("SPIFFS Mount Failed");
470 return;
471 }
472
473 listDir(SPIFFS, "/", 0);
474
475 Serial.println("------------- PERI ------------- ");
476
477 // SPI
478 SPI.begin(BOARD_SPI_SCK, BOARD_SPI_MISO, BOARD_SPI_MOSI);
479
480 // init peripheral
481 touch.setPins(BOARD_TOUCH_RST, BOARD_TOUCH_INT);
482 peri_init_st[E_PERI_INK_SCREEN] = ink_screen_init();
483 peri_init_st[E_PERI_LORA] = lora_init();
484 peri_init_st[E_PERI_TOUCH] = touch.begin(Wire, BOARD_I2C_ADDR_TOUCH, BOARD_TOUCH_SDA, BOARD_TOUCH_SCL);
485 peri_init_st[E_PERI_KYEPAD] = keypad_init(BOARD_I2C_ADDR_KEYBOARD);

491 peri_init_st[E_PERI_LTR_553ALS] = LTR553_init();
492 peri_init_st[E_PERI_A7682E] = A7682E_init();
493
494 /*
495 if(peri_init_st[E_PERI_A7682E] == false)

497 peri_init_st[E_PERI_PCM5102A] = pcm5102a_init();
498 }
