10// Set the rotation for the display
11void BaseDisplay::setRotation(int16_t r) {
12
13 // Interpret "intuitive" rotation vals
14 switch (r) {
15 // 90deg clockwise
16 case 90:
17 case -270:
18 case -3:
19 r = 1;
20 break;
21
22 // 180 deg
23 case 180:
24 case -180:
25 case -2:
26 r = 2;
27 break;
28
29 // 270 deg clockwise
30 case -90:
31 case 270:
32 case -1:
33 r = 3;
34 break;
35 }
36
37 GFX::setRotation((uint8_t) r); // Base class method
38
39 // Re-calculate window locations, for give accurate bounds info
40 setWindow( bounds.window.left(),
41 bounds.window.top(),
42 bounds.window.width(),
43 bounds.window.height(),
44 false ); // ( Do not clear the page)
45}

47// Set the image flip
48void BaseDisplay::setFlip(Flip flip) {
49 // Reverse the last flip operation applied
50 setWindow(window_left, window_top, window_right - window_left + 1, window_bottom - window_top + 1, false);
51
52 // Store the flip property, for later internal use by GFX methods
53 this->imgflip = (Flip)(flip & (Flip::HORIZONTAL | Flip::VERTICAL));
54
55 // If flipping the whole screen, not within a window, recalculate bounds
56 if (flip == Flip::HORIZONTAL || flip == Flip::VERTICAL)
57 setWindow(window_left, window_top, window_right - window_left + 1, window_bottom - window_top + 1, false);
58}

83// Window update code, with exposed clear_page argument. Prevents image clear during setRotation, with PRESERVE_IMAGE
84void BaseDisplay::setWindow(uint16_t left, uint16_t top, uint16_t width, uint16_t height, bool clear_page) {
85 uint16_t right = left + (width - 1);
86 uint16_t bottom = top + (height - 1);
87 window_left = left;
88 window_top = top;
89 window_right = right;
90 window_bottom = bottom;
91
92 // Apply flip
93 if (imgflip & Flip::HORIZONTAL) {
94 if (rotation % 2) { // If landscape
95 window_right = (drawing_height - 1) - left;
96 window_left = (drawing_height - 1) - right;
97 }
98 else { // If portrait
99 window_right = (drawing_width - 1) - left;
100 window_left = (drawing_width - 1) - right;
101 }
102 }
103 if (imgflip & Flip::VERTICAL) {
104 if (rotation % 2) { // If landscape
105 window_bottom = (drawing_width - 1) - top;
106 window_top = (drawing_width - 1) - bottom;
107 }
108 else { // If portrait
109 window_bottom = (drawing_height - 1) - top;
110 window_top = (drawing_height - 1) - bottom;
111 }
112 }
113
114 // Calculate rotated window locations
115 switch (rotation) {
116 case 0:
117 winrot_left = window_left;
118 winrot_left = winrot_left - (winrot_left % 8); // Expand box on left to fit contents
119
120 winrot_right = winrot_left + 7;
121 while(winrot_right < window_right) winrot_right += 8; // Iterate until box includes the byte where our far-left bit lives
122
123 winrot_top = window_top;
124 winrot_bottom = window_bottom;
125 break;
126
127 case 1:
128 winrot_left = (drawing_width - 1) - window_bottom;
129 winrot_left = winrot_left - (winrot_left % 8); // Expand box on left to fit contents
130
131 winrot_right = winrot_left + 7;
132 while(winrot_right < (drawing_width - 1) - window_top) winrot_right += 8; // Iterate until box includes the byte where our far-left bit lives
133
134 winrot_top = window_left;
135 winrot_bottom = window_right;
136 break;
137
138 case 2:
139 winrot_left = (drawing_width - 1) - window_right;
140 winrot_left = winrot_left - (winrot_left % 8); // Expand box on left to fit contents
141
142 winrot_right = winrot_left + 7;
143 while(winrot_right < (drawing_width - 1) - window_left) winrot_right += 8; // Iterate until box includes the byte where our far-left bit lives
144
145 winrot_bottom = (drawing_height - 1) - window_top;
146 winrot_top = (drawing_height - 1) - window_bottom;
147
148 break;
149
150
151 case 3:
152 winrot_left = window_top;
153 winrot_left = winrot_left - (winrot_left % 8); // Expand box on left to fit contents
154
155 winrot_right = winrot_left + 7;
156 while(winrot_right < window_bottom) winrot_right += 8; // Iterate until box includes the byte where our far-left bit lives
157
158 winrot_top = (drawing_height - 1) - window_right;
159 winrot_bottom = (drawing_height - 1) - window_left;
160 break;
161 } // -- Finish calculating window rotation
162
163 // Limit window to panel
164 if (window_left < 0) window_left = 0;
165 if (window_top < 0) window_top = 0;
166 if (rotation % 2) { // Landscape
167 if (window_right >= drawing_height - 1) window_right = drawing_height - 1;
168 if (window_bottom >= drawing_width - 1) window_bottom = drawing_width - 1;
169 }
170 else { // Portrait
171 if (window_right >= drawing_width - 1) window_right = drawing_width - 1;
172 if (window_bottom >= drawing_height - 1) window_bottom = drawing_height - 1;
173 }
174
175 // If preserving image, and window moves, need to reset relevant area for drawPixel, and clear
176 #if PRESERVE_IMAGE
177 if (pagefile_height == panel_height) { // If user didn't re-enable paging
178 // Specify display region handled, either in paging, or outside loop
179 page_top = winrot_top;
180 page_bottom = min((uint16_t)((winrot_top + pagefile_height) - 1), winrot_bottom);
181 pagefile_length = (page_bottom - page_top + 1) * ((winrot_right - winrot_left + 1) / 8);
182
183 if (clear_page)
184 clearPageWindow(); // This is *usually* just clearPage(), unless "partial window" is not supported.
185 }
186 #endif
187}
