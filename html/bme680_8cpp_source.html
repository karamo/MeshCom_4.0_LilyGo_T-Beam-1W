42void setupBME680()
43{
44
45 if(bWXDEBUG && bDisplayCont)
46 Serial.printf("bBME680ON:%i\n", bBME680ON);
47
48 // Don't mix BME280 or BMP280 and BME680 they share same addresses
49 if(bBME680ON && (bBMEON || bBMPON))
50 {
51 Serial.println("[INIT]...BME680 and BMx280 can't be used together!");
52 bBME680ON=false;
53 bme680_found=false;
54 }
55
56 if(!bBME680ON)
57 return;
58
59 uint8_t foundAddr = 0; // 1 = 0x76, 2 = 0x77, 3 = both
60
61 // check if the sensor is found on the I2C bus
62 Wire.beginTransmission(I2C_ADDRESS_BME680_1);
63 int error = Wire.endTransmission();
64
65 if (error == 0) {
66 Serial.println("[INIT]...BME680 sensor found at 0x76");
67 foundAddr = 1;
68 } else {
69 Serial.println("[INIT]...BME680 sensor not found at 0x76");
70 }
71 Wire.beginTransmission(I2C_ADDRESS_BME680_2);
72 error = Wire.endTransmission();
73 if (error == 0) {
74 Serial.println("[INIT]...BME680 sensor found at 0x77");
75 foundAddr += 2;
76 } else {
77 Serial.println("[INIT]...BME680 sensor not found at 0x77");
78 }
79
80 // if none found or there is another sensor on the other address return for now!!! TODO: fix this
81 if (foundAddr == 0 || foundAddr == 3)
82 {
83 Serial.println("[INIT]...Could not find BME680 sensor or Address conflict!");
84 return;
85 }
86 else
87 {
88 bme680_found = true;
89 }
90
91 // Initialize the BME680
92 switch (foundAddr)
93 {
94 case 1:
95 bme.begin(I2C_ADDRESS_BME680_1);
96 break;
97 case 2:
98 bme.begin(I2C_ADDRESS_BME680_2);
99 break;
100 }
101
102 // Set up oversampling and filter initialization
103 bme.setTemperatureOversampling(BME680_OS_8X);
104 bme.setHumidityOversampling(BME680_OS_2X);
105 bme.setPressureOversampling(BME680_OS_4X);
106 bme.setIIRFilterSize(BME680_FILTER_SIZE_3);
107 bme.setGasHeater(320, 150); // 320*C for 150 ms
108
109}
