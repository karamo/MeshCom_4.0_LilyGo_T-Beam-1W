58bool FatFile::cmpName(uint16_t index, FatLfn_t* fname, uint8_t lfnOrd) {
59 FatFile dir = *this;
60 DirLfn_t* ldir;
61 fname->reset();
62 for (uint8_t order = 1; order <= lfnOrd; order++) {
63 ldir = reinterpret_cast<DirLfn_t*>(dir.cacheDir(index - order));
64 if (!ldir) {
65 DBG_FAIL_MACRO;
66 goto fail;
67 }
68 // These should be checked in caller.
69 DBG_HALT_IF(ldir->attributes != FAT_ATTRIB_LONG_NAME);
70 DBG_HALT_IF(order != (ldir->order & 0X1F));
71 for (uint8_t i = 0; i < 13; i++) {
72 uint16_t u = getLfnChar(ldir, i);
73 if (fname->atEnd()) {
74 return u == 0;
75 }
76#if USE_UTF8_LONG_NAMES
77 uint16_t cp = fname->get16();
78 // Make sure caller checked for valid UTF-8.
79 DBG_HALT_IF(cp == 0XFFFF);
80 if (toUpcase(u) != toUpcase(cp)) {
81 return false;
82 }
83#else // USE_UTF8_LONG_NAMES
84 if (u > 0X7F || toUpper(u) != toUpper(fname->getch())) {
85 return false;
86 }
87#endif // USE_UTF8_LONG_NAMES
88 }
89 }
90 return true;
91
92fail:
93 return false;
94}

96bool FatFile::createLFN(uint16_t index, FatLfn_t* fname, uint8_t lfnOrd) {
97 FatFile dir = *this;
98 DirLfn_t* ldir;
99 uint8_t checksum = lfnChecksum(fname->sfn);
100 uint8_t fc = 0;
101 fname->reset();
102
103 for (uint8_t order = 1; order <= lfnOrd; order++) {
104 ldir = reinterpret_cast<DirLfn_t*>(dir.cacheDir(index - order));
105 if (!ldir) {
106 DBG_FAIL_MACRO;
107 goto fail;
108 }
109 dir.m_vol->cacheDirty();
110 ldir->order = order == lfnOrd ? FAT_ORDER_LAST_LONG_ENTRY | order : order;
111 ldir->attributes = FAT_ATTRIB_LONG_NAME;
112 ldir->mustBeZero1 = 0;
113 ldir->checksum = checksum;
114 setLe16(ldir->mustBeZero2, 0);
115 for (uint8_t i = 0; i < 13; i++) {
116 uint16_t cp;
117 if (fname->atEnd()) {
118 cp = fc++ ? 0XFFFF : 0;
119 } else {
120 cp = fname->get16();
121 // Verify caller checked for valid UTF-8.
122 DBG_HALT_IF(cp == 0XFFFF);
123 }
124 putLfnChar(ldir, i, cp);
125 }
126 }
127 return true;
128
129fail:
130 return false;

133bool FatFile::makeSFN(FatLfn_t* fname) {
134 bool is83;
135 // char c;
136 uint8_t c;
137 uint8_t bit = FAT_CASE_LC_BASE;
138 uint8_t lc = 0;
139 uint8_t uc = 0;
140 uint8_t i = 0;
141 uint8_t in = 7;
142 const char* dot;
143 const char* end = fname->end;
144 const char* ptr = fname->begin;
145
146 // Assume not zero length.
147 DBG_HALT_IF(end == ptr);
148 // Assume blanks removed from start and end.
149 DBG_HALT_IF(*ptr == ' ' || *(end - 1) == ' ' || *(end - 1) == '.');
150
151 // Blank file short name.
152 for (uint8_t k = 0; k < 11; k++) {
153 fname->sfn[k] = ' ';
154 }
155 // Not 8.3 if starts with dot.
156 is83 = *ptr == '.' ? false : true;
157 // Skip leading dots.
158 for (; *ptr == '.'; ptr++) {
159 }
160 // Find last dot.
161 for (dot = end - 1; dot > ptr && *dot != '.'; dot--) {
162 }
163
164 for (; ptr < end; ptr++) {
165 c = *ptr;
166 if (c == '.' && ptr == dot) {
167 in = 10; // Max index for full 8.3 name.
168 i = 8; // Place for extension.
169 bit = FAT_CASE_LC_EXT; // bit for extension.
170 } else {
171 if (sfnReservedChar(c)) {
172 is83 = false;
173 // Skip UTF-8 trailing characters.
174 if ((c & 0XC0) == 0X80) {
175 continue;
176 }
177 c = '_';
178 }
179 if (i > in) {
180 is83 = false;
181 if (in == 10 || ptr > dot) {
182 // Done - extension longer than three characters or no extension.
183 break;
184 }
185 // Skip to dot.
186 ptr = dot - 1;
187 continue;
188 }
189 if (isLower(c)) {
190 c += 'A' - 'a';
191 lc |= bit;
192 } else if (isUpper(c)) {
193 uc |= bit;
194 }
195 fname->sfn[i++] = c;
196 if (i < 7) {
197 fname->seqPos = i;
198 }
199 }
200 }
201 if (fname->sfn[0] == ' ') {
202 DBG_HALT_MACRO;
203 goto fail;
204 }
205 if (is83) {
206 fname->flags = (lc & uc) ? FNAME_FLAG_MIXED_CASE : lc;
207 } else {
208 fname->flags = FNAME_FLAG_LOST_CHARS;
209 fname->sfn[fname->seqPos] = '~';
210 fname->sfn[fname->seqPos + 1] = '1';
211 }
212 return true;
213
214fail:
215 return false;
216}

218bool FatFile::makeUniqueSfn(FatLfn_t* fname) {
219 const uint8_t FIRST_HASH_SEQ = 2; // min value is 2
220 uint8_t pos = fname->seqPos;
221 DirFat_t* dir;
222 uint16_t hex = 0;
223
224 DBG_HALT_IF(!(fname->flags & FNAME_FLAG_LOST_CHARS));
225 DBG_HALT_IF(fname->sfn[pos] != '~' && fname->sfn[pos + 1] != '1');
226
227 for (uint8_t seq = FIRST_HASH_SEQ; seq < 100; seq++) {
228 DBG_WARN_IF(seq > FIRST_HASH_SEQ);
229 hex += millis();
230 if (pos > 3) {
231 // Make space in name for ~HHHH.
232 pos = 3;
233 }
234 for (uint8_t i = pos + 4; i > pos; i--) {
235 uint8_t h = hex & 0XF;
236 fname->sfn[i] = h < 10 ? h + '0' : h + 'A' - 10;
237 hex >>= 4;
238 }
239 fname->sfn[pos] = '~';
240 rewind();
241 while (1) {
242 dir = readDirCache(true);
243 if (!dir) {
244 if (!getError()) {
245 // At EOF and name not found if no error.
246 goto done;
247 }
248 DBG_FAIL_MACRO;
249 goto fail;
250 }
251 if (dir->name[0] == FAT_NAME_FREE) {
252 goto done;
253 }
254 if (isFatFileOrSubdir(dir) && !memcmp(fname->sfn, dir->name, 11)) {
255 // Name found - try another.
256 break;
257 }
258 }
259 }
260 // fall inti fail - too many tries.
261 DBG_FAIL_MACRO;
262
263fail:
264 return false;
265
266done:
267 return true;
268}

270bool FatFile::open(FatFile* dirFile, FatLfn_t* fname, oflag_t oflag) {
271 bool fnameFound = false;
272 uint8_t lfnOrd = 0;
273 uint8_t freeFound = 0;
274 uint8_t freeNeed;
275 uint8_t order = 0;
276 uint8_t checksum = 0;
277 uint8_t ms10;
278 uint8_t nameOrd;
279 uint16_t curIndex;
280 uint16_t date;
281 uint16_t freeIndex = 0;
282 uint16_t freeTotal;
283 uint16_t time;
284 DirFat_t* dir;
285 DirLfn_t* ldir;
286 auto vol = dirFile->m_vol;
287
288 if (!dirFile->isDir() || isOpen()) {
289 DBG_FAIL_MACRO;
290 goto fail;
291 }
292 // Number of directory entries needed.
293 nameOrd = (fname->len + 12) / 13;
294 freeNeed = (fname->flags & FNAME_FLAG_NEED_LFN) ? 1 + nameOrd : 1;
295 dirFile->rewind();
296 while (1) {
297 curIndex = dirFile->m_curPosition / FS_DIR_SIZE;
298 dir = dirFile->readDirCache();
299 if (!dir) {
300 if (dirFile->getError()) {
301 DBG_FAIL_MACRO;
302 goto fail;
303 }
304 // At EOF
305 goto create;
306 }
307 if (dir->name[0] == FAT_NAME_DELETED || dir->name[0] == FAT_NAME_FREE) {
308 if (freeFound == 0) {
309 freeIndex = curIndex;
310 }
311 if (freeFound < freeNeed) {
312 freeFound++;
313 }
314 if (dir->name[0] == FAT_NAME_FREE) {
315 goto create;
316 }
317 } else {
318 if (freeFound < freeNeed) {
319 freeFound = 0;
320 }
321 }
322 // skip empty slot or '.' or '..'
323 if (dir->name[0] == FAT_NAME_DELETED || dir->name[0] == '.') {
324 lfnOrd = 0;
325 } else if (isFatLongName(dir)) {
326 ldir = reinterpret_cast<DirLfn_t*>(dir);
327 if (!lfnOrd) {
328 order = ldir->order & 0X1F;
329 if (order != nameOrd ||
330 (ldir->order & FAT_ORDER_LAST_LONG_ENTRY) == 0) {
331 continue;
332 }
333 lfnOrd = nameOrd;
334 checksum = ldir->checksum;
335 } else if (ldir->order != --order || checksum != ldir->checksum) {
336 lfnOrd = 0;
337 continue;
338 }
339 if (order == 1) {
340 if (!dirFile->cmpName(curIndex + 1, fname, lfnOrd)) {
341 lfnOrd = 0;
342 }
343 }
344 } else if (isFatFileOrSubdir(dir)) {
345 if (lfnOrd) {
346 if (1 == order && lfnChecksum(dir->name) == checksum) {
347 goto found;
348 }
349 DBG_FAIL_MACRO;
350 goto fail;
351 }
352 if (!memcmp(dir->name, fname->sfn, sizeof(fname->sfn))) {
353 if (!(fname->flags & FNAME_FLAG_LOST_CHARS)) {
354 goto found;
355 }
356 fnameFound = true;
357 }
358 } else {
359 lfnOrd = 0;
360 }
361 }
362
363found:
364 // Don't open if create only.
365 if (oflag & O_EXCL) {
366 DBG_FAIL_MACRO;
367 goto fail;
368 }
369 goto open;
370
371create:
372 // don't create unless O_CREAT and write mode
373 if (!(oflag & O_CREAT) || !isWriteMode(oflag)) {
374 DBG_WARN_MACRO;
375 goto fail;
376 }
377 // Keep found entries or start at current index if no free entries found.
378 if (freeFound == 0) {
379 freeIndex = curIndex;
380 }
381
382 while (freeFound < freeNeed) {
383 dir = dirFile->readDirCache();
384 if (!dir) {
385 if (dirFile->getError()) {
386 DBG_FAIL_MACRO;
387 goto fail;
388 }
389 // EOF if no error.
390 break;
391 }
392 freeFound++;
393 }
394 // Loop handles the case of huge filename and cluster size one.
395 freeTotal = freeFound;
396 while (freeTotal < freeNeed) {
397 // Will fail if FAT16 root.
398 if (!dirFile->addDirCluster()) {
399 DBG_FAIL_MACRO;
400 goto fail;
401 }
402 // 16-bit freeTotal needed for large cluster size.
403 freeTotal += vol->dirEntriesPerCluster();
404 }
405 if (fnameFound) {
406 if (!dirFile->makeUniqueSfn(fname)) {
407 goto fail;
408 }
409 }
410 lfnOrd = freeNeed - 1;
411 curIndex = freeIndex + lfnOrd;
412 if (!dirFile->createLFN(curIndex, fname, lfnOrd)) {
413 goto fail;
414 }
415 dir = dirFile->cacheDir(curIndex);
416 if (!dir) {
417 DBG_FAIL_MACRO;
418 goto fail;
419 }
420 // initialize as empty file
421 memset(dir, 0, sizeof(DirFat_t));
422 memcpy(dir->name, fname->sfn, 11);
423
424 // Set base-name and extension lower case bits.
425 dir->caseFlags = (FAT_CASE_LC_BASE | FAT_CASE_LC_EXT) & fname->flags;
426
427 // Set timestamps.
428 if (FsDateTime::callback) {
429 // call user date/time function
430 FsDateTime::callback(&date, &time, &ms10);
431 setLe16(dir->createDate, date);
432 setLe16(dir->createTime, time);
433 dir->createTimeMs = ms10;
434 } else {
435 setLe16(dir->createDate, FS_DEFAULT_DATE);
436 setLe16(dir->modifyDate, FS_DEFAULT_DATE);
437 setLe16(dir->accessDate, FS_DEFAULT_DATE);
438 if (FS_DEFAULT_TIME) {
439 setLe16(dir->createTime, FS_DEFAULT_TIME);
440 setLe16(dir->modifyTime, FS_DEFAULT_TIME);
441 }
442 }
443 // Force write of entry to device.
444 vol->cacheDirty();
445
446open:
447 // open entry in cache.
448 if (!openCachedEntry(dirFile, curIndex, oflag, lfnOrd)) {
449 DBG_FAIL_MACRO;
450 goto fail;
451 }
452 return true;
453
454fail:
455 return false;
456}

458bool FatFile::parsePathName(const char* path, FatLfn_t* fname,
459 const char** ptr) {
460 size_t len = 0;
461 // Skip leading spaces.
462 while (*path == ' ') {
463 path++;
464 }
465 fname->begin = path;
466 fname->len = 0;
467 while (*path && !isDirSeparator(*path)) {
468#if USE_UTF8_LONG_NAMES
469 uint32_t cp;
470 // Allow end = path + 4 since path is zero terminated.
471 path = FsUtf::mbToCp(path, path + 4, &cp);
472 if (!path) {
473 DBG_FAIL_MACRO;
474 goto fail;
475 }
476 len += cp <= 0XFFFF ? 1 : 2;
477 if (cp < 0X80 && lfnReservedChar(cp)) {
478 DBG_FAIL_MACRO;
479 goto fail;
480 }
481#else // USE_UTF8_LONG_NAMES
482 uint8_t cp = *path++;
483 if (cp >= 0X80 || lfnReservedChar(cp)) {
484 DBG_FAIL_MACRO;
485 goto fail;
486 }
487 len++;
488#endif // USE_UTF8_LONG_NAMES
489 if (cp != '.' && cp != ' ') {
490 // Need to trim trailing dots spaces.
491 fname->len = len;
492 fname->end = path;
493 }
494 }
495 if (!fname->len || fname->len > FAT_MAX_LFN_LENGTH) {
496 DBG_FAIL_MACRO;
497 goto fail;
498 }
499 // Advance to next path component.
500 for (; *path == ' ' || isDirSeparator(*path); path++) {
501 }
502 *ptr = path;
503 return makeSFN(fname);
504
505fail:
506 return false;
507}

509bool FatFile::remove() {
510 bool last;
511 uint8_t checksum;
512 FatFile dirFile;
513 DirFat_t* dir;
514 DirLfn_t* ldir;
515
516 // Cant' remove not open for write.
517 if (!isWritable()) {
518 DBG_FAIL_MACRO;
519 goto fail;
520 }
521 // Free any clusters.
522 if (m_firstCluster && !m_vol->freeChain(m_firstCluster)) {
523 DBG_FAIL_MACRO;
524 goto fail;
525 }
526 // Cache directory entry.
527 dir = cacheDirEntry(FsCache::CACHE_FOR_WRITE);
528 if (!dir) {
529 DBG_FAIL_MACRO;
530 goto fail;
531 }
532 checksum = lfnChecksum(dir->name);
533
534 // Mark entry deleted.
535 dir->name[0] = FAT_NAME_DELETED;
536
537 // Set this file closed.
538 m_attributes = FILE_ATTR_CLOSED;
539 m_flags = 0;
540
541 // Write entry to device.
542 if (!m_vol->cacheSync()) {
543 DBG_FAIL_MACRO;
544 goto fail;
545 }
546 if (!isLFN()) {
547 // Done, no LFN entries.
548 return true;
549 }
550 if (!dirFile.openCluster(this)) {
551 DBG_FAIL_MACRO;
552 goto fail;
553 }
554 for (uint8_t order = 1; order <= m_lfnOrd; order++) {
555 ldir = reinterpret_cast<DirLfn_t*>(dirFile.cacheDir(m_dirIndex - order));
556 if (!ldir) {
557 DBG_FAIL_MACRO;
558 goto fail;
559 }
560 if (ldir->attributes != FAT_ATTRIB_LONG_NAME ||
561 order != (ldir->order & 0X1F) || checksum != ldir->checksum) {
562 DBG_FAIL_MACRO;
563 goto fail;
564 }
565 last = ldir->order & FAT_ORDER_LAST_LONG_ENTRY;
566 ldir->order = FAT_NAME_DELETED;
567 m_vol->cacheDirty();
568 if (last) {
569 if (!m_vol->cacheSync()) {
570 DBG_FAIL_MACRO;
571 goto fail;
572 }
573 return true;
574 }
575 }
576 // Fall into fail.
577 DBG_FAIL_MACRO;
578
579fail:
580 return false;
581}
