89void api_wait_wake(void)
90{
91#if defined NRF52_SERIES || defined ESP32
92 // Wait until semaphore is released (FreeRTOS)
93 xSemaphoreTake(g_task_sem, portMAX_DELAY);
94#endif
95#ifdef ARDUINO_ARCH_RP2040
96 bool got_signal = false;
97 while (!got_signal)
98 {
99 osEvent event = osSignalWait(0, osWaitForever);
100
101 if (event.status == osEventSignal)
102 {
103 got_signal = true;
104 }
105 }
106#endif
107}

114void api_wake_loop(uint16_t reason)
115{
116 g_task_event_type |= reason;
117 API_LOG("API", "Waking up loop task");
118
119#if defined NRF52_SERIES || defined ESP32
120 if (g_task_sem != NULL)
121 {
122 // Wake up task to send initial packet
123 xSemaphoreGive(g_task_sem);
124 }
125#endif
126#ifdef ARDUINO_ARCH_RP2040
127 if (loop_thread != NULL)
128 {
129 osSignalSet(loop_thread, SIGNAL_WAKE);

180void api_timer_start(void)
181{
182#if defined NRF52_SERIES
183 // g_task_wakeup_timer.start();
184 if (isInISR())
185 {
186 BaseType_t xHigherPriorityTaskWoken = pdFALSE;
187 xTimerChangePeriodFromISR(g_task_wakeup_timer, mypdMS_TO_TICKS(meshcom_settings.send_repeat_time), &xHigherPriorityTaskWoken);
188 xTimerStartFromISR(g_task_wakeup_timer, &xHigherPriorityTaskWoken);
189 portYIELD_FROM_ISR(xHigherPriorityTaskWoken);
190 }
191 else
192 {
193 xTimerChangePeriod(g_task_wakeup_timer, mypdMS_TO_TICKS(meshcom_settings.send_repeat_time), 0);
194 xTimerStart(g_task_wakeup_timer, 0);
195 }
196#endif
197#if defined ARDUINO_ARCH_RP2040
198 TimerStart(&g_task_wakeup_timer);
199#endif
200#if defined ESP32
