63char* fsFmtDate(char* str, uint16_t date) {
64 str = fsFmtField(str, date & 31, 0);
65 date >>= 5;

83char* fsFmtTimeZone(char* str, int8_t tz) {
84 if (tz & 0X80) {
85 char sign;
86 if (tz & 0X40) {
87 sign = '-';
88 tz = -tz;
89 } else {
90 sign = '+';
91 tz &= 0X7F;
92 }
93 if (tz) {
94 str = fsFmtField(str, 15 * (tz % 4), 0);
95 str = fsFmtField(str, tz / 4, ':');
96 *--str = sign;
97 }
98 *--str = 'C';
99 *--str = 'T';
100 *--str = 'U';
101 }
102 return str;
103}
