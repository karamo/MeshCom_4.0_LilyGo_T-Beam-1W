44static uint8_t CRC7(const uint8_t* data, uint8_t n) {
45 uint8_t crc = 0;
46 for (uint8_t i = 0; i < n; i++) {
47 uint8_t d = data[i];
48 for (uint8_t j = 0; j < 8; j++) {
49 crc <<= 1;
50 if ((d & 0x80) ^ (crc & 0x80)) {
51 crc ^= 0x09;
52 }
53 d <<= 1;
54 }
55 }
56 return (crc << 1) | 1;
57}

127bool SharedSpiCard::begin(SdSpiConfig spiConfig) {
128 Timeout timeout;
129 // Restore state to creator.
130 initSharedSpiCard();
131 m_errorCode = SD_CARD_ERROR_NONE;
132 m_csPin = spiConfig.csPin;
133#if SPI_DRIVER_SELECT >= 2
134 m_spiDriverPtr = spiConfig.spiPort;
135 if (!m_spiDriverPtr) {
136 error(SD_CARD_ERROR_INVALID_CARD_CONFIG);
137 goto fail;
138 }
139#endif // SPI_DRIVER_SELECT
140 sdCsInit(m_csPin);
141 spiUnselect();
142 spiSetSckSpeed(1000UL * SD_MAX_INIT_RATE_KHZ);
143 spiBegin(spiConfig);
144 m_beginCalled = true;
145 uint32_t arg;
146 spiStart();
147
148 // must supply min of 74 clock cycles with CS high.
149 spiUnselect();
150 for (uint8_t i = 0; i < 10; i++) {
151 spiReceive();
152 }
153 spiSelect();
154 timeout.set(SD_INIT_TIMEOUT);
155 while (true) {
156 // command to go idle in SPI mode
157 if (cardCommand(CMD0, 0) == R1_IDLE_STATE) {
158 break;
159 }
160 if (timeout.timedOut()) {
161 error(SD_CARD_ERROR_CMD0);
162 goto fail;
163 }
164 }
165#if USE_SD_CRC
166 if (cardCommand(CMD59, 1) != R1_IDLE_STATE) {
167 error(SD_CARD_ERROR_CMD59);
168 goto fail;
169 }
170#endif // USE_SD_CRC
171 // check SD version
172 while (true) {
173 if (cardCommand(CMD8, 0x1AA) & R1_ILLEGAL_COMMAND) {
174 type(SD_CARD_TYPE_SD1);
175 break;
176 }
177 // Skip first three bytes.
178 for (uint8_t i = 0; i < 4; i++) {
179 m_status = spiReceive();
180 }
181 if (m_status == 0XAA) {
182 type(SD_CARD_TYPE_SD2);
183 break;
184 }
185 if (timeout.timedOut()) {
186 error(SD_CARD_ERROR_CMD8);
187 goto fail;
188 }
189 }
190 // initialize card and send host supports SDHC if SD2
191 arg = type() == SD_CARD_TYPE_SD2 ? 0X40000000 : 0;
192 while (cardAcmd(ACMD41, arg) != R1_READY_STATE) {
193 // check for timeout
194 if (timeout.timedOut()) {
195 error(SD_CARD_ERROR_ACMD41);
196 goto fail;
197 }
198 }
199 // if SD2 read OCR register to check for SDHC card
200 if (type() == SD_CARD_TYPE_SD2) {
201 if (cardCommand(CMD58, 0)) {
202 error(SD_CARD_ERROR_CMD58);
203 goto fail;
204 }
205 if ((spiReceive() & 0XC0) == 0XC0) {
206 type(SD_CARD_TYPE_SDHC);
207 }
208 // Discard rest of ocr - contains allowed voltage range.
209 for (uint8_t i = 0; i < 3; i++) {
210 spiReceive();
211 }
212 }
213 spiStop();
214 spiSetSckSpeed(spiConfig.maxSck);
215 return true;
216
217fail:
218 spiStop();
219 return false;
220}

238// send command and return error code. Return zero for OK
239uint8_t SharedSpiCard::cardCommand(uint8_t cmd, uint32_t arg) {
240 if (!syncDevice()) {
241 return 0XFF;
242 }
243 // select card
244 if (!m_spiActive) {
245 spiStart();
246 }
247 if (cmd != CMD0 && cmd != CMD12 && !waitReady(SD_CMD_TIMEOUT)) {
248 return 0XFF;
249 }
250#if USE_SD_CRC
251 // form message
252 uint8_t buf[6];
253 buf[0] = (uint8_t)0x40U | cmd;
254 buf[1] = (uint8_t)(arg >> 24U);
255 buf[2] = (uint8_t)(arg >> 16U);
256 buf[3] = (uint8_t)(arg >> 8U);
257 buf[4] = (uint8_t)arg;
258
259 // add CRC
260 buf[5] = CRC7(buf, 5);
261
262 // send message
263 spiSend(buf, 6);
264#else // USE_SD_CRC
265 // send command
266 spiSend(cmd | 0x40);
267
268 // send argument
269 uint8_t* pa = reinterpret_cast<uint8_t*>(&arg);
270 for (int8_t i = 3; i >= 0; i--) {
271 spiSend(pa[i]);
272 }
273
274 // send CRC - correct for CMD0 with arg zero or CMD8 with arg 0X1AA
275 spiSend(cmd == CMD0 ? 0X95 : 0X87);
276#endif // USE_SD_CRC
277
278 // discard first fill byte to avoid MISO pull-up problem.
279 spiReceive();
280
281 // there are 1-8 fill bytes before response. fill bytes should be 0XFF.
282 uint8_t n = 0;
283 do {
284 m_status = spiReceive();
285 } while (m_status & 0X80 && ++n < 10);
286 return m_status;
287}

297bool SharedSpiCard::erase(uint32_t firstSector, uint32_t lastSector) {
298 csd_t csd;
299 if (!readCSD(&csd)) {
300 goto fail;
301 }
302 // check for single sector erase
303 if (!csd.eraseSingleBlock()) {
304 // erase size mask
305 uint8_t m = csd.eraseSize() - 1;
306 if ((firstSector & m) != 0 || ((lastSector + 1) & m) != 0) {
307 // error card can't erase specified area
308 error(SD_CARD_ERROR_ERASE_SINGLE_SECTOR);
309 goto fail;
310 }
311 }
312 if (m_type != SD_CARD_TYPE_SDHC) {
313 firstSector <<= 9;
314 lastSector <<= 9;
315 }
316 if (cardCommand(CMD32, firstSector) || cardCommand(CMD33, lastSector) ||
317 cardCommand(CMD38, 0)) {
318 error(SD_CARD_ERROR_ERASE);
319 goto fail;
320 }
321 if (!waitReady(SD_ERASE_TIMEOUT)) {
322 error(SD_CARD_ERROR_ERASE_TIMEOUT);
323 goto fail;
324 }
325 spiStop();
326 return true;
327
328fail:
329 spiStop();
330 return false;
331}

354//------------------------------------------------------------------------------
355bool SharedSpiCard::readData(uint8_t* dst, size_t count) {
356#if USE_SD_CRC
357 uint16_t crc;
358#endif // USE_SD_CRC
359
360 // wait for start sector token
361 Timeout timeout(SD_READ_TIMEOUT);
362 while ((m_status = spiReceive()) == 0XFF) {
363 if (timeout.timedOut()) {
364 error(SD_CARD_ERROR_READ_TIMEOUT);
365 goto fail;
366 }
367 }
368 if (m_status != DATA_START_SECTOR) {
369 error(SD_CARD_ERROR_READ_TOKEN);
370 goto fail;
371 }
372 // transfer data
373 if ((m_status = spiReceive(dst, count))) {
374 error(SD_CARD_ERROR_DMA);
375 goto fail;
376 }
377
378#if USE_SD_CRC
379 // get crc
380 crc = (spiReceive() << 8) | spiReceive();
381 if (crc != CRC_CCITT(dst, count)) {
382 error(SD_CARD_ERROR_READ_CRC);
383 goto fail;
384 }
385#else // USE_SD_CRC
386 // discard crc
387 spiReceive();
388 spiReceive();
389#endif // USE_SD_CRC
390 return true;
391
392fail:
393 spiStop();
394 return false;
395}

600#if USE_SD_CRC

602#else // USE_SD_CRC

604#endif // USE_SD_CRC
605 spiSend(token);
606 spiSend(src, 512);
607 spiSend(crc >> 8);
608 spiSend(crc & 0XFF);
609
610 m_status = spiReceive();
611 if ((m_status & DATA_RES_MASK) != DATA_RES_ACCEPTED) {
612 error(SD_CARD_ERROR_WRITE_DATA);
613 goto fail;
614 }
615 return true;
616
617fail:

619 return false;
