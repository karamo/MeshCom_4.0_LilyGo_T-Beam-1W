69 // Note: All config***() function must be called before begin()

74 Bluefruit.configPrphConn(250, BLE_GAP_EVENT_LENGTH_MIN, 16, 16);
75#endif

77 Bluefruit.begin(1, 0);
78 // BLE Pin

80
81 // Set max power. Accepted values are: (min) -40, -20, -16, -12, -8, -4, 0, 2, 3, 4, 5, 6, 7, 8 (max)
82 // RAK4631 - nRF52840 max 4dBm
83 Bluefruit.setTxPower(0);
84
85#if NO_BLE_LED > 0

90#ifdef _VARIANT_ISP4520_

95#else
97 sprintf(helper_string, "%s-%02x%02x-%s", g_ble_dev_name, dmac[1], dmac[0], meshcom_settings.node_call); // Anzeige mit callsign

110 // Configure and Start Device Information Service

152 * - Interval: fast mode = 20 ms, slow mode = 152.5 ms

266void settings_rx_callback(uint16_t conn_hdl, BLECharacteristic *chr, uint8_t *data, uint16_t len)
267{
268 API_LOG("SETT", "Settings received");
269
270 delay(1000);
271
272 // Check the characteristic
273 if (chr->uuid == g_lora_data.uuid)
274 {
275 if (len != sizeof(s_meshcom_settings))
276 {
277 API_LOG("SETT", "Received settings have wrong size %d", len);
278 return;
279 }
280
281 s_meshcom_settings *rcvdSettings = (s_meshcom_settings *)data;
282 if ((rcvdSettings->valid_mark_1 != 0xAA) || (rcvdSettings->valid_mark_2 != MESHCOM_DATA_MARKER))
283 {
284 API_LOG("SETT", "Received settings data do not have required markers");
285 return;
286 }
287
288 // Save new LoRa settings
289 memcpy((void *)&meshcom_settings, data, sizeof(s_meshcom_settings));
290
291 // Save new settings
292 save_settings();
293
294 // Update settings
295 g_lora_data.write((void *)&meshcom_settings, sizeof(s_meshcom_settings));
296
297 // Inform connected device about new settings
298 g_lora_data.notify((void *)&meshcom_settings, sizeof(s_meshcom_settings));
299
300 /*KBC
301 if (meshcom_settings.resetRequest)
302 {
303 API_LOG("SETT", "Initiate reset");
304 delay(1000);
305 sd_nvic_SystemReset();
306 }
307 */
308
309 // Notify task about the event
310 if (g_task_sem != NULL)
311 {
312 g_task_event_type |= BLE_CONFIG;
313 API_LOG("SETT", "Waking up loop task");
314 xSemaphoreGive(g_task_sem);
315 }
316 }
