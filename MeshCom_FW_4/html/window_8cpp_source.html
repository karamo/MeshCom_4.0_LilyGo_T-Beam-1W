47uint16_t WindowBounds::getWindowBounds(WindowBounds::side request) {
48
49 // Boolean LUT (x:side, y:rotation): after considering rotation, does requested edge need to measure from opposite edge.
50 static const uint8_t rotswap_lut[4] = { 0b0000,
51 0b0101,
52 0b1111,
53 0b1010 };
54
55 // Handle setFlip() - first part
56 // If flipping, we need to find the opposite edge, and right at the end, measure from the opposite side
57 if ( (request % 2) && (*imgflip & HORIZONTAL) ) { // If we're flipping horizontal, and this edge needs flipping
58 request = (WindowBounds::side)((request + 2) % 4);
59 }
60 if ( !(request % 2) && (*imgflip & VERTICAL) ) { // If vertical flip
61 request = (WindowBounds::side)((request + 2) % 4);
62 } // -------------- End Flip Part 1 -----------
63
64 uint16_t rotated_request = (request + *rotation) % 4; // Rotate the WindowBounds::side value
65 uint16_t result; // Build in this varaible
66
67 // Given our request side and rotation, do we need to measure from opposite edge
68 bool rotswap = rotswap_lut[*rotation] & (1 << request);
69
70 // Start by simply picking a rotated edge
71 result = *edges[rotated_request];
72
73 // Handle a special case; funny issues with unusual drawing_width
74 if (rotated_request == R)
75 result = min(result, (uint16_t)(drawing_width - 1));
76
77 // If required by LUT, find display width or height, and subtract the edge distance
78 if (rotswap) {
79 uint16_t minuend = (rotated_request % 2) ? (drawing_width - 1) : (drawing_height - 1);
80 result = minuend - result;
81 }
82
83 // Handle setFlip() - 2nd part
84 if ( (request % 2) && (*imgflip & HORIZONTAL) ) { // If we're flipping horizontal, and this edge needs flipping
85 uint16_t minuend = (rotated_request % 2) ? (drawing_width - 1) : (drawing_height - 1);
86
87 // Handle another special case caused by strange drawing_width
88 if (result > minuend)
89 result = 0;
90 else
91 result = minuend - result; // (this is the normal case)
92 }
93 if ( !(request % 2) && (*imgflip & VERTICAL) ) { // If vertical flip
94 uint16_t minuend = (rotated_request % 2) ? (drawing_width - 1) : (drawing_height - 1);
95 result = minuend - result;
96 } // ----------- end flip part 2 -------------
97
98 return result;
99}
