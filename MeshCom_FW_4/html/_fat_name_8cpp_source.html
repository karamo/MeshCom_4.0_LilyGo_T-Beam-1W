54size_t FatFile::getName7(char* name, size_t size) {
55 FatFile dir;
56 DirLfn_t* ldir;
57 size_t n = 0;
58 if (!isOpen()) {
59 DBG_FAIL_MACRO;
60 goto fail;
61 }
62 if (!isLFN()) {
63 return getSFN(name, size);
64 }
65 if (!dir.openCluster(this)) {
66 DBG_FAIL_MACRO;
67 goto fail;
68 }
69 for (uint8_t order = 1; order <= m_lfnOrd; order++) {
70 ldir = reinterpret_cast<DirLfn_t*>(dir.cacheDir(m_dirIndex - order));
71 if (!ldir) {
72 DBG_FAIL_MACRO;
73 goto fail;
74 }
75 if (ldir->attributes != FAT_ATTRIB_LONG_NAME ||
76 order != (ldir->order & 0X1F)) {
77 DBG_FAIL_MACRO;
78 goto fail;
79 }
80 for (uint8_t i = 0; i < 13; i++) {
81 uint16_t c = getLfnChar(ldir, i);
82 if (c == 0) {
83 goto done;
84 }
85 if ((n + 1) >= size) {
86 DBG_FAIL_MACRO;
87 goto fail;
88 }
89 name[n++] = c >= 0X7F ? '?' : c;
90 }
91 }
92done:
93 name[n] = 0;
94 return n;
95
96fail:
97 name[0] = '\0';
98 return 0;
99}

101size_t FatFile::getName8(char* name, size_t size) {
102 char* end = name + size;
103 char* str = name;
104 char* ptr;
105 FatFile dir;
106 DirLfn_t* ldir;
107 uint16_t hs = 0;
108 uint32_t cp;
109 if (!isOpen()) {
110 DBG_FAIL_MACRO;
111 goto fail;
112 }
113 if (!isLFN()) {
114 return getSFN(name, size);
115 }
116 if (!dir.openCluster(this)) {
117 DBG_FAIL_MACRO;
118 goto fail;
119 }
120 for (uint8_t order = 1; order <= m_lfnOrd; order++) {
121 ldir = reinterpret_cast<DirLfn_t*>(dir.cacheDir(m_dirIndex - order));
122 if (!ldir) {
123 DBG_FAIL_MACRO;
124 goto fail;
125 }
126 if (ldir->attributes != FAT_ATTRIB_LONG_NAME ||
127 order != (ldir->order & 0X1F)) {
128 DBG_FAIL_MACRO;
129 goto fail;
130 }
131 for (uint8_t i = 0; i < 13; i++) {
132 uint16_t c = getLfnChar(ldir, i);
133 if (hs) {
134 if (!FsUtf::isLowSurrogate(c)) {
135 DBG_FAIL_MACRO;
136 goto fail;
137 }
138 cp = FsUtf::u16ToCp(hs, c);
139 hs = 0;
140 } else if (!FsUtf::isSurrogate(c)) {
141 if (c == 0) {
142 goto done;
143 }
144 cp = c;
145 } else if (FsUtf::isHighSurrogate(c)) {
146 hs = c;
147 continue;
148 } else {
149 DBG_FAIL_MACRO;
150 goto fail;
151 }
152 // Save space for zero byte.
153 ptr = FsUtf::cpToMb(cp, str, end - 1);
154 if (!ptr) {
155 DBG_FAIL_MACRO;
156 goto fail;
157 }
158 str = ptr;
159 }
160 }
161done:
162 *str = '\0';
163 return str - name;
164
165fail:
166 *name = 0;
167 return 0;
168}

170size_t FatFile::getSFN(char* name, size_t size) {
171 char c;
172 uint8_t j = 0;
173 uint8_t lcBit = FAT_CASE_LC_BASE;
174 uint8_t* ptr;
175 DirFat_t* dir;
176 if (!isOpen()) {
177 DBG_FAIL_MACRO;
178 goto fail;
179 }
180 if (isRoot()) {
181 if (size < 2) {
182 DBG_FAIL_MACRO;
183 goto fail;
184 }
185 name[0] = '/';
186 name[1] = '\0';
187 return 1;
188 }
189 // cache entry
190 dir = cacheDirEntry(FsCache::CACHE_FOR_READ);
191 if (!dir) {
192 DBG_FAIL_MACRO;
193 goto fail;
194 }
195 ptr = dir->name;
196 // format name
197 for (uint8_t i = 0; i < 12; i++) {
198 if (i == 8) {
199 if (*ptr == ' ') {
200 break;
201 }
202 lcBit = FAT_CASE_LC_EXT;
203 c = '.';
204 } else {
205 c = *ptr++;
206 if ('A' <= c && c <= 'Z' && (lcBit & dir->caseFlags)) {
207 c += 'a' - 'A';
208 }
209 if (c == ' ') {
210 continue;
211 }
212 }
213 if ((j + 1u) >= size) {
214 DBG_FAIL_MACRO;
215 goto fail;
216 }
217 name[j++] = c;
218 }
219 name[j] = '\0';
220 return j;
221
222fail:
223 name[0] = '\0';
224 return 0;
225}
