55void SdSpiArduinoDriver::send(const uint8_t* buf, size_t count) {
56 // If not a multiple of four. Command with CRC used six byte send.
57 while (count % 4) {
58 send(*buf++);
59 count--;
60 }
61 // Convert byte array to 4 byte array.
62 uint32_t myArray[count / 4]; // NOLINT
63 for (int x = 0; x < count / 4; x++) {
64 myArray[x] = ((uint32_t)buf[(x * 4) + 3] << (8 * 3)) |
65 ((uint32_t)buf[(x * 4) + 2] << (8 * 2)) |
66 ((uint32_t)buf[(x * 4) + 1] << (8 * 1)) |
67 ((uint32_t)buf[(x * 4) + 0] << (8 * 0));
68 }
69 m_spi->transfer(reinterpret_cast<void*>(myArray), count);
70}
