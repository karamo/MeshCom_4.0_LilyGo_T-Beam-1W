52bool FatFormatter::format(FsBlockDevice* dev, uint8_t* secBuf, print_t* pr) {
53 bool rtn;
54 m_dev = dev;
55 m_secBuf = secBuf;
56 m_pr = pr;
57 m_sectorCount = m_dev->sectorCount();
58 m_capacityMB = (m_sectorCount + SECTORS_PER_MB - 1) / SECTORS_PER_MB;
59
60 if (m_capacityMB <= 6) {
61 writeMsg("Card is too small.\r\n");
62 return false;
63 } else if (m_capacityMB <= 16) {
64 m_sectorsPerCluster = 2;
65 } else if (m_capacityMB <= 32) {
66 m_sectorsPerCluster = 4;
67 } else if (m_capacityMB <= 64) {
68 m_sectorsPerCluster = 8;
69 } else if (m_capacityMB <= 128) {
70 m_sectorsPerCluster = 16;
71 } else if (m_capacityMB <= 1024) {
72 m_sectorsPerCluster = 32;
73 } else if (m_capacityMB <= 32768) {
74 m_sectorsPerCluster = 64;
75 } else {
76 // SDXC cards
77 m_sectorsPerCluster = 128;
78 }
79 rtn = m_sectorCount < 0X400000 ? makeFat16() : makeFat32();
80 if (rtn) {
81 writeMsg("Format Done\r\n");
82 } else {
83 writeMsg("Format Failed\r\n");
84 }
85 return rtn;
86}

88bool FatFormatter::initFatDir(uint8_t fatType, uint32_t sectorCount) {
89 size_t n;
90 memset(m_secBuf, 0, BYTES_PER_SECTOR);
91 writeMsg("Writing FAT ");
92 for (uint32_t i = 1; i < sectorCount; i++) {
93 if (!m_dev->writeSector(m_fatStart + i, m_secBuf)) {
94 return false;
95 }
96 if ((i % (sectorCount / 32)) == 0) {
97 writeMsg(".");
98 }
99 }
100 writeMsg("\r\n");
101 // Allocate reserved clusters and root for FAT32.
102 m_secBuf[0] = 0XF8;
103 n = fatType == 16 ? 4 : 12;
104 for (size_t i = 1; i < n; i++) {
105 m_secBuf[i] = 0XFF;
106 }
107 return m_dev->writeSector(m_fatStart, m_secBuf) &&
108 m_dev->writeSector(m_fatStart + m_fatSize, m_secBuf);
109}

111void FatFormatter::initPbs() {
112 PbsFat_t* pbs = reinterpret_cast<PbsFat_t*>(m_secBuf);
113 memset(m_secBuf, 0, BYTES_PER_SECTOR);
114 pbs->jmpInstruction[0] = 0XEB;
115 pbs->jmpInstruction[1] = 0X76;
116 pbs->jmpInstruction[2] = 0X90;
117 for (uint8_t i = 0; i < sizeof(pbs->oemName); i++) {
118 pbs->oemName[i] = ' ';
119 }
120 setLe16(pbs->bpb.bpb16.bytesPerSector, BYTES_PER_SECTOR);
121 pbs->bpb.bpb16.sectorsPerCluster = m_sectorsPerCluster;
122 setLe16(pbs->bpb.bpb16.reservedSectorCount, m_reservedSectorCount);
123 pbs->bpb.bpb16.fatCount = 2;
124 // skip rootDirEntryCount
125 // skip totalSectors16
126 pbs->bpb.bpb16.mediaType = 0XF8;
127 // skip sectorsPerFat16
128 // skip sectorsPerTrack
129 // skip headCount
130 setLe32(pbs->bpb.bpb16.hidddenSectors, m_relativeSectors);
131 setLe32(pbs->bpb.bpb16.totalSectors32, m_totalSectors);
132 // skip rest of bpb
133 setLe16(pbs->signature, PBR_SIGNATURE);
134}

136bool FatFormatter::makeFat16() {
137 uint32_t nc;
138 PbsFat_t* pbs = reinterpret_cast<PbsFat_t*>(m_secBuf);
139
140 for (m_dataStart = 2 * BU16;; m_dataStart += BU16) {
141 nc = (m_sectorCount - m_dataStart) / m_sectorsPerCluster;
142 m_fatSize = (nc + 2 + (BYTES_PER_SECTOR / 2) - 1) / (BYTES_PER_SECTOR / 2);
143 uint32_t r = BU16 + 1 + 2 * m_fatSize + FAT16_ROOT_SECTOR_COUNT;
144 if (m_dataStart >= r) {
145 m_relativeSectors = m_dataStart - r + BU16;
146 break;
147 }
148 }
149 // check valid cluster count for FAT16 volume
150 if (nc < 4085 || nc >= 65525) {
151 writeMsg("Bad cluster count\r\n");
152 return false;
153 }
154 m_reservedSectorCount = 1;
155 m_fatStart = m_relativeSectors + m_reservedSectorCount;
156 m_totalSectors =
157 nc * m_sectorsPerCluster + 2 * m_fatSize + m_reservedSectorCount + 32;
158 if (m_totalSectors < 65536) {
159 m_partType = 0X04;
160 } else {
161 m_partType = 0X06;
162 }
163 // write MBR
164 if (!writeMbr()) {
165 return false;
166 }
167 initPbs();
168 setLe16(pbs->bpb.bpb16.rootDirEntryCount, FAT16_ROOT_ENTRY_COUNT);
169 setLe16(pbs->bpb.bpb16.sectorsPerFat16, m_fatSize);
170 pbs->bpb.bpb16.physicalDriveNumber = 0X80;
171 pbs->bpb.bpb16.extSignature = EXTENDED_BOOT_SIGNATURE;
172 setLe32(pbs->bpb.bpb16.volumeSerialNumber, 1234567);
173 for (size_t i = 0; i < sizeof(pbs->bpb.bpb16.volumeLabel); i++) {
174 pbs->bpb.bpb16.volumeLabel[i] = ' ';
175 }
176 pbs->bpb.bpb16.volumeType[0] = 'F';
177 pbs->bpb.bpb16.volumeType[1] = 'A';
178 pbs->bpb.bpb16.volumeType[2] = 'T';
179 pbs->bpb.bpb16.volumeType[3] = '1';
180 pbs->bpb.bpb16.volumeType[4] = '6';
181 if (!m_dev->writeSector(m_relativeSectors, m_secBuf)) {
182 return false;
183 }
184 return initFatDir(16, m_dataStart - m_fatStart);
185}

187bool FatFormatter::makeFat32() {
188 uint32_t nc;
189 PbsFat_t* pbs = reinterpret_cast<PbsFat_t*>(m_secBuf);
190 FsInfo_t* fsi = reinterpret_cast<FsInfo_t*>(m_secBuf);
191
192 m_relativeSectors = BU32;
193 for (m_dataStart = 2 * BU32;; m_dataStart += BU32) {
194 nc = (m_sectorCount - m_dataStart) / m_sectorsPerCluster;
195 m_fatSize = (nc + 2 + (BYTES_PER_SECTOR / 4) - 1) / (BYTES_PER_SECTOR / 4);
196 uint32_t r = m_relativeSectors + 9 + 2 * m_fatSize;
197 if (m_dataStart >= r) {
198 break;
199 }
200 }
201 // error if too few clusters in FAT32 volume
202 if (nc < 65525) {
203 writeMsg("Bad cluster count\r\n");
204 return false;
205 }
206 m_reservedSectorCount = m_dataStart - m_relativeSectors - 2 * m_fatSize;
207 m_fatStart = m_relativeSectors + m_reservedSectorCount;
208 m_totalSectors = nc * m_sectorsPerCluster + m_dataStart - m_relativeSectors;
209 // type depends on address of end sector
210 // max CHS has lba = 16450560 = 1024*255*63
211 if ((m_relativeSectors + m_totalSectors) <= 16450560) {
212 // FAT32 with CHS and LBA
213 m_partType = 0X0B;
214 } else {
215 // FAT32 with only LBA
216 m_partType = 0X0C;
217 }
218 if (!writeMbr()) {
219 return false;
220 }
221 initPbs();
222 setLe32(pbs->bpb.bpb32.sectorsPerFat32, m_fatSize);
223 setLe32(pbs->bpb.bpb32.fat32RootCluster, 2);
224 setLe16(pbs->bpb.bpb32.fat32FSInfoSector, 1);
225 setLe16(pbs->bpb.bpb32.fat32BackBootSector, 6);
226 pbs->bpb.bpb32.physicalDriveNumber = 0X80;
227 pbs->bpb.bpb32.extSignature = EXTENDED_BOOT_SIGNATURE;
228 setLe32(pbs->bpb.bpb32.volumeSerialNumber, 1234567);
229 for (size_t i = 0; i < sizeof(pbs->bpb.bpb32.volumeLabel); i++) {
230 pbs->bpb.bpb32.volumeLabel[i] = ' ';
231 }
232 pbs->bpb.bpb32.volumeType[0] = 'F';
233 pbs->bpb.bpb32.volumeType[1] = 'A';
234 pbs->bpb.bpb32.volumeType[2] = 'T';
235 pbs->bpb.bpb32.volumeType[3] = '3';
236 pbs->bpb.bpb32.volumeType[4] = '2';
237 if (!m_dev->writeSector(m_relativeSectors, m_secBuf) ||
238 !m_dev->writeSector(m_relativeSectors + 6, m_secBuf)) {
239 return false;
240 }
241 // write extra boot area and backup
242 memset(m_secBuf, 0, BYTES_PER_SECTOR);
243 setLe32(fsi->trailSignature, FSINFO_TRAIL_SIGNATURE);
244 if (!m_dev->writeSector(m_relativeSectors + 2, m_secBuf) ||
245 !m_dev->writeSector(m_relativeSectors + 8, m_secBuf)) {
246 return false;
247 }
248 // write FSINFO sector and backup
249 setLe32(fsi->leadSignature, FSINFO_LEAD_SIGNATURE);
250 setLe32(fsi->structSignature, FSINFO_STRUCT_SIGNATURE);
251 setLe32(fsi->freeCount, 0XFFFFFFFF);
252 setLe32(fsi->nextFree, 0XFFFFFFFF);
253 if (!m_dev->writeSector(m_relativeSectors + 1, m_secBuf) ||
254 !m_dev->writeSector(m_relativeSectors + 7, m_secBuf)) {
255 return false;
256 }
257 return initFatDir(32, 2 * m_fatSize + m_sectorsPerCluster);
258}

260bool FatFormatter::writeMbr() {
261 memset(m_secBuf, 0, BYTES_PER_SECTOR);
262 MbrSector_t* mbr = reinterpret_cast<MbrSector_t*>(m_secBuf);
263
264#if USE_LBA_TO_CHS
265 lbaToMbrChs(mbr->part->beginCHS, m_capacityMB, m_relativeSectors);
266 lbaToMbrChs(mbr->part->endCHS, m_capacityMB,
267 m_relativeSectors + m_totalSectors - 1);
268#else // USE_LBA_TO_CHS
269 mbr->part->beginCHS[0] = 1;
270 mbr->part->beginCHS[1] = 1;
271 mbr->part->beginCHS[2] = 0;
272 mbr->part->endCHS[0] = 0XFE;
273 mbr->part->endCHS[1] = 0XFF;
274 mbr->part->endCHS[2] = 0XFF;
275#endif // USE_LBA_TO_CHS
276
277 mbr->part->type = m_partType;
278 setLe32(mbr->part->relativeSectors, m_relativeSectors);
279 setLe32(mbr->part->totalSectors, m_totalSectors);
280 setLe16(mbr->signature, MBR_SIGNATURE);
281 return m_dev->writeSector(0, m_secBuf);
282}
