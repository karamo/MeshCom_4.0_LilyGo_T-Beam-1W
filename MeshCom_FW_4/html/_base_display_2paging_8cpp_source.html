9// Used with a WHILE loop, to break the graphics into small parts, and draw them one at a time
10bool BaseDisplay::calculating() {
11
12 // NOTE: this loop runs BEFORE every new page.
13 // This means that it is actually processing data generated in the last loop
14 // No real reason for this over a do while, just personal preference
15
16 // Beginning of first loop (initial setup)
17 // --------------------------------------
18 if (page_cursor == 0) {
19
20 // Init display, if needed
21 if (fastmode_state == NOT_SET) {
22 fastmodeOff();
23 clearAllMemories(); // Fill whole memory, incase updating window after reset (static)
24 }
25
26 // Grab memory, if it doesn't persist between updates
27 if (!PRESERVE_IMAGE || pagefile_height < panel_height)
28 grabPageMemory();
29
30 // Specify display region handled, either in paging, or outside loop
31 page_top = winrot_top;
32 page_bottom = min((uint16_t)((winrot_top + pagefile_height) - 1), winrot_bottom);
33 pagefile_length = (page_bottom - page_top + 1) * ((winrot_right - winrot_left + 1) / 8);
34
35 // This is usually just clearPage(), unless "partial window" is not supported
36 clearPageWindow();
37
38 // Track state of display memory (re:customPowerOn)
39 display_cleared = false;
40 just_restarted = false;
41
42 // Preserve the initial state - the config set before we begin loop
43 storeDrawingConfig();
44 }
45
46 // End of each loop (start of next)
47 // --------------------------------
48 else {
49 // Return to the config which was present at the start of the first loop - incase user setCursor(), etc
50 restoreDrawingConfig();
51
52 // Check if the last page contained any part of the window
53 if (!(winrot_bottom < page_top || winrot_top > page_bottom))
54 writePage(); // Send off the old page
55
56 // Calculate memory locations for the new page
57 page_top += pagefile_height;
58 page_bottom = min((uint16_t)((page_top + pagefile_height) - 1), winrot_bottom);
59 pagefile_length = (page_bottom - page_top + 1) * ((winrot_right - winrot_left+1) / 8);
60 }
61
62 // Check whether loop should continue
63 // ----------------------------------
64 if (page_top > winrot_bottom) {
65 page_cursor = 0; // Reset for next time
66
67 // Release the memory, if not preserved
68 if (!PRESERVE_IMAGE || pagefile_height < panel_height)
69 freePageMemory();
70 else {
71 // Reset page dimensions now, incase big MCU wants to draw outside loop
72 page_top = winrot_top;
73 page_bottom = min((uint16_t)((winrot_top + pagefile_height) - 1), winrot_bottom);
74 pagefile_length = (page_bottom - page_top + 1) * ((winrot_right - winrot_left + 1) / 8);
75 }
76
77 // Fastmode OFF or TURBO, (single pass)
78 // ----------------------------------
79 if (fastmode_state == OFF || fastmode_state == TURBO) {
80 if(!saving_to_sd)
81 activate();
82
83 return false;
84 }
85
86 // Fastmode ON
87 // ------------
88 else {
89 // If PRESERVE_IMAGE: no need to re-render - we have the whole image in RAM
90 // --------------
91
92 if (PRESERVE_IMAGE && pagefile_height == panel_height) {
93 if (!saving_to_sd) {
94 activate();
95 fastmode_secondpass = true;
96
97 writePage();
98 endImageTxQuiet(); // Controller accepts the "old image data" without refresh - for differential update
99 fastmode_secondpass = false; // Reset for next time
100 return false;
101 }
102 }
103
104 // If Paging
105 // (we need to render all the pieces twice..)
106 // --------------
107
108 // First pass
109 if (fastmode_secondpass == false) {
110 if (!saving_to_sd)
111 activate();
112
113 fastmode_secondpass = true;
114 return true; // Re-calculate the whole display again
115 }
116
117 // Second Pass
118 else {
119 if (!saving_to_sd)
120 endImageTxQuiet(); // Display accepts the "old image data" without updating
121
122 fastmode_secondpass = false; // Reset state for next time
123 return false;
124 }
125 }
126 }
127 else {
128 // This is usually just clearPage(), unless "partial window" is not supported
129 clearPageWindow();
130 page_cursor++;
131 return true; // Keep looping
132 }
133}

135// Clear the data arrays in between pages
136void BaseDisplay::clearPage() {
137 uint8_t black_byte = (default_color & WHITE) * 255; // We're filling in bulk here; bits are either all on or all off
138 for (uint16_t i = 0; i < page_bytecount; i++)
139 page_black[i] = black_byte;
140
141 // Repeat for red
142 if (supportsColor(RED)) {
143 uint8_t red_byte = ((default_color & RED) >> 1) * 255;
144 for (uint16_t i = 0; i < page_bytecount; i++)
145 page_red[i] = red_byte;
146 }
147}

156void BaseDisplay::storeDrawingConfig() {
157 before_paging_font = gfxFont;
158 before_paging_text_color = (Color) textcolor;
159 before_paging_rotation = (Rotation) rotation;
160 before_paging_cursor_x = getCursorX();

164// Restore the drawing config at the start of each paging loop - allows setCursor() before DRAW()
165void BaseDisplay::restoreDrawingConfig() {
166 // Compare these first - they take extra work to set
167 if (gfxFont != before_paging_font)
168 setFont(before_paging_font);
169 if (rotation != before_paging_rotation)
170 setRotation(before_paging_rotation);
171
172 setTextColor(before_paging_text_color);
173 setCursor(before_paging_cursor_x, before_paging_cursor_y);
174}
