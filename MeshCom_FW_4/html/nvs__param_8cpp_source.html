99union nvs_data nvs_param_get(NVSDataID id, NVSType type)
100{
101 union nvs_data t = {0};
102 prefs.begin("system");
103
104 int i = nvs_param_id_is_find(id);
105 if(i == -1) return t;
106
107 switch (nvs_default[i].type)
108 {
109 case NVS_I8: t.i8 = prefs.getChar(nvs_default[i].key); break;
110 case NVS_U8: t.u8 = prefs.getUChar(nvs_default[i].key); break;
111 case NVS_I16: t.i16 = prefs.getShort(nvs_default[i].key); break;
112 case NVS_U16: t.u16 = prefs.getUShort(nvs_default[i].key); break;
113 case NVS_I32: t.i32 = prefs.getInt(nvs_default[i].key); break;
114 case NVS_U32: t.u32 = prefs.getUInt(nvs_default[i].key); break;
115 case NVS_I64: t.i64 = prefs.getLong64(nvs_default[i].key); break;
116 case NVS_U64: t.u64 = prefs.getULong64(nvs_default[i].key); break;
117 case NVS_FLOAT: t.ff = prefs.getFloat(nvs_default[i].key); break;
118 case NVS_DOUBLE: t.dd = prefs.getDouble(nvs_default[i].key); break;
119 case NVS_BLOB: t.bb = prefs.getBool(nvs_default[i].key); break;
120 case NVS_STR: t.str = (prefs.getString(nvs_default[i].key)).c_str(); break;
121 default:
122 break;
123 }
124 // Serial.printf("[GET] ID=%d, Type=%d, data=%d\n", id, type, t.u8);
125 prefs.end();
126 return t;
127}
