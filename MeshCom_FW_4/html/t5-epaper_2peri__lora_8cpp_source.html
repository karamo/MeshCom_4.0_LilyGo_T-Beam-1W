183void lora_set_mode(int mode)
184{
185 if(mode == LORA_MODE_SEND){
186 radio.setPacketSentAction(set_transmit_flag);
187 Serial.println(F("[LORA] Sending first packet ... "));
188 transmissionState = radio.startTransmit("Hello World!");
189 } else if(mode == LORA_MODE_RECV){
190 radio.setPacketReceivedAction(set_receive_flag);
191 Serial.println(F("[LORA] Starting to listen ... "));
192 receivedState = radio.startReceive();
193 if (receivedState == RADIOLIB_ERR_NONE) {
194 Serial.println(F("success!"));
195 } else {
196 Serial.print(F("failed, code "));
197 Serial.println(receivedState);
198 }
199 }
200 lora_mode = mode;
201}
