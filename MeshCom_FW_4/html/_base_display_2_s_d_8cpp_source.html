54// Create a blank 24bit bitmap on SD card
55void BaseDisplay::initBMP(const char* filename) {
56
57 // Open for writing
58 sd = new SDWrapper();
59 sd->begin(pin_cs_card, display_spi);
60 sd->openFile(filename, true);
61
62 // Macro: write data. Undefined at end of method.
63 #define wr(b) ( sd->write(b) )
64
65 // File header
66
67 // 2 Bytes - identify as bitmap
68 wr('B');
69 wr('M');
70
71 // 4 Bytes - byte size, set later
72 wr(0); wr(0); wr(0); wr(0);
73
74 // 4 Bytes - reserved
75 wr(0); wr(0); wr(0); wr(0);
76
77 // 0x0A - 4 Bytes - start of image data
78 wr(0x36); wr(0); wr(0); wr(0);
79
80 // Image header
81
82 // 4 bytes - header size
83 wr(0x28); wr(0); wr(0); wr(0);
84
85 // 4 bytes - image width
86 for (uint8_t b = 0; b < 32; b+=8) {
87 wr( (drawing_width >> b) & 0xFF );
88 }
89
90 // 4 bytes - image height
91 for (uint8_t b = 0; b < 32; b+=8)
92 wr( (panel_height >> b) & 0xFF );
93
94 // 2 bytes - "planes" - must be zero
95 wr(0); wr(0);
96
97 // 2 bytes - bits per pixel - 24
98 wr(24); wr(0);
99
100 // 8 bytes - compression info - irrelevant
101 for (uint8_t i = 0; i < 8; i++)
102 wr(0);
103
104 // 4 bytes - preferred ppm - x
105 wr(0x12); wr(0x0B); wr(0); wr(0); // MSPaint: 4619 pixels per meter, apparently?
106
107 // 4 bytes - preferred ppm - y
108 wr(0x12); wr(0x0B); wr(0); wr(0); // MSPaint: 4619 pixels per meter, apparently?
109
110 // 8 bytes - colormap - unused
111 for (uint8_t i = 0; i < 8; i++)
112 wr(0);
113
114 // Image data
115
116 // Calculate row width (with padding)
117 uint16_t padding = (4 - ((drawing_width*3) % 4)) % 4; // How many bytes of padding, to round up to 4. BMP format is padded to 4 bytes
118 uint32_t row_width = ((drawing_width*3) + padding); // This is too large for uint16_t. Notice the typecast.
119
120 // Get default color in RGB
121 uint8_t B, G, R;
122
123 switch (default_color) {
124 case WHITE:
125 default:
126 B = 0xFF; G = 0xFF; R = 0xFF;
127 break;
128
129 case BLACK:
130 B = 0; G = 0; R = 0;
131 break;
132
133 case RED:
134 B = 0; G = 0; R = 0xFF;
135 break;
136 }
137
138 // Fill with default color
139 for (uint16_t y = 0; y < panel_height; y++) {
140 for (uint16_t x = 0; x < drawing_width * 3; x += 3) {
141 wr(B);
142 wr(G);
143 wr(R);
144 }
145
146 // Add padding at end of row (not a multiple of 3)
147 for (uint16_t x = drawing_width * 3; x < row_width; x++)
148 wr(0);
149 }
150
151 // Go back and set the "byte size" parameter
152 uint32_t size = sd->fileSize();
153
154 sd->seek(2);
155
156 // 4 bytes - file size
157 for (uint8_t b = 0; b < 32; b+=8)
158 wr( (size >> b) & 0xFF );
159
160 // Done
161
162 #undef wr
163
164 delete sd;
165}

173// Draw a Monochrome .bmp file from SD card
174void BaseDisplay::drawMonoBMP(int16_t left, int16_t top, const char* filename, Color foreground_color, Color background_color) {
175
176 // The SD class instance takes a lot of RAM; create as needed (some platforms only)
177 sd = new SDWrapper();
178
179 // Open card
180 sd->begin(pin_cs_card, display_spi);
181
182 // Open image
183 sd->openFile(filename, false);
184
185 // Grab metadata from image
186 uint16_t width = sd->BMPWidth();
187 uint16_t height = sd->BMPHeight();
188 uint16_t image_start = sd->BMPStart();
189 bool color_invert = sd->BMPColorInverted();
190
191
192 // Columns
193 for(uint16_t y = 0; y < height; y++) {
194
195 // Calculate the byte offset of the new row
196 uint16_t padding = (32 - (width % 32)) % 32; // How many bits of padding, to round up to 32. BMP format is padded to 4 bytes
197 uint32_t row_start = image_start + (y* (width + padding) / 8 );
198
199 // Move to that location in the .bmp file
200 sd->seek(row_start);
201
202 // Rows
203 for(uint16_t x = 0; x < width; x+=8) {
204
205 // Grab data
206 uint8_t data = sd->read();
207
208 // Each bit is a separate pixel
209 for(int8_t b = 7; b >= 0; b--) {
210
211 // Is the image bit low or high?
212 bool val = (data >> b) & 1;
213
214 // Invert the bit, if weird .bmp
215 if(color_invert)
216 val = !val;
217
218 // Apply user's color
219 Color c;
220 if (val == 0)
221 c = foreground_color;
222 else if (background_color != (Color) -1)
223 c = background_color;
224 else
225 continue; // No background requested
226
227 // Flip the row order (.bmp nonsense), and add the bitcount to the x bytes, for pixel coords
228 uint16_t source_x = x + (7-b);
229 uint16_t source_y = (height - 1) - y;
230
231 // Don't draw the padding at end of rows
232 if (source_x < width)
233 drawPixel(source_x + left, source_y + top, c);
234 }
235 }
236 }
237
238 // End of method: delete SD
239 delete sd;
240}

334char* BaseDisplay::getIterableFilename(const char* prefix, uint32_t number) {
335
336 // Cursor for the new "filename" string
337 uint8_t cursor = 0;
338
339 // 8 chars + .bmp + null-term
340 static char filename[8 + 4 + 1]; // (static: lazy, and we only need to use one file at a time, so)
341
342 // Copy prefix into the new string, counting
343 for(; cursor < 6; cursor++) {
344 filename[cursor] = prefix[cursor];
345
346 // All the prefix already copied
347 if (cursor == strlen(prefix) - 1)
348 break;
349 }
350
351 uint8_t cursor_final = cursor; // Final position for a digit
352 cursor = 7;
353
354 // Check if number too large
355 number = min(number, (uint32_t)pow(10, (7 - cursor_final)) - 1);
356
357 // Add digits (or zeros), from smallest to largest
358 while (cursor > cursor_final) {
359 filename[cursor] = number % 10;
360 number /= 10;
361
362 filename[cursor] += 48; // Convert: (int) 0 to (char)'0'
363
364 cursor--; // Step one char back (00# -> 0#0 -# #00)
365 }
366
367 // Add the extension
368 strcpy(&filename[8], ".bmp");
369
370 return filename;
371}

382// Load a "canvas" image file from SD, direct to screen
383void BaseDisplay::loadFullscreenBMP(const char* filename) {
384 begin();
385
386 // Method writes direct to display. Have to make sure hardware init is done
387 if (fastmode_state == NOT_SET)
388 fastmodeOff();
389
390 // Need to return to fullscreen to load - store the current window and restore later
391 uint16_t oldwin_left = bounds.window.left();
392 uint16_t oldwin_top = bounds.window.top();
393 uint16_t oldwin_width = bounds.window.width();
394 uint16_t oldwin_height = bounds.window.height();
395 fullscreen();
396
397 // The SD class instance takes a lot of RAM; create as needed (some platforms only)
398 sd = new SDWrapper();
399
400 // Open card
401 sd->begin(pin_cs_card, display_spi);
402
403 // Open image
404 sd->openFile(filename, false);
405
406 // Tell the display that we will write the full screen
407 int16_t sx, sy, ex, ey;
408 calculateMemoryArea(sx, sy, ex, ey, 0, 0, panel_width - 1, panel_height - 1); // Virtual, derived class
409 setMemoryArea(sx, sy, ex, ey);
410
411 // Send the data to the display
412 sendCommand(0x24);
413 send24BitBMP(BLACK);
414
415 // Write second ram: fastmodeOFF
416 if (fastmode_state == OFF) {
417 // Write to "RED" ram
418 sendCommand(0x26);
419
420 // If display has RED, detect the red pixels, otherwise just use the black again
421 send24BitBMP(supportsColor(RED) ? RED : BLACK);
422
423 // Display the result
424 activate();
425 }
426
427 // If fastmodeON
428 else if (fastmode_state == ON) {
429 // Update the display first,
430 activate();
431
432 // Then send the data again, before final update
433 // setMemoryArea(sx, sy, ex, ey);
434 sendCommand(0x26);
435 send24BitBMP(BLACK);
436 endImageTxQuiet();
437 }
438
439 // If fastmode TURBO, for some reason
440 else if (fastmode_state == TURBO)
441 activate();
442
443 // Free memory from SD instance
444 delete sd;
445
446 // Restore the previous window setting
447 setWindow(oldwin_left, oldwin_top, oldwin_width, oldwin_height);
448}

461 // Backend for the SAVE_TO_SD loop
462 bool BaseDisplay::savingBMP(const char* filename) {
463
464 // Start of SAVE_TO_SD loop
465 if (!saving_to_sd) {
466
467 // Store the filename, for writePageToBMP()
468 sd_filename = filename;
469
470 // First loop only - repair canvas if corrupt
471 if (!fullscreenBMPValid(filename, true))
472 initBMP(filename);
473
474 // Signals that next savingBMP setup is done. Also signals writePageToBMP() to intercept the outgoing gfx data
475 saving_to_sd = true;
476
477 // Now, offload onto calculating() (main "paging" loop). SD class is (usually) instantiated in writePageToBMP()
478 return calculating();
479 }
480
481 // Subequent loops
482 else {
483
484 // If the usual paging loop is still doing its thing, loop around
485 if (calculating())
486 return true;
487
488 // Otherwise, stop intercepting writePage()
489 saving_to_sd = false;
490
491 // No need for another loop
492 return false;
493 }
494
495 }

525// Feed bitmap data directly into display - helper method behind loadFullscreenBMP()
526void BaseDisplay::send24BitBMP(Color target) {
527
528 // Grab metadata
529 uint16_t width = sd->BMPWidth();
530 uint16_t height = sd->BMPHeight();
531 uint16_t image_start = sd->BMPStart();
532
533 // Rows
534 for(int16_t y = (int16_t) height - 1; y >= 0; y--) { // Cast to suppress warning, signed so y can be < 0
535
536 // Calculate the byte offset of the new row
537 uint16_t padding = (4 - ((width*3) % 4)) % 4; // How many bytes of padding, to round up to 4. BMP format is padded to 4 bytes
538 uint32_t row_start = image_start + ((uint32_t)y * ((width*3) + padding)); // This is too large for uint16_t. Notice the typecast.
539
540 // Move to row location in the .bmp file
541 sd->seek(row_start);
542
543 // Columns
544 for(uint16_t x = 0; x < width; x+=8) {
545
546 // Clear byte
547 uint8_t display_data = 0x0; // Default bit state: black
548
549 // Each bit of display - 3 bytes of .bmp
550 for (int8_t b = 7; b >= 0; b--) {
551
552 uint8_t B = sd->read();
553 uint8_t G = sd->read();
554 uint8_t R = sd->read();
555
556 // Determine the color of the pixel
557 Color pixel = parseColor(B, G, R);
558
559 // Set the image bit accordingly
560
561 if (target == BLACK) {
562 if (pixel == WHITE || pixel == RED)
563 display_data |= (1<<b);
564 }
565
566 else if (target == RED) {
567 if (pixel == RED)
568 display_data |= (1<<b);
569 }
570
571 // Special case: take pity on 2.13" displays - accept 122px wide images
572 if (drawing_width == 122 && width == 122 && x + (7 - b) == drawing_width - 1) {
573 x = width; // Don't try load x 122 to 127, it doesn't exist
574 break; // Accept the rest of the byte as blank
575 }
576 }
577 // Transfer 8 pixels to the screen
578 sendData(display_data);
579 }
580 }
581}

583// Decide which of the available display colors best matches a 24bit Bitmap pixel
584Color BaseDisplay::parseColor(uint8_t B, uint8_t G, uint8_t R) {
585
586 if (B < R/2 && G < R/2 && R > 64) // Pixel shouldn't be too dark, and should be dominated by red
587 return RED;
588 else if (B < 127 && G < 127 && R < 127) // 50% threshold for black / white
589 return BLACK;
590 else
591 return WHITE;
592}

594void BaseDisplay::writePageToBMP() {
595
596 // Instantiate SD, but only inside method. (drawing operations might need to use the card too)
597 sd = new SDWrapper(); // Needs to be deleted at end of method
598
599 // Open card
600 sd->begin(pin_cs_card, display_spi);
601
602 // Open image. Filename stored in savingBMP()
603 sd->openFile(this->sd_filename, true);
604
605 // Get the bounds of the page we are about to write
606 uint16_t left = winrot_left;
607 uint16_t right = winrot_right;
608 uint16_t top = (panel_height - 1) - page_top; // Inverted, .bmp rows come out reverse order..
609 uint16_t bottom = (panel_height -1) - page_bottom;
610
611 // Pre-calculate some dimension info
612 uint16_t padding = (drawing_width*3) % 4; // We *might* need row padding, if we crop to 122px for 2.13" displays. BMP format is padded to 4bytes
613 uint16_t row_width = (drawing_width*3) + padding; // Width of the row, 24bit pixels
614 uint32_t image_start = sd->BMPStart(); // Initial offset of image data in file
615
616 // Counters for the page data: pagefile, pagefile bits
617 uint16_t pbyte = 0, pbit = 7;
618
619
620 // Work backwards through .bmp rows, as pagefile works forwards
621 for (int16_t y = top; y >= (int16_t)bottom; y--) {
622
623 // Seek() the start of the .bmp row
624 uint32_t row_start = image_start + ((uint32_t)y * row_width); // Cast to prevent overflow
625 uint32_t left_offset = (left * 3);
626
627 sd->seek(row_start + left_offset);
628
629 // 3 Bytes for each pixel
630 for(uint16_t x = left; x <= right; x++) {
631
632 // Decode color info from the individual pagefiles
633 // (This is not ideal)
634 uint8_t B=0, G=0, R=0; // Default BLACK
635 uint8_t black_bit = (page_black[pbyte] >> pbit) & 1; // Grab from black pagefile
636
637 // Mono display
638 if (!supportsColor(RED)) {
639 if (black_bit == 1) {
640 B = G = R = 0xFF; // output: white
641 }
642 }
643
644 // 3 Color Red
645 else if (supportsColor(RED)) {
646 uint8_t red_bit = (page_red[pbyte] >> pbit) & 1; // Also grab from red pagefile
647 if (black_bit == 1 && red_bit == 0) {
648 B = G = R = 0xFF; // output: white
649 }
650 else if (black_bit == 1 && red_bit == 1 ) {
651 B = 0; G = 0; R = 0xFF; // output: red
652 }
653 }
654
655 // Write the 24bits of color info to the SD
656 sd->write(B);
657 sd->write(G);
658 sd->write(R);
659
660 // Special case: take pity on 2.13" displays - crop bmp at 122px wide
661 // Is set by using drawing_width in row_start, and in initBMP()
662 if (drawing_width == 122 && x == drawing_width - 1) {
663 // Pretend that we have finished reading the last byte
664 x = right;
665 pbit = 0;
666
667 // Add padding (only needed in this specific situation)
668 for(uint8_t i = 0; i < padding; i++)
669 sd->write(0);
670 }
671
672 // Move pagefile counters
673 if (pbit > 0)
674 pbit --;
675 else {
676 // wrap around
677 pbit = 7;
678 pbyte++;
679 }
680 } // End of row
681 }
682
683 // Finished with the SD card.
684 delete sd;
685}

716// Check if canvas .bmp is valid, or corrupt
717bool BaseDisplay::fullscreenBMPValid(const char* filename, bool purge) {
718
719 bool exists = true;
720 bool isValid = true;
721
722 // Create SD, open card
723 sd = new SDWrapper();
724 sd->begin(pin_cs_card, display_spi);
725
726 // Might as well check if file exists first
727 if (!sd->exists(filename)) {
728 isValid = false;
729 exists = false;
730 }
731
732 else {
733 sd->openFile(filename);
734
735 uint16_t width = sd->BMPWidth();
736 uint16_t height = sd->BMPHeight();
737 uint32_t reported_size = sd->BMPReportedSize();
738 uint32_t true_size = sd->fileSize();
739
740 // Check width, height, and filesize from bitmap header
741 if ( (width != drawing_width && width != panel_width) || height != panel_height || reported_size != true_size || true_size == 0 )
742 isValid = false;
743 }
744
745 // Delete, if necessary, and requested
746 if (!isValid && purge && exists)
747 sd->remove(filename);
748
749 // Tidy up
750 delete sd;
751
752 return isValid;
753}
