88// Print text in the center of the display, with optional offset
89void BaseDisplay::printCenter(const char* text, int16_t offset_x, int16_t offset_y) {
90 // Find dimensions of the text
91 int16_t text_off_x(0), text_off_y(0); // How far is text top left from the centerpoint
92 uint16_t width(0), height(0);
93 getTextBounds(text, 0, 0, &text_off_x, &text_off_y, &width, &height);
94
95 // Move the cursor into position
96 uint16_t left = bounds.window.centerX() - text_off_x;
97 uint16_t top = bounds.window.centerY() - text_off_y;
98 setCursor(
99 left - (width / 2) + offset_x,
100 top - (height / 2) + offset_y
101 );
102
103 // Print the text
104 print(text);
105}

131// Print a double in the center of the display, optionally specifying decimal places and offset
132void BaseDisplay::printCenter(double value, uint8_t decimal_places, int16_t offset_x, int16_t offset_y) {
133 uint8_t length = 0;
134 uint8_t digits_before_decimal = log(abs(value)) + 1;
135
136 // Space for minus sign
137 if (value < 0)
138 length++;
139
140 // Digits before decimal point
141 length += digits_before_decimal;
142
143 // Space for decimal point
144 if (decimal_places > 0)
145 length += decimal_places;
146
147 // Null terminator
148 length++;
149
150 // Get the string (dynamic length), then pass through
151 char *text = new char[length];
152
153 // sprintf(text, )
154 dtostrf(value, 0, decimal_places, text); // Length without the null-term
155
156 printCenter(text, offset_x, offset_y);
157 delete[] text;
158}

182inline GFXglyph *pgm_read_glyph_ptr(const GFXfont *gfxFont, uint8_t c) {
183#ifdef __AVR__
184 return &(((GFXglyph *)pgm_read_pointer(&gfxFont->glyph))[c]);
185#else
186 // expression in __AVR__ section may generate "dereferencing type-punned
187 // pointer will break strict-aliasing rules" warning In fact, on other
188 // platforms (such as STM32) there is no need to do this pointer magic as
189 // program memory may be read in a usual way So expression may be simplified
190 return gfxFont->glyph + c;
191#endif // __AVR__
192}

200 // as STM32) there is no need to do this pointer magic as program memory may
201 // be read in a usual way So expression may be simplified

214size_t BaseDisplay::write(uint8_t c) {
215
216 if (!gfxFont) { // 'Classic' built-in font
217
218 if (c == '\n') { // Newline?
219 // Newline: either below the first line, or at window edge, if window moved
220 cursor_x = max( (int16_t)bounds.window.left(), cursor_placed_x);
221 cursor_y += textsize_y * 8; // advance y one line
222 }
223 else if (c != '\r') { // Ignore carriage returns
224 if (wrap && ((cursor_x + textsize_x * 6) > (int16_t)bounds.window.right())) { // Off right?
225 cursor_x = max( (int16_t)bounds.window.left(), cursor_placed_x); // Reset x
226 cursor_y += textsize_y * 8; // advance y one line
227 }
228 drawChar(cursor_x, cursor_y, c, textcolor, textbgcolor, textsize_x, textsize_y);
229 cursor_x += textsize_x * 6; // Advance x one char
230 }
231
232 }
233 else { // Custom font
234
235 if (c == '\n') {
236 // Newline: either below the first line, or at window edge, if window moved
237 cursor_x = max( (int16_t)bounds.window.left(), cursor_placed_x);
238 cursor_y += (int16_t)textsize_y * (uint8_t)pgm_read_byte(&gfxFont->yAdvance);
239 }
240 else if (c != '\r') {
241 uint8_t first = pgm_read_byte(&gfxFont->first);
242 if ((c >= first) && (c <= (uint8_t)pgm_read_byte(&gfxFont->last))) {
243 GFXglyph *glyph = pgm_read_glyph_ptr(gfxFont, c - first);
244 uint8_t w = pgm_read_byte(&glyph->width);
245 uint8_t h = pgm_read_byte(&glyph->height);
246
247 if ((w > 0) && (h > 0)) { // Is there an associated bitmap?
248 int16_t xo = (int8_t)pgm_read_byte(&glyph->xOffset); // sic
249
250 if (wrap && ((cursor_x + textsize_x * (xo + w)) > (int16_t)bounds.window.right())) { // Cursor beyond right edge
251 cursor_x = max( (int16_t)bounds.window.left(), cursor_placed_x);
252 cursor_y += (int16_t)textsize_y * (uint8_t)pgm_read_byte(&gfxFont->yAdvance);
253 }
254 drawChar(cursor_x, cursor_y, c, textcolor, textbgcolor, textsize_x, textsize_y);
255 }
256 cursor_x += (uint8_t)pgm_read_byte(&glyph->xAdvance) * (int16_t)textsize_x;
257 }
258 }
259 }
260 return 1;
261}

263void BaseDisplay::charBounds(unsigned char c, int16_t *x, int16_t *y, int16_t *minx, int16_t *miny, int16_t *maxx, int16_t *maxy)
264{
265 if (gfxFont) {
266 if (c == '\n') { // Newline
267 *x = max( (int16_t)bounds.window.left(), cursor_placed_x); // Reset x to line start, advance y by one line
268 *y += textsize_y * (uint8_t)pgm_read_byte(&gfxFont->yAdvance);
269 }
270
271 else if (c != '\r') { // Not a carriage return; is normal char
272 uint8_t first = pgm_read_byte(&gfxFont->first),
273 last = pgm_read_byte(&gfxFont->last);
274
275 if ((c >= first) && (c <= last)) { // Char present in this font?
276 GFXglyph *glyph = pgm_read_glyph_ptr(gfxFont, c - first);
277
278 uint8_t gw = pgm_read_byte(&glyph->width),
279 gh = pgm_read_byte(&glyph->height),
280 xa = pgm_read_byte(&glyph->xAdvance);
281 int8_t xo = pgm_read_byte(&glyph->xOffset),
282 yo = pgm_read_byte(&glyph->yOffset);
283
284 if (wrap && ((*x + (((int16_t)xo + gw) * textsize_x)) > (int16_t)bounds.window.right())) { // Cursor beyond right edge
285 *x = max( (int16_t)bounds.window.left(), cursor_placed_x) + xa; // Reset x to line start, advance y by one line
286 *y += textsize_y * (uint8_t)pgm_read_byte(&gfxFont->yAdvance);
287 }
288
289 int16_t tsx = (int16_t)textsize_x,
290 tsy = (int16_t)textsize_y,
291 x1 = *x + xo * tsx,
292 y1 = *y + yo * tsy,
293 x2 = x1 + gw * tsx - 1,
294 y2 = y1 + gh * tsy - 1;
295
296 if (x1 < *minx)
297 *minx = x1;
298 if (y1 < *miny)
299 *miny = y1;
300 if (x2 > *maxx)
301 *maxx = x2;
302 if (y2 > *maxy)
303 *maxy = y2;
304
305 *x += xa * tsx;
306 }
307 }
308 }
309
310 else { // Default font
311
312 if (c == '\n') { // Newline
313 *x = max( (int16_t)bounds.window.left(), cursor_placed_x); // Reset x to line start, advance y by one line
314 *y += textsize_y * 8; // advance y one line
315 // min/max x/y unchaged -- that waits for next 'normal' character
316 }
317
318 else if (c != '\r') { // Normal char; ignore carriage returns
319
320 if (wrap && ((*x + textsize_x * 6) > (int16_t)bounds.window.right())) { // Cursor beyond right edge
321 cursor_x = max( (int16_t)bounds.window.left(), cursor_placed_x); // Reset x
322 *y += textsize_y * 8; // advance y one line
323 }
324
325 int x2 = *x + textsize_x * 6 - 1, // Lower-right pixel of char
326 y2 = *y + textsize_y * 8 - 1;
327 if (x2 > *maxx)
328 *maxx = x2; // Track max x, y
329 if (y2 > *maxy)
330 *maxy = y2;
331 if (*x < *minx)
332 *minx = *x; // Track min x, y
333 if (*y < *miny)
334 *miny = *y;
335 *x += textsize_x * 6; // Advance x one char
336 }
337 }
338}
