61static uint16_t exFatDirChecksum(const uint8_t* data, uint16_t checksum) {
62 bool skip = data[0] == EXFAT_TYPE_FILE;
63 for (size_t i = 0; i < 32; i += i == 1 && skip ? 3 : 1) {
64 checksum = ((checksum << 15) | (checksum >> 1)) + data[i];
65 }
66 return checksum;
67}

69bool ExFatFile::addCluster() {
70 uint32_t find = m_vol->bitmapFind(m_curCluster ? m_curCluster + 1 : 0, 1);
71 if (find < 2) {
72 DBG_FAIL_MACRO;
73 goto fail;
74 }
75 if (!m_vol->bitmapModify(find, 1, 1)) {
76 DBG_FAIL_MACRO;
77 goto fail;
78 }
79 if (m_curCluster == 0) {
80 m_flags |= FILE_FLAG_CONTIGUOUS;
81 goto done;
82 }
83 if (isContiguous()) {
84 if (find == (m_curCluster + 1)) {
85 goto done;
86 }
87 // No longer contiguous so make FAT chain.
88 m_flags &= ~FILE_FLAG_CONTIGUOUS;
89
90 for (uint32_t c = m_firstCluster; c < m_curCluster; c++) {
91 if (!m_vol->fatPut(c, c + 1)) {
92 DBG_FAIL_MACRO;
93 goto fail;
94 }
95 }
96 }
97 // New cluster is EOC.
98 if (!m_vol->fatPut(find, EXFAT_EOC)) {
99 DBG_FAIL_MACRO;
100 goto fail;
101 }
102 // Connect new cluster to existing chain.
103 if (m_curCluster) {
104 if (!m_vol->fatPut(m_curCluster, find)) {
105 DBG_FAIL_MACRO;
106 goto fail;
107 }
108 }
109
110done:
111 m_curCluster = find;
112 return true;
113
114fail:
115 return false;
116}

118bool ExFatFile::addDirCluster() {
119 uint32_t sector;
120 uint32_t dl = isRoot() ? m_vol->rootLength() : m_dataLength;
121 uint8_t* cache;
122 dl += m_vol->bytesPerCluster();
123 if (dl >= 0X4000000) {
124 DBG_FAIL_MACRO;
125 goto fail;
126 }
127 if (!addCluster()) {
128 DBG_FAIL_MACRO;
129 goto fail;
130 }
131 sector = m_vol->clusterStartSector(m_curCluster);
132 for (uint32_t i = 0; i < m_vol->sectorsPerCluster(); i++) {
133 cache =
134 m_vol->dataCachePrepare(sector + i, FsCache::CACHE_RESERVE_FOR_WRITE);
135 if (!cache) {
136 DBG_FAIL_MACRO;
137 goto fail;
138 }
139 memset(cache, 0, m_vol->bytesPerSector());
140 }
141 if (!isRoot()) {
142 m_flags |= FILE_FLAG_DIR_DIRTY;
143 m_dataLength += m_vol->bytesPerCluster();
144 m_validLength += m_vol->bytesPerCluster();
145 }
146 return sync();
147
148fail:
149 return false;

388bool ExFatFile::syncDir() {
389 DirFile_t* df;
390 DirStream_t* ds;
391 uint8_t* cache;
392 uint16_t checksum = 0;
393
394 for (uint8_t is = 0; is <= m_setCount; is++) {
395 cache = dirCache(is, FsCache::CACHE_FOR_READ);
396 if (!cache) {
397 DBG_FAIL_MACRO;
398 goto fail;
399 }
400 switch (cache[0]) {
401 case EXFAT_TYPE_FILE:
402 df = reinterpret_cast<DirFile_t*>(cache);
403 setLe16(df->attributes, m_attributes & FS_ATTRIB_COPY);
404 if (FsDateTime::callback) {
405 uint16_t date, time;
406 uint8_t ms10;
407 FsDateTime::callback(&date, &time, &ms10);
408 df->modifyTimeMs = ms10;
409 setLe16(df->modifyTime, time);
410 setLe16(df->modifyDate, date);
411 setLe16(df->accessTime, time);
412 setLe16(df->accessDate, date);
413 }
414 m_vol->dataCacheDirty();
415 break;
416
417 case EXFAT_TYPE_STREAM:
418 ds = reinterpret_cast<DirStream_t*>(cache);
419 if (isContiguous()) {
420 ds->flags |= EXFAT_FLAG_CONTIGUOUS;
421 } else {
422 ds->flags &= ~EXFAT_FLAG_CONTIGUOUS;
423 }
424 setLe64(ds->validLength, m_validLength);
425 setLe32(ds->firstCluster, m_firstCluster);
426 setLe64(ds->dataLength, m_dataLength);
427 m_vol->dataCacheDirty();
428 break;
429
430 case EXFAT_TYPE_NAME:
431 break;
432
433 default:
434 DBG_FAIL_MACRO;
435 goto fail;
436 break;
437 }
438 checksum = exFatDirChecksum(cache, checksum);
439 }
440 df = reinterpret_cast<DirFile_t*>(
441 m_vol->dirCache(&m_dirPos, FsCache::CACHE_FOR_WRITE));
442 if (!df) {
443 DBG_FAIL_MACRO;
444 goto fail;
445 }
446 setLe16(df->setChecksum, checksum);
447 if (!m_vol->cacheSync()) {
448 DBG_FAIL_MACRO;
449 goto fail;
450 }
451 return true;
452
453fail:
454 m_error |= WRITE_ERROR;
455 return false;
456}

458bool ExFatFile::timestamp(uint8_t flags, uint16_t year, uint8_t month,
459 uint8_t day, uint8_t hour, uint8_t minute,
460 uint8_t second) {
461 DirFile_t* df;
462 uint8_t* cache;
463 uint16_t checksum = 0;
464 uint16_t date;
465 uint16_t time;
466 uint8_t ms10;
467
468 if (!isFile() || year < 1980 || year > 2107 || month < 1 || month > 12 ||
469 day < 1 || day > 31 || hour > 23 || minute > 59 || second > 59) {
470 DBG_FAIL_MACRO;
471 goto fail;
472 }
473 // update directory entry
474 if (!sync()) {
475 DBG_FAIL_MACRO;
476 goto fail;
477 }
478
479 date = FS_DATE(year, month, day);
480 time = FS_TIME(hour, minute, second);
481 ms10 = second & 1 ? 100 : 0;
482
483 for (uint8_t is = 0; is <= m_setCount; is++) {
484 cache = dirCache(is, FsCache::CACHE_FOR_READ);
485 if (!cache) {
486 DBG_FAIL_MACRO;
487 goto fail;
488 }
489 switch (cache[0]) {
490 case EXFAT_TYPE_FILE:
491 df = reinterpret_cast<DirFile_t*>(cache);
492 setLe16(df->attributes, m_attributes & FS_ATTRIB_COPY);
493 m_vol->dataCacheDirty();
494 if (flags & T_ACCESS) {
495 setLe16(df->accessTime, time);
496 setLe16(df->accessDate, date);
497 }
498 if (flags & T_CREATE) {
499 df->createTimeMs = ms10;
500 setLe16(df->createTime, time);
501 setLe16(df->createDate, date);
502 }
503 if (flags & T_WRITE) {
504 df->modifyTimeMs = ms10;
505 setLe16(df->modifyTime, time);
506 setLe16(df->modifyDate, date);
507 }
508 break;
509
510 case EXFAT_TYPE_STREAM:
511 break;
512
513 case EXFAT_TYPE_NAME:
514 break;
515
516 default:
517 DBG_FAIL_MACRO;
518 goto fail;
519 break;
520 }
521 checksum = exFatDirChecksum(cache, checksum);
522 }
523 df = reinterpret_cast<DirFile_t*>(
524 m_vol->dirCache(&m_dirPos, FsCache::CACHE_FOR_WRITE));
525 if (!df) {
526 DBG_FAIL_MACRO;
527 goto fail;
528 }
529 setLe16(df->setChecksum, checksum);
530 if (!m_vol->cacheSync()) {
531 DBG_FAIL_MACRO;
532 goto fail;
533 }
534 return true;
535
536fail:
537 return false;
538}

540bool ExFatFile::truncate() {
541 uint32_t toFree;
542 // error if not a normal file or read-only
543 if (!isWritable()) {
544 DBG_FAIL_MACRO;
545 goto fail;
546 }
547 if (m_firstCluster == 0) {
548 return true;
549 }
550 if (isContiguous()) {
551 uint32_t nc = 1 + ((m_dataLength - 1) >> m_vol->bytesPerClusterShift());
552 if (m_curCluster) {
553 toFree = m_curCluster + 1;
554 nc -= 1 + m_curCluster - m_firstCluster;
555 } else {
556 toFree = m_firstCluster;
557 m_firstCluster = 0;
558 }
559 if (nc && !m_vol->bitmapModify(toFree, nc, 0)) {
560 DBG_FAIL_MACRO;
561 goto fail;
562 }
563 } else {
564 // need to free chain
565 if (m_curCluster) {
566 toFree = 0;
567 int8_t fg = m_vol->fatGet(m_curCluster, &toFree);
568 if (fg < 0) {
569 DBG_FAIL_MACRO;
570 goto fail;
571 }
572 if (fg) {
573 // current cluster is end of chain
574 if (!m_vol->fatPut(m_curCluster, EXFAT_EOC)) {
575 DBG_FAIL_MACRO;
576 goto fail;
577 }
578 }
579 } else {
580 toFree = m_firstCluster;
581 m_firstCluster = 0;
582 }
583 if (toFree) {
584 if (!m_vol->freeChain(toFree)) {
585 DBG_FAIL_MACRO;
586 goto fail;
587 }
588 }
589 }
590 m_dataLength = m_curPosition;
591 m_validLength = m_curPosition;
592 m_flags |= FILE_FLAG_DIR_DIRTY;
593 return sync();
594
595fail:
596 return false;
597}

599size_t ExFatFile::write(const void* buf, size_t nbyte) {
600 // convert void* to uint8_t* - must be before goto statements
601 const uint8_t* src = reinterpret_cast<const uint8_t*>(buf);
602 uint8_t* cache;
603 uint8_t cacheOption;
604 uint16_t sectorOffset;
605 uint32_t sector;
606 uint32_t clusterOffset;
607
608 // number of bytes left to write - must be before goto statements
609 size_t toWrite = nbyte;
610 size_t n;
611 // error if not an open file or is read-only
612 if (!isWritable()) {
613 DBG_FAIL_MACRO;
614 goto fail;
615 }
616 // seek to end of file if append flag
617 if ((m_flags & FILE_FLAG_APPEND)) {
618 if (!seekSet(m_validLength)) {
619 DBG_FAIL_MACRO;
620 goto fail;
621 }
622 }
623 while (toWrite) {
624 clusterOffset = m_curPosition & m_vol->clusterMask();
625 sectorOffset = clusterOffset & m_vol->sectorMask();
626 if (clusterOffset == 0) {
627 // start of new cluster
628 if (m_curCluster != 0) {
629 int fg;
630
631 if (isContiguous()) {
632 uint32_t lc = m_firstCluster;
633 lc += (m_dataLength - 1) >> m_vol->bytesPerClusterShift();
634 if (m_curCluster < lc) {
635 m_curCluster++;
636 fg = 1;
637 } else {
638 fg = 0;
639 }
640 } else {
641 fg = m_vol->fatGet(m_curCluster, &m_curCluster);
642 if (fg < 0) {
643 DBG_FAIL_MACRO;
644 goto fail;
645 }
646 }
647 if (fg == 0) {
648 // add cluster if at end of chain
649 if (!addCluster()) {
650 DBG_FAIL_MACRO;
651 goto fail;
652 }
653 }
654 } else {
655 if (m_firstCluster == 0) {
656 // allocate first cluster of file
657 if (!addCluster()) {
658 DBG_FAIL_MACRO;
659 goto fail;
660 }
661 m_firstCluster = m_curCluster;
662 } else {
663 m_curCluster = m_firstCluster;
664 }
665 }
666 }
667 // sector for data write
668 sector = m_vol->clusterStartSector(m_curCluster) +
669 (clusterOffset >> m_vol->bytesPerSectorShift());
670
671 if (sectorOffset != 0 || toWrite < m_vol->bytesPerSector()) {
672 // partial sector - must use cache
673 // max space in sector
674 n = m_vol->bytesPerSector() - sectorOffset;
675 // lesser of space and amount to write
676 if (n > toWrite) {
677 n = toWrite;
678 }
679
680 if (sectorOffset == 0 && m_curPosition >= m_validLength) {
681 // start of new sector don't need to read into cache
682 cacheOption = FsCache::CACHE_RESERVE_FOR_WRITE;
683 } else {
684 // rewrite part of sector
685 cacheOption = FsCache::CACHE_FOR_WRITE;
686 }
687 cache = m_vol->dataCachePrepare(sector, cacheOption);
688 if (!cache) {
689 DBG_FAIL_MACRO;
690 goto fail;
691 }
692 uint8_t* dst = cache + sectorOffset;
693 memcpy(dst, src, n);
694 if (m_vol->bytesPerSector() == (n + sectorOffset)) {
695 // Force write if sector is full - improves large writes.
696 if (!m_vol->dataCacheSync()) {
697 DBG_FAIL_MACRO;
698 goto fail;
699 }
700 }
701#if USE_MULTI_SECTOR_IO
702 } else if (toWrite >= 2 * m_vol->bytesPerSector()) {
703 // use multiple sector write command
704 uint32_t ns = toWrite >> m_vol->bytesPerSectorShift();
705 // Limit writes to current cluster.
706 uint32_t maxNs = m_vol->sectorsPerCluster() -
707 (clusterOffset >> m_vol->bytesPerSectorShift());
708 if (ns > maxNs) {
709 ns = maxNs;
710 }
711 n = ns << m_vol->bytesPerSectorShift();
712 if (!m_vol->cacheSafeWrite(sector, src, ns)) {
713 DBG_FAIL_MACRO;
714 goto fail;
715 }
716#endif // USE_MULTI_SECTOR_IO
717 } else {
718 n = m_vol->bytesPerSector();
719 if (!m_vol->cacheSafeWrite(sector, src)) {
720 DBG_FAIL_MACRO;
721 goto fail;
722 }
723 }
724 m_curPosition += n;
725 src += n;
726 toWrite -= n;
727 if (m_curPosition > m_validLength) {
728 m_flags |= FILE_FLAG_DIR_DIRTY;
729 m_validLength = m_curPosition;
730 }
731 }
732 if (m_curPosition > m_dataLength) {
733 m_dataLength = m_curPosition;
734 // update fileSize and insure sync will update dir entry
735 m_flags |= FILE_FLAG_DIR_DIRTY;
736 } else if (FsDateTime::callback) {
737 // insure sync will update modified date and time
738 m_flags |= FILE_FLAG_DIR_DIRTY;
739 }
740 return nbyte;
741
742fail:
743 // return for write error
744 m_error |= WRITE_ERROR;
745 return 0;
746}
