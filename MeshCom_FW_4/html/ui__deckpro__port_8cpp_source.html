211void ui_wifi_get_scan_info(ui_wifi_scan_info_t *list, int list_len)
212{
213 int n = WiFi.scanNetworks();
214 if(n > list_len)
215 n = list_len;
216
217 memset(list, 0, (sizeof(*list) * list_len));
218 for(int i = 0; i < n; i++)
219 {
220 const char *str = WiFi.SSID(i).c_str();
221 if(is_chinese_utf8(str))
222 continue;
223 strncpy(list[i].name, WiFi.SSID(i).c_str(), 16);
224 list[i].rssi = WiFi.RSSI(i);
225 }
226}
