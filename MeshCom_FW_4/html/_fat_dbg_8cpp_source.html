48static void printHex(print_t* pr, uint8_t w, uint16_t h) {
49 char buf[5];
50 char* ptr = buf + sizeof(buf);
51 *--ptr = 0;
52 for (uint8_t i = 0; i < w; i++) {
53 char c = h & 0XF;
54 *--ptr = c < 10 ? c + '0' : c + 'A' - 10;
55 h >>= 4;
56 }
57 pr->write(ptr);
58}

96static bool printFatDir(print_t* pr, DirFat_t* dir) {
97 DirLfn_t* ldir = reinterpret_cast<DirLfn_t*>(dir);
98 if (!dir->name[0]) {
99 pr->println(F("Unused"));
100 return false;
101 } else if (dir->name[0] == FAT_NAME_DELETED) {
102 pr->println(F("Deleted"));
103 } else if (isFatFileOrSubdir(dir)) {
104 pr->print(F("SFN: "));
105 for (uint8_t i = 0; i < 11; i++) {
106 printHex(pr, dir->name[i]);
107 pr->write(' ');
108 }
109 pr->write(' ');
110 pr->write(dir->name, 11);
111 pr->println();
112 pr->print(F("attributes: 0X"));
113 printHexLn(pr, dir->attributes);
114 pr->print(F("caseFlags: 0X"));
115 printHexLn(pr, dir->caseFlags);
116 uint32_t fc = ((uint32_t)getLe16(dir->firstClusterHigh) << 16) |
117 getLe16(dir->firstClusterLow);
118 pr->print(F("firstCluster: "));
119 pr->println(fc, HEX);
120 pr->print(F("fileSize: "));
121 pr->println(getLe32(dir->fileSize));
122 } else if (isFatLongName(dir)) {
123 pr->print(F("LFN: "));
124 for (uint8_t i = 0; i < 13; i++) {
125 uint16_t c = getLfnChar(ldir, i);
126 if (15 < c && c < 128) {
127 pr->print(static_cast<char>(c));
128 } else {
129 pr->print("0X");
130 pr->print(c, HEX);
131 }
132 pr->print(' ');
133 }
134 pr->println();
135 pr->print(F("order: 0X"));
136 pr->println(ldir->order, HEX);
137 pr->print(F("attributes: 0X"));
138 pr->println(ldir->attributes, HEX);
139 pr->print(F("checksum: 0X"));
140 pr->println(ldir->checksum, HEX);
141 } else {
142 pr->println(F("Other"));
143 }
144 pr->println();
145 return true;
146}

148void FatFile::dmpFile(print_t* pr, uint32_t pos, size_t n) {
149 char text[17];
150 text[16] = 0;
151 if (n >= 0XFFF0) {
152 n = 0XFFF0;
153 }
154 if (!seekSet(pos)) {
155 return;
156 }
157 for (size_t i = 0; i <= n; i++) {
158 if ((i & 15) == 0) {
159 if (i) {
160 pr->write(' ');
161 pr->write(text);
162 if (i == n) {
163 break;
164 }
165 }
166 pr->write('\r');
167 pr->write('\n');
168 if (i >= n) {
169 break;
170 }
171 printHex(pr, 4, i);
172 pr->write(' ');
173 }
174 int16_t h = read();
175 if (h < 0) {
176 break;
177 }
178 pr->write(' ');
179 printHex(pr, 2, h);
180 text[i & 15] = ' ' <= h && h < 0X7F ? h : '.';
181 }
182 pr->write('\r');
183 pr->write('\n');
184}

241 uint16_t nf = fatType() == 16 ? 256 : fatType() == 32 ? 128 : 0;

243 pr->println(F("Invalid fatType"));
244 return;
245 }
246 pr->println(F("FAT:"));
247 uint32_t sector = m_fatStartSector + start;
248 uint32_t cluster = nf * start;
249 for (uint32_t i = 0; i < count; i++) {
250 uint8_t* pc = fatCachePrepare(sector + i, FsCache::CACHE_FOR_READ);
251 if (!pc) {
252 pr->println(F("cache read failed"));
253 return;
254 }
255 for (size_t k = 0; k < nf; k++) {
256 if (0 == cluster % 8) {
257 if (k) {
258 pr->println();
259 }
260 printHex(pr, cluster);
261 }
262 cluster++;
263 pr->write(' ');
264 uint32_t v = fatType() == 32 ? getLe32(pc + 4 * k) : getLe16(pc + 2 * k);
265 printHex(pr, v);
266 }
267 pr->println();
268 }
