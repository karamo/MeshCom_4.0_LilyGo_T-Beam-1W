33static char toUpper(char c) { return 'a' <= c && c <= 'z' ? c - 'a' + 'A' : c; }

35inline uint16_t exFatHash(char c, uint16_t hash) {
36 uint8_t u = toUpper(c);
37 hash = ((hash << 15) | (hash >> 1)) + u;
38 hash = ((hash << 15) | (hash >> 1));
39 return hash;
40}

42inline uint16_t exFatHash(uint16_t u, uint16_t hash) {
43 uint16_t c = toUpcase(u);
44 hash = ((hash << 15) | (hash >> 1)) + (c & 0XFF);
45 hash = ((hash << 15) | (hash >> 1)) + (c >> 8);
46 return hash;
47}

104size_t ExFatFile::getName8(char* name, size_t count) {
105 char* end = name + count;
106 char* str = name;
107 char* ptr;
108 DirName_t* dn;
109 uint16_t hs = 0;
110 uint32_t cp;
111 if (!isOpen()) {
112 DBG_FAIL_MACRO;
113 goto fail;
114 }
115 for (uint8_t is = 2; is <= m_setCount; is++) {
116 dn = reinterpret_cast<DirName_t*>(dirCache(is, FsCache::CACHE_FOR_READ));
117 if (!dn || dn->type != EXFAT_TYPE_NAME) {
118 DBG_FAIL_MACRO;
119 goto fail;
120 }
121 for (uint8_t in = 0; in < 15; in++) {
122 uint16_t c = getLe16(dn->unicode + 2 * in);
123 if (hs) {
124 if (!FsUtf::isLowSurrogate(c)) {
125 DBG_FAIL_MACRO;
126 goto fail;
127 }
128 cp = FsUtf::u16ToCp(hs, c);
129 hs = 0;
130 } else if (!FsUtf::isSurrogate(c)) {
131 if (c == 0) {
132 goto done;
133 }
134 cp = c;
135 } else if (FsUtf::isHighSurrogate(c)) {
136 hs = c;
137 continue;
138 } else {
139 DBG_FAIL_MACRO;
140 goto fail;
141 }
142 // Save space for zero byte.
143 ptr = FsUtf::cpToMb(cp, str, end - 1);
144 if (!ptr) {
145 DBG_FAIL_MACRO;
146 goto fail;
147 }
148 str = ptr;
149 }
150 }
151done:
152 *str = '\0';
153 return str - name;
154
155fail:
156 *name = 0;
157 return 0;
158}

160bool ExFatFile::hashName(ExName_t* fname) {
161 uint16_t hash = 0;
162 fname->reset();
163#if USE_UTF8_LONG_NAMES
164 fname->nameLength = 0;
165 while (!fname->atEnd()) {
166 uint16_t u = fname->get16();
167 if (u == 0XFFFF) {
168 DBG_FAIL_MACRO;
169 goto fail;
170 }
171 hash = exFatHash(u, hash);
172 fname->nameLength++;
173 }
174#else // USE_UTF8_LONG_NAMES
175 while (!fname->atEnd()) {
176 // Convert to byte for smaller exFatHash.
177 char c = fname->getch();
178 hash = exFatHash(c, hash);
179 }
180 fname->nameLength = fname->end - fname->begin;
181#endif // USE_UTF8_LONG_NAMES
182 fname->nameHash = hash;
183 if (!fname->nameLength || fname->nameLength > EXFAT_MAX_NAME_LENGTH) {
184 DBG_FAIL_MACRO;
185 goto fail;
186 }
187 return true;
188
189fail:
190 return false;
191}
