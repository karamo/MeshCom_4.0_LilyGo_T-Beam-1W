111 inline __attribute__((always_inline))
112 bool MODE_CPHA(uint8_t mode) {return (mode & 1) != 0;}
113 inline __attribute__((always_inline))
114 bool MODE_CPOL(uint8_t mode) {return (mode & 2) != 0;}
115 inline __attribute__((always_inline))
116 void receiveBit(uint8_t bit, uint8_t* data) {
117 if (MODE_CPHA(Mode)) {
118 fastDigitalWrite(SckPin, !MODE_CPOL(Mode));
119 }
120 nop;
121 nop;
122 fastDigitalWrite(SckPin,
123 MODE_CPHA(Mode) ? MODE_CPOL(Mode) : !MODE_CPOL(Mode));
124 if (fastDigitalRead(MisoPin)) *data |= 1 << bit;
125 if (!MODE_CPHA(Mode)) {
126 fastDigitalWrite(SckPin, MODE_CPOL(Mode));
127 }
128 }
129 //----------------------------------------------------------------------------
130 inline __attribute__((always_inline))
131 void sendBit(uint8_t bit, uint8_t data) {
132 if (MODE_CPHA(Mode)) {
133 fastDigitalWrite(SckPin, !MODE_CPOL(Mode));
134 }
135 fastDigitalWrite(MosiPin, data & (1 << bit));
136 fastDigitalWrite(SckPin,
137 MODE_CPHA(Mode) ? MODE_CPOL(Mode) : !MODE_CPOL(Mode));
138 nop;
139 nop;
140 if (!MODE_CPHA(Mode)) {
141 fastDigitalWrite(SckPin, MODE_CPOL(Mode));
142 }
143 }
144 //----------------------------------------------------------------------------
145 inline __attribute__((always_inline))
146 void transferBit(uint8_t bit, uint8_t* rxData, uint8_t txData) {
147 if (MODE_CPHA(Mode)) {
148 fastDigitalWrite(SckPin, !MODE_CPOL(Mode));
149 }
150 fastDigitalWrite(MosiPin, txData & (1 << bit));
151 fastDigitalWrite(SckPin,
152 MODE_CPHA(Mode) ? MODE_CPOL(Mode) : !MODE_CPOL(Mode));
153 if (fastDigitalRead(MisoPin)) *rxData |= 1 << bit;
154 if (!MODE_CPHA(Mode)) {
155 fastDigitalWrite(SckPin, MODE_CPOL(Mode));
156 }
157 }
158 //----------------------------------------------------------------------------
159};
