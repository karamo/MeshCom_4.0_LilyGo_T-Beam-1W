27void BaseDisplay::begin() {
28
29 // Only begin() once
30 if (begun)
31 return;
32 else
33 begun = true;
34
35 // Set the digital pins that supplement the SPI interface
36 pinMode(pin_cs, OUTPUT);
37 pinMode(pin_dc, OUTPUT);
38 pinMode(pin_busy, INPUT); // NOTE: do not use internal pullups: incompatible logic levels
39
40 // Prepare SPI
41 display_spi = Platform::getSPI();
42 digitalWrite(pin_cs, HIGH);
43 Platform::beginSPI(display_spi, pin_sdi, pin_miso, pin_clk);
44
45 // SAMD21: change SPI pins if requested
46 #ifdef __SAMD21G18A__
47 if (pin_sdi != DEFAULT_SDI || pin_clk != DEFAULT_CLK)
48 pin_miso = Platform::setSPIPins(pin_sdi, pin_clk);
49 #endif
50
51 // All in one boards: power on the peripherals, and hard reset to clear away whatever traumatic experience the display has been through
52 #if ALL_IN_ONE
53 Platform::VExtOn();
54 Platform::toggleResetPin();
55 wait();
56 #endif
57
58 // If PRESERVE_IMAGE possible, memory was allocated in constructor
59 if (PRESERVE_IMAGE && pagefile_height == panel_height)
60 writePage(); // Make sure display memory is also blanked
61}

83void BaseDisplay::initDrawingParams() {
84 // Default drawing config must be set early, as user-config may be issued before begin() is auto-called.
85
86 // Calculate pagefile size
87 pagefile_height = constrain(pagefile_height, (uint16_t)1, (uint16_t)MAX_PAGE_HEIGHT);
88 page_bytecount = panel_width * pagefile_height / 8;
89
90 // If unpaged drawing possible, allocate the memory, and set library to draw fullscreen
91 if (PRESERVE_IMAGE && pagefile_height == panel_height)
92 grabPageMemory();
93
94 // Default GFX options
95 fullscreen();
96 setBackgroundColor(WHITE);
97 setTextColor(BLACK);
98}
