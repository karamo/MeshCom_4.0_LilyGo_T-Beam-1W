135bool istream::getDouble(double* value) {
136 bool got_digit = false;
137 bool got_dot = false;
138 bool neg;
139 int16_t c;
140 bool expNeg = false;
141 int16_t exp = 0;
142 int16_t fracExp = 0;
143 uint32_t frac = 0;
144 pos_t endPos;
145 double pow10;
146 double v;
147
148 getpos(&endPos);
149 c = readSkip();
150 neg = c == '-';
151 if (c == '-' || c == '+') {
152 c = getch();
153 }
154 while (1) {
155 if (isdigit(c)) {
156 got_digit = true;
157 if (frac < uint32_max / 10) {
158 frac = frac * 10 + (c - '0');
159 if (got_dot) {
160 fracExp--;
161 }
162 } else {
163 if (!got_dot) {
164 fracExp++;
165 }
166 }
167 } else if (!got_dot && c == '.') {
168 got_dot = true;
169 } else {
170 break;
171 }
172 if (fracExp < -EXP_LIMIT || fracExp > EXP_LIMIT) {
173 goto fail;
174 }
175 c = getch(&endPos);
176 }
177 if (!got_digit) {
178 goto fail;
179 }
180 if (c == 'e' || c == 'E') {
181 c = getch();
182 expNeg = c == '-';
183 if (c == '-' || c == '+') {
184 c = getch();
185 }
186 while (isdigit(c)) {
187 if (exp > EXP_LIMIT) {
188 goto fail;
189 }
190 exp = exp * 10 + (c - '0');
191 c = getch(&endPos);
192 }
193 }
194 v = static_cast<double>(frac);
195 exp = expNeg ? fracExp - exp : fracExp + exp;
196 expNeg = exp < 0;
197 if (expNeg) {
198 exp = -exp;
199 }
200 pow10 = 10.0;
201 while (exp) {
202 if (exp & 1) {
203 if (expNeg) {
204 // check for underflow
205 if (v < DBL_MIN * pow10 && frac != 0) {
206 goto fail;
207 }
208 v /= pow10;
209 } else {
210 // check for overflow
211 if (v > DBL_MAX / pow10) {
212 goto fail;
213 }
214 v *= pow10;
215 }
216 }
217 pow10 *= pow10;
218 exp >>= 1;
219 }
220 setpos(&endPos);
221 *value = neg ? -v : v;
222 return true;
223
224fail:
225 // error restore position to last good place
226 setpos(&endPos);
227 setstate(failbit);
228 return false;
229}

261bool istream::getNumber(uint32_t posMax, uint32_t negMax, uint32_t* num) {
262 int16_t c;
263 int8_t any = 0;
264 int8_t have_zero = 0;
265 uint8_t neg;
266 uint32_t val = 0;
267 uint32_t cutoff;
268 uint8_t cutlim;
269 pos_t endPos;
270 uint8_t f = flags() & basefield;
271 uint8_t base = f == oct ? 8 : f != hex ? 10 : 16;
272 getpos(&endPos);
273 c = readSkip();
274
275 neg = c == '-' ? 1 : 0;
276 if (c == '-' || c == '+') {
277 c = getch();
278 }
279
280 if (base == 16 && c == '0') { // TESTSUITE
281 c = getch(&endPos);
282 if (c == 'X' || c == 'x') {
283 c = getch();
284 // remember zero in case no hex digits follow x/X
285 have_zero = 1;
286 } else {
287 any = 1;
288 }
289 }
290 // set values for overflow test
291 cutoff = neg ? negMax : posMax;
292 cutlim = cutoff % base;
293 cutoff /= base;
294
295 while (1) {
296 if (isdigit(c)) {
297 c -= '0';
298 } else if (isalpha(c)) {
299 c -= isupper(c) ? 'A' - 10 : 'a' - 10;
300 } else {
301 break;
302 }
303 if (c >= base) {
304 break;
305 }
306 if (val > cutoff || (val == cutoff && c > cutlim)) {
307 // indicate overflow error
308 any = -1;
309 break;
310 }
311 val = val * base + c;
312 c = getch(&endPos);
313 any = 1;
314 }
315 setpos(&endPos);
316 if (any > 0 || (have_zero && any >= 0)) {
317 *num = neg ? -val : val;
318 return true;
319 }
320 setstate(failbit);
321 return false;
322}
