63void BHI260AP_get_val(int val_type, float *x, float *y, float *z)
64{
65 bhy.update();

67 switch (val_type) {
68 case 1:
69 *x = accel_data.x * accel_factor;
70 *y = accel_data.y * accel_factor;
71 *z = accel_data.z * accel_factor;

73 case 2 :
74 *x = gyro_data.x * gyro_factor;
75 *y = gyro_data.y * gyro_factor;
76 *z = gyro_data.z * gyro_factor;

91bool BHI260AP_init(void)
92{
93 // Set the reset pin and interrupt pin, if any
94 bhy.setPins(BOARD_GYROSCOPDE_RST, BOARD_GYROSCOPDE_INT);
95
96 if (!bhy.init(Wire, BOARD_I2C_SDA, BOARD_I2C_SCL, BHI260AP_SLAVE_ADDRESS_L)) {
97 Serial.print("Failed to init BHI260AP - ");
98 Serial.println(bhy.getError());
99 return false;
100 }
101
102 // Output all available sensors to Serial
103 // bhy.printSensors(Serial);
104
105 float sample_rate = 100.0; /* Read out hintr_ctrl measured at 100Hz */
106 uint32_t report_latency_ms = 0; /* Report immediately */
107
108 // Enable acceleration
109 bhy.configure(SENSOR_ID_ACC_PASS, sample_rate, report_latency_ms);
110 // Enable gyroscope
111 bhy.configure(SENSOR_ID_GYRO_PASS, sample_rate, report_latency_ms);
112
113 // Set the acceleration sensor result callback function
114 bhy.onResultEvent(SENSOR_ID_ACC_PASS, accel_process_callback);
115
116 // Set the gyroscope sensor result callback function
117 bhy.onResultEvent(SENSOR_ID_GYRO_PASS, gyro_process_callback);
118
119 // Set the Magnetometer sensor result callback function
120 bhy.onResultEvent(SENSOR_ID_MAG_PASS, magn_process_callback);
121
122 return true;
123}
