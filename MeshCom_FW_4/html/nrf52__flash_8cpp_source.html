258boolean save_settings(void)
259{
260 bool result = true;
261 // Read saved content
262 // printf("settings_name:%s\n", settings_name);
263
264 lora_file.open(settings_name, FILE_O_READ);
265 if (!lora_file)
266 {
267 DEBUG_MSG("FLASH", "File doesn't exist, force format");
268 delay(100);
269 flash_reset();
270 lora_file.open(settings_name, FILE_O_READ);
271 }
272 lora_file.read((uint8_t *)&g_flash_content, sizeof(s_meshcom_settings));
273 lora_file.close();
274
275 DEBUG_MSG("FLASH", "close");
276
277 //printf("g_flash_content%s\n", g_flash_content.node_call);
278
279 if (memcmp((void *)&g_flash_content, (void *)&meshcom_settings, sizeof(s_meshcom_settings)) != 0)
280 {
281 DEBUG_MSG("FLASH", "Flash content changed, writing new data");
282 delay(100);
283
284 InternalFS.remove(settings_name);
285
286 if (lora_file.open(settings_name, FILE_O_WRITE))
287 {
288 lora_file.write((uint8_t *)&meshcom_settings, sizeof(s_meshcom_settings));
289 lora_file.flush();
290 }
291 else
292 {
293 result = false;
294 }
295 lora_file.close();
296 }
297
298 log_settings();
299
300 //Test only Serial.println("\nflash save...");
301
302 return result;
303}
