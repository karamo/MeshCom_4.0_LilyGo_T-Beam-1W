273{
274 //Serial.println("read_batt");
275
276 is_receiving = true;
277
278 float raw = 0.0;
279
280 #if defined(NRF52_SERIES)
281
282 analogReference(AR_INTERNAL_3_0);
283
284 uint32_t adc_reading = 0;
285 //Multisampling
286 for (int i = 0; i < NO_OF_SAMPLES; i++)
287 {
288 int raw = analogRead(vbat_pin);
289 adc_reading += raw;
290 }
291
292 adc_reading /= NO_OF_SAMPLES;
293
294 raw = (float)adc_reading;
295
296 //Serial.printf("Raw: %d\n", adc_reading);
297
298 #elif defined(BOARD_TLORA_OLV216)
299
300 pinMode(23, OUTPUT);
301 pinMode(vbat_pin, INPUT);
302
303 raw = (float)(analogRead(vbat_pin)) / 4095*2*3.3*1.1;
304
305 // Serial.printf("ADC analog value = <%f>\n", raw);
306
307 #elif defined(BOARD_E290)
308
309 uint16_t battery_levl = analogRead(vbat_pin);
310
311 if(bDisplayCont)
312 Serial.printf("ADC analog value = <%i>\n", battery_levl);
313
314 raw = (float)battery_levl;
315
316 #elif defined(BOARD_T_DECK) || defined(BOARD_T_DECK_PLUS)
317
318 uint16_t adcValue = analogRead(vbat_pin);
319 raw = adcValue;
320
321 // TRACKER
322 #elif defined(BOARD_TRACKER)
323 #ifdef BATTERY_PIN
324 int adc_value = analogRead(BATTERY_PIN);
325
326 double voltage = (adc_value * 3.3 ) / 4095.0;
327
328 double inputDivider = (1.0 / (390.0 + 100.0)) * 100.0; // The voltage divider is a 390k + 100k resistor in series, 100k on the low side.
329
330 float milliVolt = ((voltage / inputDivider) + 0.285)*1000.0;
331
332 is_receiving = false;
333
334 return milliVolt; // Yes, this offset is excessive, but the ADC on the ESP32s3 is quite inaccurate and noisy. Adjust to own measurements.
335 #else
336 is_receiving = false;
337 return (float)0.0;
338 #endif
339 #elif defined(BOARD_HELTEC_V3) || defined(BOARD_STICK_V3)
340
341 // ADC resolution
342 const int resolution = 12;
343 const int adcMax = pow(2,resolution) - 1;
344 const float adcMaxVoltage = 3.3;
345 // On-board voltage divider
346 const int R1 = 390;
347 const int R2 = 100;
348 // Calibration measurements
349 const float measuredVoltage = 4.2;
350 const float reportedVoltage = 4.095;
351 // Calibration factor
352 const float factor = (adcMaxVoltage / adcMax) * ((R1 + R2)/(float)R2) * (measuredVoltage / reportedVoltage);
353
354 //V3.1 digitalWrite(ADC_CTRL_PIN,LOW);
355 digitalWrite(ADC_CTRL_PIN, HIGH);
356
357 delay(100);
358 int analogValue = analogRead(vbat_pin);
359
360 //V3.1 digitalWrite(ADC_CTRL_PIN, HIGH);
361 digitalWrite(ADC_CTRL_PIN, LOW);
362
363 float floatVoltage = factor * analogValue;
364 uint16_t voltage = (int)(floatVoltage * 1000.0);
365
366 if(bDEBUG && bDisplayInfo)
367 {
368 Serial.print("[readBatteryVoltage] ADC : ");
369 Serial.println(analogValue);
370 Serial.print("[readBatteryVoltage] Float : ");
371 Serial.println(floatVoltage,3);
372 Serial.print("[readBatteryVoltage] milliVolts : ");
373 Serial.println(voltage);
374 }
375
376 raw = floatVoltage * 1000.0;
377
378 #elif defined(BOARD_E22_S3)
379
380 uint16_t analogValue = analogReadMilliVolts(BATTERY_PIN);
381
382 raw = (float)analogValue * fBattFaktor;
383
384 if(bDisplayCont)
385 {
386 Serial.printf("%s [BATT]...reading: %u factor: %.4f voltage: %.2f mV\n", getTimeString().c_str(), analogValue, fBattFaktor, raw);
387 }
388
389 #else
390
391 int adc_value = 0;
392 int adc_reading = 0;
393
394 //Multisampling
395 for (int i = 0; i < NO_OF_SAMPLES; i++)
396 {
397 if (unit == ADC_UNIT_1)
398 adc_value = adc1_get_raw((adc1_channel_t)channel);
399 else
400 adc2_get_raw((adc2_channel_t)channel, width, &adc_value);
401
402 adc_reading += adc_value;
403 }
404
405 adc_reading /= NO_OF_SAMPLES;
406
407 //Convert adc_reading to voltage in mV
408 uint32_t voltage = esp_adc_cal_raw_to_voltage(adc_reading, adc_chars);
409
410 raw = voltage;
411
412 #endif
413
414 // take it als mV
415 #if defined(NRF52_SERIES)
416 raw = raw * 1.25717;
417 #elif defined(BOARD_TBEAM) || defined(BOARD_SX1268)
418 raw = raw * 10.7687;
419 #elif defined(BOARD_HELTEC)
420 raw = raw * 24.80;
421 #elif defined(BOARD_HELTEC_V3) || defined(BOARD_STICK_V3) || defined(BOARD_TRACKER)
422 // all done
423 #elif defined(BOARD_E22_S3)
424 // all done
425 #elif defined(BOARD_TLORA_OLV216)
426 raw = raw * 1000.0; // convert to volt
427 #elif defined(BOARD_E290)
428 raw = raw * 4.13173653;
429 #elif defined(BOARD_T_DECK) || defined(BOARD_T_DECK_PLUS)
430 raw = raw * 1.7209; //1.66051;
431 #else
432 raw = raw * 24.80;
433 #endif
434
435 if(bDisplayCont)
436 {
437 Serial.print("[readBatteryVoltage] raw mV : ");
438 Serial.println(raw);
439 }
440
441 is_receiving = false;
442
443 return raw;
444}

460uint8_t mv_to_percent(float mvolts)
461{
462 if (mvolts < max_batt * 0.785F) // 80% (3300)
463 return 0;
464
465 if (mvolts < max_batt * 0.857F) // (3600)
466 {
467 mvolts -= max_batt * 0.785F; // (3300)
468 return mvolts / 30;
469 }
470
471 if (mvolts > max_batt) // 4200
472 {
473 return 100;
474 }
475
476 mvolts -= max_batt * 0.857F; // 3600
477 uint8_t rproz = 10 + (mvolts * 0.15F);
478
479 return rproz;
