62 int ADCatten = (int)meshcom_settings.node_analog_atten;
63 float Atten = 1.0;
64 switch (ADCatten) {
65 case ADC_ATTEN_DB_0: { Atten = 1.0000; break; }
66 case ADC_ATTEN_DB_2_5: { Atten = 1.3335; break; }
67 case ADC_ATTEN_DB_6: { Atten = 1.9953; break; }
68 case ADC_ATTEN_DB_12: { Atten = 3.9811; break; }
69 }
70 int ADCintervall = (int)meshcom_settings.node_analog_alpha % 100;
71 if ((analog_oversample_timer + std::max(2,ADCintervall)) < millis()) //min. 2ms, max. 99ms
72 {
73 //digitalWrite(BOARD_LED, LOW); // OE3WAS für TEST Timing
74 //digitalWrite(BOARD_LED, HIGH); // OE3WAS für TEST
75 //digitalWrite(BOARD_LED, LOW); // OE3WAS für TEST

77 // optimal wäre es, den Wert von Vref aus der EFUSE auslesen zu können
78 // aber nicht jeder ESP32/ESP32-S3 enthält so einen Wert
79 float vref = 1100.0; // bereich 1000 .. 1200 [mV]
80 ADCalpha = meshcom_settings.node_analog_alpha - (int)meshcom_settings.node_analog_alpha; // 0.001 .. 0.999
81
82 ADCraw = analogReadRaw(meshcom_settings.node_analog_pin);
83 SampleCount++;
84 // 12bit value [0 .. 4095] als default angenommen, muss ggf angepasst werden
85 float raw = (float)ADCraw * (vref / 4095.0) * meshcom_settings.node_analog_faktor; // raw Wert [mV] nach Kalibrierung
86 raw *= Atten; // Abschwächer berücksichtigen
87 raw = ADCslope * raw + ADCoffset; // hier jetzt Offset & Slope verarbeiten
88 if (ADCexp1pre==0) {ADCexp1pre = raw;} //langsamen Start beschleunigen
89 if (ADCexp12pre==0) {ADCexp12pre = raw;}
90
91 // Glättung berechnen
92 ADCexp1 = ADCalpha * raw + (1.0-ADCalpha) * ADCexp1pre;
93 ADCexp12 = ADCalpha * ADCexp1 + (1.0-ADCalpha) * ADCexp12pre;
94 ADCexp2 = ((2.0-ADCalpha) * ADCexp1 - ADCexp12) / (1.0-ADCalpha);
95
96 ADCexp1pre = ADCexp1;
97 ADCexp12pre = ADCexp12;
98 analog_oversample_timer = millis();
99
100 //digitalWrite(BOARD_LED, HIGH); // OE3WAS für TEST
