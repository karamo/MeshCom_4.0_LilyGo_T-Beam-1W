38void setBrightness(uint8_t value)
39{
40 if (current_brightness_level == value) {
41 // just reset to same level - nothing to do
42 return;
43 }
44
45 if (value > BRIGHTNESS_STEPS) {
46 // cycle through
47 value = value % BRIGHTNESS_STEPS;
48 }
49
50 if (value == 0)
51 {
52 // turn display off
53 digitalWrite(TDECK_TFT_BACKLIGHT, 0);
54 delay(3);
55 current_brightness_level = 0;
56 return;
57 }
58
59 if (current_brightness_level == 0)
60 {
61 // display off - turn on first
62 digitalWrite(TDECK_TFT_BACKLIGHT, 1);
63 current_brightness_level = BRIGHTNESS_STEPS;
64 delayMicroseconds(30);
65 }
66
67 int from = BRIGHTNESS_STEPS - current_brightness_level;
68 int to = BRIGHTNESS_STEPS - value;
69 int num = (BRIGHTNESS_STEPS + to - from) % BRIGHTNESS_STEPS;
70
71 for (int i = 0; i < num; i++)
72 {
73 digitalWrite(TDECK_TFT_BACKLIGHT, 0);
74 digitalWrite(TDECK_TFT_BACKLIGHT, 1);
75 }
76

78 pre_sleep_brightness_level = current_brightness_level;
