29const uint16_t MIN_2X_BAUD = F_CPU / (4 * (2 * 0XFFF + 1)) + 1;

33void MinimumSerial::begin(uint32_t baud) {
34 uint16_t baud_setting;
35 // don't worry, the compiler will squeeze out F_CPU != 16000000UL
36 if ((F_CPU != 16000000UL || baud != 57600) && baud > MIN_2X_BAUD) {
37 // Double the USART Transmission Speed
38 UCSR0A = 1 << U2X0;
39 baud_setting = (F_CPU / 4 / baud - 1) / 2;
40 } else {
41 // hardcoded exception for compatibility with the bootloader shipped
42 // with the Duemilanove and previous boards and the firmware on the 8U2
43 // on the Uno and Mega 2560.
44 UCSR0A = 0;
45 baud_setting = (F_CPU / 8 / baud - 1) / 2;
46 }
47 // assign the baud_setting
48 UBRR0H = baud_setting >> 8;
49 UBRR0L = baud_setting;
50 // enable transmit and receive
51 UCSR0B |= (1 << TXEN0) | (1 << RXEN0);
52}
