67void ostream::putDouble(double n) {
68 uint8_t nd = precision();
69 double round = 0.5;
70 char sign;
71 char buf[13]; // room for sign, 10 digits, '.', and zero byte
72 char *ptr = buf + sizeof(buf) - 1;
73 char *str = ptr;
74 // terminate string
75 *ptr = '\0';
76
77 // get sign and make nonnegative
78 if (n < 0.0) {
79 sign = '-';
80 n = -n;
81 } else {
82 sign = flags() & showpos ? '+' : '\0';
83 }
84 // check for larger than uint32_t
85 if (n > 4.0E9) {
86 putPgm(PSTR("BIG FLT"));
87 return;
88 }
89 // round up and separate int and fraction parts
90 for (uint8_t i = 0; i < nd; ++i) {
91 round *= 0.1;
92 }
93 n += round;
94 uint32_t intPart = n;
95 double fractionPart = n - intPart;
96
97 // format intPart and decimal point
98 if (nd || (flags() & showpoint)) {
99 *--str = '.';
100 }
101 str = fmtNum(intPart, str, 10);
102
103 // calculate length for fill
104 uint8_t len = sign ? 1 : 0;
105 len += nd + ptr - str;
106
107 // extract adjust field
108 fmtflags adj = flags() & adjustfield;
109 if (adj == internal) {
110 if (sign) {
111 putch(sign);
112 }
113 do_fill(len);
114 } else {
115 // do fill for right
116 fill_not_left(len);
117 if (sign) {
118 *--str = sign;
119 }
120 }
121 putstr(str);
122 // output fraction
123 while (nd-- > 0) {
124 fractionPart *= 10.0;
125 int digit = static_cast<int>(fractionPart);
126 putch(digit + '0');
127 fractionPart -= digit;
128 }
129 // do fill if not done above
130 do_fill(len);
131}
