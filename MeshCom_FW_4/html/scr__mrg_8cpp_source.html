57static void scr_mgr_active(scr_card_t *card) // 设置屏幕卡片为活跃状态

130bool scr_mgr_switch(int id, bool anim) // 清空栈，然后切换到指定 id的屏幕卡片上
131{
132 scr_card_t *tgt_card = scr_mgr_find_by_id(id);
133 scr_card_t *stack_scr = NULL;
134 lv_obj_t *curr_obj = NULL;
135
136 if(tgt_card == NULL) // 没有找到该屏幕
137 return false;
138
139 if(scr_stack_top != NULL) { // 如果有多张屏幕卡片叠在一起，就先记录顶层卡片
140 curr_obj = scr_stack_top->obj;
141 stack_scr = scr_stack_top->prev;
142 scr_mgr_remove(scr_stack_top);
143 lv_mem_free((void *)scr_stack_top);
144 scr_stack_top = stack_scr;
145 }
146 while(scr_stack_top != NULL) { // 然后清除所有卡片
147 stack_scr = scr_stack_top->prev;
148 curr_obj = scr_stack_top->obj;
149 scr_mgr_remove(scr_stack_top);
150 lv_mem_free((void *)scr_stack_top);
151 scr_stack_top = stack_scr;
152 }
153
154 stack_scr = (scr_card_t *)lv_mem_alloc(sizeof(scr_card_t));
155 stack_scr->id = tgt_card->id;
156 // stack_scr->obj = tgt_card->life->create(NULL);
157 stack_scr->obj = scr_mgr_default_style(tgt_card);
158 stack_scr->st = SCR_MGR_STATE_CREATED;
159 stack_scr->life = tgt_card->life;
160 stack_scr->prev = NULL;
161 stack_scr->next = NULL;
162 scr_stack_root = stack_scr;
163 scr_stack_top = stack_scr;
164
165 scr_mgr_active(stack_scr); // 设置屏幕卡片为活跃状态
166
167 if(scr_anim_sw != LV_SCR_LOAD_ANIM_NONE && anim){
168 lv_scr_load_anim(stack_scr->obj, scr_anim_sw, scr_anim_time, 0, true);
169 } else{
170 lv_scr_load(stack_scr->obj);
171 if(curr_obj)
172 lv_obj_del(curr_obj);
173 }
174 return true;
175}

177bool scr_mgr_push(int id, bool anim)
178{
179 scr_card_t *tgt_card = scr_mgr_find_by_id(id);
180 scr_card_t *stack_scr = NULL;
181
182 if(tgt_card == NULL){ // 没有找到该屏幕
183 return false;
184 }
185
186 if((scr_stack_top != NULL) && (tgt_card->id == scr_stack_top->id)){ // 当前push屏幕和最顶层屏幕一样
187 return false;
188 }
189
190 stack_scr = (scr_card_t *)lv_mem_alloc(sizeof(scr_card_t));
191 stack_scr->id = tgt_card->id;
192 // stack_scr->obj = tgt_card->life->create(NULL);
193 stack_scr->obj = scr_mgr_default_style(tgt_card);
194 stack_scr->st = SCR_MGR_STATE_CREATED;
195 stack_scr->life = tgt_card->life;
196 if(scr_stack_top == NULL){
197 stack_scr->prev = NULL;
198 stack_scr->next = NULL;
199 scr_stack_root = stack_scr;
200 scr_stack_top = stack_scr;
201 } else{
202 scr_mgr_inactive(scr_stack_top);
203 scr_stack_top->next = stack_scr;
204 stack_scr->prev = scr_stack_top;
205 scr_stack_top->next = NULL;
206 scr_stack_top = stack_scr;
207 }
208
209 scr_mgr_active(stack_scr);
210
211 if(scr_anim_push != LV_SCR_LOAD_ANIM_NONE && anim){
212 lv_scr_load_anim(stack_scr->obj, scr_anim_push, scr_anim_time, 0, false);
213 } else{
214 lv_scr_load(stack_scr->obj);
215 }
216 return true;
217}

219bool scr_mgr_pop(bool anim)
220{
221 scr_card_t *dst_item = NULL;
222 lv_obj_t *cur_obj;
223
224 if(scr_stack_top == NULL || scr_stack_top == scr_stack_root){
225 return false;
226 }
227
228 cur_obj = scr_stack_top->obj;
229 dst_item = scr_stack_top->prev;
230 scr_mgr_remove(scr_stack_top);
231 lv_mem_free((void *)scr_stack_top);
232 scr_stack_top = dst_item;
233
234 scr_mgr_active(dst_item);
235
236 if(scr_anim_pop != LV_SCR_LOAD_ANIM_NONE && anim){
237 lv_scr_load_anim(dst_item->obj, scr_anim_pop, scr_anim_time, 0, true);
238 } else{
239 lv_scr_load(dst_item->obj);
240 if (cur_obj) {
241 lv_obj_del(cur_obj);
242 }
243 }
244 return false;
245}
