564{
565 byte buffer[1];

567 #if defined (BOARD_HELTEC_V3)

602 }

610 }
611

615

617 // 0x16 == T-BEAM 1.3" 1306
618 // 0x28 == E22 1.3" 1306
619 // 0x28 == T-BEAM 1.3" SUPREME 1306
620
621 // 0x03 == T-BEAM 0.9" = SSD1306
622 // 0x04 == T-LORA 0.9" 868MHz = SH1106 OE3WAS
623 // 0x06 == SSD1306
624 // 0x07 == T-LORA 0.9" type 1 = SSD1306
625 // 0x07 == T-LORA 0.9" type 2 = SSD1306
626 // 0x09 == HELTEC V3 type 1
627 // 0x3F == HELTEC V3 type 2
628
629 // check 1.3"

631 if(checkByte == 0x28 || checkByte == 0x16 || checkByte == 0x00)
632 {
633 Serial.println(F("[INIT]...OLED Display is SSD1306"));
634 return 2;
635 }
636
637 // cheched 0.9"
638 Serial.println(F("[INIT]...OLED Display is SH1106"));
