334void ElegantOTAClass::loop() {
335 // Check if 2 seconds have passed since _reboot_request_millis was set
336 if (_reboot && millis() - _reboot_request_millis > 2500) {
337 _reboot = false;
338 Serial.println("Rebooting...\n");
339 #if defined(ESP8266) || defined(ESP32)
