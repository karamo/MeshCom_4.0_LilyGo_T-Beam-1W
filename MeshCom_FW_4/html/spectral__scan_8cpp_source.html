1/*
2 RadioLib SX126x Spectrum Scan Example
3
4 This example shows how to perform a spectrum power scan using SX126x.
5 The output is in the form of scan lines, each line has 33 power bins.
6 First power bin corresponds to -11 dBm, the second to -15 dBm and so on.
7 Higher number of samples in a bin corresponds to more power received
8 at that level. The example performs frequency sweep over a given range.
9
10 To show the results in a plot, run the Python script
11 RadioLib/extras/SX126x_Spectrum_Scan/SpectrumScan.py
12
13 WARNING: This functionality is experimental and requires a binary patch
14 to be uploaded to the SX126x device. There may be some undocumented
15 side effects!
16
17 For default module settings, see the wiki page
18 https://github.com/jgromes/RadioLib/wiki/Default-configuration#sx126x---lora-modem
19
20 For full API reference, see the GitHub Pages
21 https://jgromes.github.io/RadioLib/
22*/

50int sx126x_spectral_init_scan(float freq)
51{
52 // initialize SX1262 FSK modem at the initial frequency
53 int state = radio.beginFSK(freq);
54 if (state != RADIOLIB_ERR_NONE)
55 {
56 return state;
57 }
58
59 // upload a patch to the SX1262 to enable spectral scan
60 // NOTE: this patch is uploaded into volatile memory,
61 // and must be re-uploaded on every power up
62 state = radio.uploadPatch(sx126x_patch_scan, sizeof(sx126x_patch_scan));
63 if (state != RADIOLIB_ERR_NONE)
64 {
65 return state;
66 }
67
68 // configure scan bandwidth to 234.4 kHz
69 // and disable the data shaping
70 state = radio.setRxBandwidth(234.3); //any other frequwency does not work - maybe those values are somehow fixed in the binary blob of the scan patch
71 //ToDo: check if that patch can be recompiled allowing a narrower bandwidth, see: https://github.com/Lora-net/sx1302_hal/blob/master/util_spectral_scan/src/spectral_scan.c
72 state |= radio.setDataShaping(RADIOLIB_SHAPING_NONE);
73 if (state != RADIOLIB_ERR_NONE)
74 {
75 return state;
76 }
77
78 // if we reach this point, everythung should be okay
79 return RADIOLIB_ERR_NONE;
80}

106uint16_t *sx126x_spectral_scan_freq(float freq, uint16_t samples)
107{
108 uint16_t *results = new uint16_t[RADIOLIB_SX126X_SPECTRAL_SCAN_RES_SIZE]; // initialize array with zeros, we will return the array containing zeros if anything goes wrong
109 if (radio.setFrequency(freq) == RADIOLIB_ERR_NONE)
110 {
111 // start spectral scan
112 if (radio.spectralScanStart(samples) == RADIOLIB_ERR_NONE)
113 { // if scan started without an error, wait for spectral scan to finish
114 while (radio.spectralScanGetStatus() != RADIOLIB_ERR_NONE){
115 delay(10);
116 }
117
118 // read the results
119 if (radio.spectralScanGetResult(results) == RADIOLIB_ERR_NONE){
120 return results;
121 }
122 }
123 }
124
125 // if we reach this point, something went wrong
126 return results; // return an array of zeros
127}

138void sx126x_spectral_scan()
139{
140
141 float freqStart = 430.0; // scan start frequency in MHz
142 const float freqEnd = 440.2; // scan end frequency in MHz
143 const float samples = 2048; // amount of samples to be taken
144
145 sx126x_spectral_init_scan(freqStart);
146
147 while (freqStart <= freqEnd)
148 {
149 Serial.print("FREQ ");
150 Serial.println(freqStart, 2);
151 Serial.print(F("[SX1262] Starting spectral scan ... \n"));
152 // get the results
153 uint16_t *res;
154 res = sx126x_spectral_scan_freq(freqStart, samples);
155
156 // we have some results, print it
157 Serial.print("SCAN ");
158 for (uint8_t i = 0; i < RADIOLIB_SX126X_SPECTRAL_SCAN_RES_SIZE; i++)
159 {
160 Serial.print(res[i]);
161 Serial.print(',');
162 }
163 Serial.println(" END");
164
165 // wait a little bit before the next scan
166 delay(100);
167
168 // set the next frequency
169 // the frequency step should be slightly smaller
170 // or the same as the Rx bandwidth set in setup
171 freqStart += 0.2;
172
173 // yield(); //pet the watchdog so it stays calm
174 }
175 Serial.println("SCAN END");
176
177 sx126x_spectral_finish_scan();
178}
