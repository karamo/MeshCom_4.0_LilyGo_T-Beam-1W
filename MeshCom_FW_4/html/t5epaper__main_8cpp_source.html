476static bool bq25896_init(void)
477{
478 bool result = PPM.init(Wire, BOARD_SDA, BOARD_SCL, BQ25896_SLAVE_ADDRESS);
479 if (result == false) {
480 // while (1) {
481 // Serial.println("PPM is not online...");
482 // delay(1000);
483 // }
484 return false;
485 }
486
487 // Set the minimum operating voltage. Below this voltage, the PPM will protect
488 PPM.setSysPowerDownVoltage(3300);
489
490 // Set input current limit, default is 500mA
491 PPM.setInputCurrentLimit(3250);
492
493 Serial.printf("getInputCurrentLimit: %d mA\n", PPM.getInputCurrentLimit());
494
495 // Disable current limit pin
496 PPM.disableCurrentLimitPin();
497
498 // Set the charging target voltage, Range:3840 ~ 4608mV ,step:16 mV
499 PPM.setChargeTargetVoltage(4208);
500
501 // Set the precharge current , Range: 64mA ~ 1024mA ,step:64mA
502 PPM.setPrechargeCurr(64);
503
504 // The premise is that Limit Pin is disabled, or it will only follow the maximum charging current set by Limi tPin.
505 // Set the charging current , Range:0~5056mA ,step:64mA
506 PPM.setChargerConstantCurr(1024);
507
508 // Get the set charging current
509 PPM.getChargerConstantCurr();
510 Serial.printf("getChargerConstantCurr: %d mA\n", PPM.getChargerConstantCurr());
511
512
513 // To obtain voltage data, the ADC must be enabled first
514 PPM.enableMeasure();
515
516 // Turn on charging function
517 // If there is no battery connected, do not turn on the charging function
518 PPM.enableCharge();
519
520 // Turn off charging function
521 // If USB is used as the only power input, it is best to turn off the charging function,
522 // otherwise the VSYS power supply will have a sawtooth wave, affecting the discharge output capability.
523 // PPM.disableCharge();
524
525
526 // The OTG function needs to enable OTG, and set the OTG control pin to HIGH
527 // After OTG is enabled, if an external power supply is plugged in, OTG will be turned off
528
529 PPM.enableOTG();
530 PPM.disableOTG();
531 // pinMode(OTG_ENABLE_PIN, OUTPUT);
532 // digitalWrite(OTG_ENABLE_PIN, HIGH);
533
534 return result;
535}
